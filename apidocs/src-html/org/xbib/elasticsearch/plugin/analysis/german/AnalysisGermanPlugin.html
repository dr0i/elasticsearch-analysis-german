001package org.xbib.elasticsearch.plugin.analysis.german;
002
003import org.elasticsearch.common.collect.ImmutableList;
004import org.elasticsearch.common.component.LifecycleComponent;
005import org.elasticsearch.common.inject.Module;
006import org.elasticsearch.index.analysis.AnalysisModule;
007import org.elasticsearch.plugins.AbstractPlugin;
008import org.xbib.elasticsearch.index.analysis.baseform.BaseformTokenFilterFactory;
009import org.xbib.elasticsearch.index.analysis.combo.ComboAnalysisBinderProcessor;
010import org.xbib.elasticsearch.index.analysis.concat.ConcatTokenFilterFactory;
011import org.xbib.elasticsearch.index.analysis.decompound.DecompoundTokenFilterFactory;
012import org.xbib.elasticsearch.index.analysis.german.GermanNormalizationFilterFactory;
013import org.xbib.elasticsearch.index.analysis.icu.IcuAnalysisBinderProcessor;
014import org.xbib.elasticsearch.index.analysis.langdetect.LangdetectModule;
015import org.xbib.elasticsearch.index.analysis.langdetect.LangdetectService;
016import org.xbib.elasticsearch.index.analysis.sortform.SortformAnalyzerProvider;
017import org.xbib.elasticsearch.index.analysis.sortform.SortformTokenFilterFactory;
018import org.xbib.elasticsearch.index.analysis.worddelimiter.WordDelimiterFilter2Factory;
019import org.xbib.elasticsearch.index.analysis.worddelimiter.WordDelimiterFilterFactory;
020import org.xbib.elasticsearch.indices.analysis.icu.IcuIndicesAnalysisModule;
021
022import java.util.Collection;
023
024import static org.elasticsearch.common.collect.Lists.newArrayList;
025
026public class AnalysisGermanPlugin extends AbstractPlugin {
027
028    @Override
029    public String name() {
030        return "analysis-german-" +
031                Build.getInstance().getVersion() + "-" +
032                Build.getInstance().getShortHash();
033    }
034
035    @Override
036    public String description() {
037        return "German language related analysis support";
038    }
039
040    @Override
041    public Collection<Class<? extends Module>> modules() {
042        return ImmutableList.<Class<? extends Module>>of(IcuIndicesAnalysisModule.class);
043    }
044
045    /**
046     * Automatically called with the analysis module.
047     */
048    public void onModule(AnalysisModule module) {
049        module.addProcessor(new IcuAnalysisBinderProcessor());
050        module.addProcessor(new ComboAnalysisBinderProcessor());
051        module.addAnalyzer("sortform", SortformAnalyzerProvider.class);
052        module.addTokenFilter("german_normalize", GermanNormalizationFilterFactory.class);
053        module.addTokenFilter("decompound", DecompoundTokenFilterFactory.class);
054        module.addTokenFilter("baseform", BaseformTokenFilterFactory.class);
055        module.addTokenFilter("worddelimiter", WordDelimiterFilterFactory.class);
056        module.addTokenFilter("worddelimiter2", WordDelimiterFilter2Factory.class);
057        module.addTokenFilter("sortform", SortformTokenFilterFactory.class);
058        module.addTokenFilter("concat", ConcatTokenFilterFactory.class);
059    }
060
061    @Override
062    public Collection<Class<? extends LifecycleComponent>> services() {
063        Collection<Class<? extends LifecycleComponent>> services = newArrayList();
064        services.add(LangdetectService.class);
065        return services;
066    }
067
068    @Override
069    public Collection<Class<? extends Module>> indexModules() {
070        Collection<Class<? extends Module>> modules = newArrayList();
071        modules.add(LangdetectModule.class);
072        return modules;
073    }
074
075}
