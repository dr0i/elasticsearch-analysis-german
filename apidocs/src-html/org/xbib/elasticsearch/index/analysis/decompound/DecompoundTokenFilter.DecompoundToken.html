002package org.xbib.elasticsearch.index.analysis.decompound;
003
004import org.apache.lucene.analysis.TokenFilter;
005import org.apache.lucene.analysis.TokenStream;
006import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
007import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
008import org.apache.lucene.analysis.tokenattributes.PositionIncrementAttribute;
009import org.apache.lucene.util.AttributeSource;
010
011import java.io.IOException;
012import java.util.LinkedList;
013
014public class DecompoundTokenFilter extends TokenFilter {
015
016    protected final LinkedList<DecompoundToken> tokens;
017
018    protected final Decompounder decomp;
019
020    protected final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
021
022    protected final OffsetAttribute offsetAtt = addAttribute(OffsetAttribute.class);
023
024    private final PositionIncrementAttribute posIncAtt = addAttribute(PositionIncrementAttribute.class);
025
026    private AttributeSource.State current;
027
028    protected DecompoundTokenFilter(TokenStream input, Decompounder decomp) {
029        super(input);
030        this.tokens = new LinkedList<DecompoundToken>();
031        this.decomp = decomp;
032    }
033
034    @Override
035    public final boolean incrementToken() throws IOException {
036        if (!tokens.isEmpty()) {
037            assert current != null;
038            DecompoundToken token = tokens.removeFirst();
039            restoreState(current);
040            termAtt.setEmpty().append(token.txt);
041            offsetAtt.setOffset(token.startOffset, token.endOffset);
042            posIncAtt.setPositionIncrement(0);
043            return true;
044        }
045        if (input.incrementToken()) {
046            decompound();
047            if (!tokens.isEmpty()) {
048                current = captureState();
049            }
050            return true;
051        } else {
052            return false;
053        }
054    }
055
056    protected void decompound() {
057        int start = offsetAtt.startOffset();
058        CharSequence term = new String(termAtt.buffer(), 0, termAtt.length());
059        for (String s : decomp.decompound(term.toString())) {
060            int len = s.length();
061            tokens.add(new DecompoundToken(s, start, len));
062            start += len;
063        }
064    }
065
066    @Override
067    public void reset() throws IOException {
068        super.reset();
069        tokens.clear();
070        current = null;
071    }
072
073    protected class DecompoundToken {
074
075        public final CharSequence txt;
076        public final int startOffset;
077        public final int endOffset;
078
079        public DecompoundToken(CharSequence txt, int offset, int length) {
080            this.txt = txt;
081            int startOff = DecompoundTokenFilter.this.offsetAtt.startOffset();
082            int endOff = DecompoundTokenFilter.this.offsetAtt.endOffset();
083            if (endOff - startOff != DecompoundTokenFilter.this.termAtt.length()) {
084                this.startOffset = startOff;
085                this.endOffset = endOff;
086            } else {
087                this.startOffset = offset;
088                this.endOffset = offset + length;
089            }
090        }
091    }
092}
