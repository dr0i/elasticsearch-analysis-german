226    private String reverse(String s) {
227        char[] ret = new char[s.length()];
228        StringBuilder torev = new StringBuilder(s);
229        for (int i = torev.length() - 1, j = 0; i >= 0; i--, j++) {
230            ret[j] = torev.charAt(i);
231        }
232        return new String(ret);
233    }

235    private String voted(List<String> classes) {
236        if (classes == null) {
237            return null;
238        }
239        int sum = 0;
240        int maxval = 0, actval;
241        String maxclass = "undecided";
242        String actclass;
243        for (String cl : classes) {
244            StringTokenizer st = new StringTokenizer(cl, "=");
245            actclass = st.nextToken();
246            if (st.hasMoreTokens()) {
247                actval = new Integer(st.nextToken());
248            } else {
249                actval = 0;
250            }
251            sum += actval;
252            if (actval > maxval) {
253                maxval = actval;
254                maxclass = actclass;
255            }
256            if ((actval == maxval) && !actclass.equals(maxclass) && !actclass.isEmpty()) {
257                maxclass += ";" + actclass;
258            }
259        }
260        if (((double) maxval / (double) sum) >= this.thresh) {
261            return maxclass;
262        } else {
263            return "undecided";
264        }
265    }

267    private List<String> add(List<String> one, List<String> two) {
268        List list = new ArrayList();
269        hash = new HashMap();
270        String clas;
271        String snr;
272        int nr;
273        int nr2;
274        String cont;
275        for (String s : one) {
276            StringTokenizer st = new StringTokenizer(s, "=");
277            clas = st.nextToken();
278            snr = st.nextToken();
279            hash.put(clas, snr);
280        }
281        for (String s : two) {
282            StringTokenizer st = new StringTokenizer(s, "=");
283            clas = st.nextToken();
284            if (st.hasMoreTokens()) {
285                snr = st.nextToken();
286                nr = Integer.parseInt(snr);
287            } else {
288                nr = 0;
289            }
290            cont = (String) hash.get(clas);
291            if (cont != null) {
292                nr2 = Integer.parseInt(cont);
293                nr += nr2;
294            }
295            snr = Integer.toString(nr);
296            hash.put(clas, snr);
297        }
298
299        for (String c : hash.keySet()) {
300            String instr = c + "=" + hash.get(c);
301            list.add(instr);
302        }
303        return list;
304    }

306    private Node insert(Node k1, Node k2) {
307        String w0;
308        String w1;
309        String w2;
310        if (k1 == null) {
311            return k2;
312        }
313        int pos;
314        int min;
315
316        if (k1.content.length() < k2.content.length()) {
317            min = k1.content.length();
318        } else {
319            min = k2.content.length();
320        }
321        for (pos = 0; pos < min; pos++) {
322            if (k1.content.charAt(pos) != k2.content.charAt(pos)) {
323                break;
324            }
325        }
326        w0 = k2.content.substring(0, pos);
327        w1 = k1.content.substring(pos, k1.content.length());
328        w2 = k2.content.substring(pos, k2.content.length());
329        if (w2.length() == 0) {
330            k1.classes(add(k1.classes(), k2.classes()));
331            return k1;
332        }
333        if (w1.length() == 0) {
334            k2.content = w2;
335            Node goalpos = getChild(k1, w2);
336            if (goalpos == null) {
337                k1.children().add(k2);
338            } else {
339                k1.children().remove(goalpos);
340                k1.children().add(insert(goalpos, k2));
341            }
342            k1.classes(add(k1.classes(), k2.classes()));
343            return k1;
344        } else {
345            Node h = new Node(w0);
346            k2.content = w2;
347            h.children().add(k2);
348            k1.content = w1;
349            h.children().add(k1);
350            h.classes(add(k1.classes(), k2.classes()));
351            return h;
352        }
353    }

364    private Node lookup(Node k, String w) {
365        Node node;
366        String w1;
367        String w2;
368        if (k == null) {
369            return null;
370        }
371        int min;
372        int pos;
373        if (k.content.length() < w.length()) {
374            min = k.content.length();
375        } else {
376            min = w.length();
377        }
378        for (pos = 0; pos < min; pos++) {
379            if (k.content.charAt(pos) != w.charAt(pos)) {
380                break;
381            }
382        }
383        w1 = k.content.substring(pos, k.content.length());
384        w2 = w.substring(pos, w.length());
385        if (w2.length() != 0) {
386            node = lookup(getChild(k, w2), w2);
387            if (node != null) {
388                return node;
389            } else {
390                return k;
391            }
392        }
393        if (w1.length() != 0) {
394            return k;
395        }
396        return k;
397    }

407    private Node pruneNode(Node node) {
408        String vklass;
409        String aklass;
410        StringTokenizer st;
411        ArrayList temp;
412        if (node.children().size() == 0) {
413            node.content = node.content.substring(0, 1);
414        } else if (node.classes().size() == 1) {
415            node.content = node.content.substring(0, 1);
416            node.children().clear();
417        } else {
418            vklass = voted(node.classes());
419            temp = new ArrayList();
420            for (Node akk : node.children()) {
421                if (akk.classes().size() == 1) {
422                    aklass = akk.classes().get(0);
423                    st = new StringTokenizer(aklass, "=");
424                    aklass = st.nextToken();
425                    if (aklass.equals(vklass)) {
426                    } else {
427                        akk = pruneNode(akk);
428                        temp.add(akk);
429                    }
430                } else {
431                    akk = pruneNode(akk);
432                    temp.add(akk);
433                }
434            }
435            node.children(temp);
436        }
437        return node;
438    }

466        }
467        if (ignorecase) {
468            word = word.toLowerCase();
469        }
470        if (reverse) {
471            word = reverse(word);
472        }
473        Node k = find(word + "<");
474        double valsum = 0;
475        double goalval = 0;
476        String actclass;
477        int actval;
478        for (String s : k.classes()) {
479            StringTokenizer st = new StringTokenizer(s, "=");
480            actclass = st.nextToken();
481            actval = new Integer(st.nextToken()).intValue();
482            valsum += actval;
483            if (actclass.equals(cla)) {
484                goalval = actval;
485            }
486        }
487        if (valsum > 0) {
488            ret = goalval / valsum;
489        }
490        return ret;
491    }

525    private Node get(String word, int mode) {
526        String currentWord = word;
527        int i = 0;
528        StringBuilder currentLabel;
529        String exlabel = "";
530        List currentClasses;
531        while (stringtree[i] != attentionNode) {
532            exlabel += stringtree[i];
533            i++;
534        }
535        while (true) {
536            currentClasses = new ArrayList();
537            currentLabel = new StringBuilder();
538            i++;
539            i++;
540            while (stringtree[i] != ']') {
541                StringBuilder currentClass = new StringBuilder();
542                while ((stringtree[i] != ';') && (stringtree[i] != ']')) {
543                    currentClass.append(stringtree[i]);
544                    i++;
545                }
546                if (stringtree[i] != ']') {
547                    i++;
548                }
549                currentClasses.add(currentClass.toString());
550            }
551            if (currentWord.length() == 0) {
552                break;
553            }
554            if ((i + 1) == stringtree.length) {
555                if (mode == EXACT) {
556                    exlabel = null;
557                    currentClasses = null;
558                }
559                break;
560            }
561            i++;
562            while (stringtree[i] != currentWord.charAt(0)) {
563                if (stringtree[i] == attentionNode) {
564                    if (mode == EXACT) {
565                        exlabel = null;
566                        currentClasses = null;
567                    }
568                    break;
569                }
570                while (stringtree[i] != attentionNumber) {
571                    i++;
572                }
573                i++;
574                i += offset;
575            }
576            if (stringtree[i] == attentionNode) {
577                break;
578            }
579            while (stringtree[i] != attentionNumber) {
580                currentLabel.append(stringtree[i]);
581                i++;
582            }
583            i++;
584            if (currentLabel.length() > currentWord.length()) {
585                if (mode == EXACT) {
586                    exlabel = null;
587                    currentClasses = null;
588                } else if (mode == LOWER) {
589                    exlabel = currentLabel.toString();
590                    currentClasses = getClassesAt(string2int(new String(stringtree, i, offset)));
591                }
592                break;
593            }
594            String w1 = currentWord.substring(0, currentLabel.length());
595            String w2 = currentWord.substring(currentLabel.length());
596            if (!(w1.equals(currentLabel.toString()))) {
597                if (mode == EXACT) {
598                    exlabel = null;
599                    currentClasses = null;
600                } else if (mode == LOWER) {
601                    exlabel = currentLabel.toString();
602                    currentClasses = getClassesAt(string2int(new String(stringtree, i, offset)));
603                }
604                break;
605            }
606            int o = string2int(new String(stringtree, i, offset));
607            currentWord = w2;
608            i = o;
609            exlabel = currentLabel.toString();
610        }
611        Node k = new Node(exlabel);
612        k.classes(currentClasses);
613        return k;
614    }

635    private void insert(Node k) {
636        if (root == null) {
637            root = getObjectTree(stringtree);
638        }
639        stringtree = null;
640        root.classes(add(root.classes(), k.classes()));
641        Node gpos = getChild(root, k.content);
642        if (gpos == null) {
643            gpos = k;
644            root.children().add(gpos);
645        } else {
646            root.children().remove(gpos);
647            root.children().add(insert(gpos, k));
648        }
649    }

687        if (i < treestring.length) {
688            while (treestring[i] != this.attentionNode) {

697                    childOffset.append(treestring[i]);
698                    i++;
699                }
700                addStringToMap(m, treestring, string2int(childOffset.toString()), childContent);
701            }
702        }

939    private StringBuilder tree2string(Node aktKnoten, int startPos) {
940        StringBuilder ret = new StringBuilder();
941        ret.append(this.attentionNode);
942        int relPos = 1;
943        StringBuilder tmp;
944        tmp = formatClasses(aktKnoten);
945        relPos += tmp.length();
946        ret.append(tmp);
947        tmp = formatChildrenContents(aktKnoten, relPos);
948        relPos += tmp.length();
949        ret.append(tmp);
950        int vorigerTeilbaum = 0;
951        for (Node aktKind : aktKnoten.children()) {
952            relPos += vorigerTeilbaum;
953            formatPosition(ret, aktKind.pos, relPos + startPos);
954            tmp = tree2string(aktKind, relPos + startPos);
955            vorigerTeilbaum = tmp.length();
956            ret.append(tmp);
957        }
958        return ret;
959    }

961    private StringBuilder formatClasses(Node node) {
962        StringBuilder ret = new StringBuilder("[");
963        int k = 0;
964        for (String s : node.classes()) {
965            k++;
966            if (k != 1) {
967                ret.append(";");
968            }
969            ret.append(s);
970        }
971        ret.append("]");
972        return ret;
973    }

976        StringBuilder ret = new StringBuilder("");

979            ret.append(child.content);
980            ret.append(this.attentionNumber);

982            child.pos = relPos + startPos;
983            ret.append(int2string(0));
984        }

992    private String int2string(int i) {
993        StringBuilder ret = new StringBuilder();
994        int rest = i;
995        for (int e = this.offset - 1; e >= 0; e--) {
996            int k = rest / ((int) (Math.exp(e * Math.log(this.basis))));
997            rest = rest % ((int) (Math.exp(e * Math.log(this.basis))));
998            char c = (char) (this.startchar + k);
999            ret.append(c);
1000        }
1001        return ret.toString();
1002    }

1004    private int string2int(String s) {
1005        int ret = 0;
1006        for (int i = 0; i < this.offset; i++) {
1007            char c = s.charAt(i);
1008            int k = (c) - this.startchar;
1009            ret += k
1010                    * ((int) Math.exp((this.offset - i - 1)
1011                    * Math.log(this.basis)));
1012        }
1013        return ret;
1014    }
