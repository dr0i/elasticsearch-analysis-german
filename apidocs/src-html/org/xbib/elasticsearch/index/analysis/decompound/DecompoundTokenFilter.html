001package org.xbib.elasticsearch.index.analysis.decompound;
002
003import org.apache.lucene.analysis.TokenFilter;
004import org.apache.lucene.analysis.TokenStream;
005import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
006import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
007import org.apache.lucene.analysis.tokenattributes.PositionIncrementAttribute;
008import org.apache.lucene.util.AttributeSource;
009
010import java.io.IOException;
011import java.util.LinkedList;
012
013public class DecompoundTokenFilter extends TokenFilter {
014
015    protected final LinkedList<DecompoundToken> tokens;
016
017    protected final Decompounder decomp;
018
019    protected final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
020
021    protected final OffsetAttribute offsetAtt = addAttribute(OffsetAttribute.class);
022
023    private final PositionIncrementAttribute posIncAtt = addAttribute(PositionIncrementAttribute.class);
024
025    private AttributeSource.State current;
026
027    protected DecompoundTokenFilter(TokenStream input, Decompounder decomp) {
028        super(input);
029        this.tokens = new LinkedList<DecompoundToken>();
030        this.decomp = decomp;
031    }
032
033    @Override
034    public final boolean incrementToken() throws IOException {
035        if (!tokens.isEmpty()) {
036            assert current != null;
037            DecompoundToken token = tokens.removeFirst();
038            restoreState(current);
039            termAtt.setEmpty().append(token.txt);
040            offsetAtt.setOffset(token.startOffset, token.endOffset);
041            posIncAtt.setPositionIncrement(0);
042            return true;
043        }
044        if (input.incrementToken()) {
045            decompound();
046            if (!tokens.isEmpty()) {
047                current = captureState();
048            }
049            return true;
050        } else {
051            return false;
052        }
053    }
054
055    protected void decompound() {
056        int start = offsetAtt.startOffset();
057        CharSequence term = new String(termAtt.buffer(), 0, termAtt.length());
058        for (String s : decomp.decompound(term.toString())) {
059            int len = s.length();
060            tokens.add(new DecompoundToken(s, start, len));
061            start += len;
062        }
063    }
064
065    @Override
066    public void reset() throws IOException {
067        super.reset();
068        tokens.clear();
069        current = null;
070    }
071
072    protected class DecompoundToken {
073
074        public final CharSequence txt;
075        public final int startOffset;
076        public final int endOffset;
077
078        public DecompoundToken(CharSequence txt, int offset, int length) {
079            this.txt = txt;
080            int startOff = DecompoundTokenFilter.this.offsetAtt.startOffset();
081            int endOff = DecompoundTokenFilter.this.offsetAtt.endOffset();
082            if (endOff - startOff != DecompoundTokenFilter.this.termAtt.length()) {
083                this.startOffset = startOff;
084                this.endOffset = endOff;
085            } else {
086                this.startOffset = offset;
087                this.endOffset = offset + length;
088            }
089        }
090    }
091}
