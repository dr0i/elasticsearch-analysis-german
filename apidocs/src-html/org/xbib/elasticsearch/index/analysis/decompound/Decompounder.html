002package org.xbib.elasticsearch.index.analysis.decompound;
003
004import java.io.IOException;
005import java.io.InputStream;
006import java.util.ArrayList;
007import java.util.List;
008import java.util.StringTokenizer;
009
010public class Decompounder {
011
012    private CompactPatriciaTrie kompvvTree;
013    private CompactPatriciaTrie kompvhTree;
014    private CompactPatriciaTrie grfTree;
015
016    public Decompounder(InputStream kompvv, InputStream kompvh, InputStream gfred, double threshold)
017            throws IOException, ClassNotFoundException {
018        kompvvTree = new CompactPatriciaTrie();
019        kompvvTree.load(kompvv);
020        kompvvTree.setIgnoreCase(true);
021        kompvvTree.setThreshold(threshold);
022        kompvhTree = new CompactPatriciaTrie();
023        kompvhTree.load(kompvh);
024        kompvhTree.setIgnoreCase(true);
025        kompvhTree.setThreshold(threshold);
026        grfTree = new CompactPatriciaTrie();
027        grfTree.load(gfred);
028        grfTree.setIgnoreCase(true);
029        grfTree.setThreshold(threshold); // previous value = 0.46
030    }
031
032    public Decompounder(CompactPatriciaTrie kompvv, CompactPatriciaTrie kompvh, CompactPatriciaTrie gfred) {
033        kompvvTree = kompvv;
034        kompvhTree = kompvh;
035        grfTree = gfred;
036    }
037    
038    private String reverse(String torev) {
039        String ret = "";
040        for (int i = torev.length(); i > 0; i--) {
041            ret += torev.substring(i - 1, i);
042        }
043        return ret;
044    }
045
046    public List<String> decompound(String word) {
047        word = reduceToBaseForm(word);
048        List<String> list = new ArrayList<String>();
049        String classvv = kompvvTree.classify(word + "<");
050        String classvh = kompvhTree.classify(reverse(word) + "<");
051        StringBuilder numStrvv = new StringBuilder();
052        StringBuilder numStrvh = new StringBuilder();
053        StringBuilder suffixvv = new StringBuilder();
054        StringBuilder suffixvh = new StringBuilder();
055        String vvpart1 = "";
056        String vhpart1 = "";
057        String vvpart2 = "";
058        String vhpart2 = "";
059        int numvv = 0;
060        int numvh = 0;
061        boolean vhOk = true;
062        boolean vvOk = true;
063        if (classvv.equals("undecided")) {
064            vvOk = false;
065        }
066        if (classvh.equals("undecided")) {
067            vhOk = false;
068        }
069        if (vvOk) {
070            for (int i = 0; i < classvv.length(); i++) {
071                char c = classvv.charAt(i);
072                if ((c <= '9') && (c >= '0')) {
073                    numStrvv.append(c);
074                } else {
075                    suffixvv.append(c);
076                }
077            }
078        }
079        if (vhOk) {
080            for (int i = 0; i < classvh.length(); i++) {
081                char c = classvh.charAt(i);
082                if ((c <= '9') && (c >= '0')) {
083                    numStrvh.append(c);
084                } else {
085                    suffixvh.append(c);
086                }
087            }
088        }
089
090        if (vvOk) {
091            numvv = Integer.parseInt(numStrvv.toString());
092        }
093        if (vhOk) {
094            numvh = Integer.parseInt(numStrvh.toString());
095        }
096
097        if (vvOk) {
098            if (numvv >= word.length()) {
099                vvOk = false;
100            }
101        }
102
103        if (vhOk) {
104            if (numvh >= word.length()) {
105                vhOk = false;
106            }
107        }
108
109        if (vvOk) {
110            for (int i = 0; i < suffixvv.length(); i++) {
111                if (word.length() > (numvv + i)) {
112                    if (suffixvv.charAt(i) != word.charAt(numvv + i)) {
113                        vvOk = false;
114                    }
115                } else {
116                    vvOk = false;
117                }
118            }
119        }
120        if (vhOk) {
121            for (int i = 0; i < suffixvh.length(); i++) {
122                if (suffixvh.charAt(i) != word.charAt(numvh + 1 + i)) {
123                    vvOk = false;
124                }
125            }
126        }
127        if (vvOk) {
128            vvpart1 = word.substring(0, numvv);
129            vvpart2 = word.substring(numvv + suffixvv.length(), word.length());
130            if (vvpart2.length() <= 3) {
131                vvOk = false;
132            }
133        }
134        if (vhOk) {
135            vhpart1 = word.substring(0, word.length() - numvh);
136            vhpart2 = word.substring(word.length() - (numvh + suffixvh.length()), word.length());
137            if (vhpart1.length() <= 3) {
138                vhOk = false;
139            }
140
141        }
142        if (vvOk && vhOk) {
143            if (vvpart1.equals(vhpart1)) {
144                list.add(vvpart1);
145                if (vhpart2.length() < vvpart2.length()) {
146                    list.add(vhpart2);
147                } else if (vhpart2.length() > vvpart2.length()) {
148                    list.add(vvpart2);
149                }
150            } else if ((vhpart1.length() - vvpart1.length()) < 3) {
151                list.add(vvpart1);
152                if (vhpart2.length() < vvpart2.length()) {
153                    list.add(vhpart2);
154                } else if (vhpart2.length() > vvpart2.length()) {
155                    list.add(vvpart2);
156                }
157            }
158            else {
159                list.add(vvpart1);
160                list.add(word.substring(vvpart1.length() + suffixvv.length(), word.length() - numvh));
161                list.add(vhpart2);
162            }
163            if (vvpart2.equals(vhpart2)) {
164                list.add(vvpart2);
165            }
166
167        } else if (vvOk && !vhOk) {
168            list.add(vvpart1);
169            list.add(vvpart2);
170        } else if (vhOk && !vvOk) {
171            list.add(vhpart1);
172            list.add(vhpart2);
173        } else {
174            list.add(word);
175        }
176        List<String> retvec2 = new ArrayList();
177        List<String> l;
178        if (list.size() > 1) {
179            for (String s : list) {
180                l = decompound(s);
181                retvec2.addAll(l);
182            }
183        }
184        else {
185            retvec2 = list;
186        }
187        return retvec2;
188    }
189
190    public String reduceToBaseForm(String word) {
191        String result = word;
192        String baseForm = grfTree.classify(reverse(word));
193        if (!baseForm.equals("undecided")) {
194            StringTokenizer st = new StringTokenizer(baseForm, ",");
195            baseForm = st.nextToken();
196            StringBuilder numStr = new StringBuilder();
197            StringBuilder suffix = new StringBuilder();
198            for (int i = 0; i < baseForm.length(); i++) {
199                char c = baseForm.charAt(i);
200                if ((c <= '9') && (c >= '0')) {
201                    numStr.append(c);
202                } else {
203                    suffix.append(c);
204                }
205            }
206            if (numStr.length() > 0) {
207                int cutpos = Integer.parseInt(numStr.toString());
208                if (cutpos > result.length()) {
209                    cutpos = result.length();
210                }
211                result = result.substring(0, result.length() - cutpos) + suffix;
212            }
213        }
214        return result;
215    }
