002package org.xbib.elasticsearch.index.analysis.decompound;
003
004import org.apache.lucene.analysis.TokenStream;
005import org.elasticsearch.ElasticsearchIllegalArgumentException;
006import org.elasticsearch.common.inject.Inject;
007import org.elasticsearch.common.inject.assistedinject.Assisted;
008import org.elasticsearch.common.settings.Settings;
009import org.elasticsearch.env.Environment;
010import org.elasticsearch.index.Index;
011import org.elasticsearch.index.analysis.AbstractTokenFilterFactory;
012import org.elasticsearch.index.settings.IndexSettings;
013
014import java.io.IOException;
015
016public class DecompoundTokenFilterFactory extends AbstractTokenFilterFactory {
017
018    private final Decompounder decompounder;
019
020    @Inject
021    public DecompoundTokenFilterFactory(Index index,
022            @IndexSettings Settings indexSettings, Environment env,
023            @Assisted String name, @Assisted Settings settings) {
024        super(index, indexSettings, name, settings);
025        this.decompounder = createDecompounder(env, settings);
026    }
027
028    @Override
029    public TokenStream create(TokenStream tokenStream) {
030        return new DecompoundTokenFilter(tokenStream, decompounder);
031    }
032
033    private Decompounder createDecompounder(Environment env, Settings settings) {
034        try {
035            String forward = settings.get("forward", "/decompound/kompVVic.tree");
036            String backward = settings.get("backward", "/decompound/kompVHic.tree");
037            String reduce = settings.get("reduce", "/decompound/grfExt.tree");
038            double threshold = settings.getAsDouble("threshold", 0.51);
039            return new Decompounder(env.resolveConfig(forward).openStream(),
040                    env.resolveConfig(backward).openStream(),
041                    env.resolveConfig(reduce).openStream(),
042                    threshold);
043        } catch (ClassNotFoundException e) {
044            throw new ElasticsearchIllegalArgumentException("decompounder resources in settings not found: " + settings, e);
045        } catch (IOException e) {
046            throw new ElasticsearchIllegalArgumentException("decompounder resources in settings not found: " + settings, e);
047        }
048    }
049}
