001package org.xbib.elasticsearch.index.analysis.sortform;
002
003import org.apache.lucene.analysis.TokenFilter;
004import org.apache.lucene.analysis.TokenStream;
005import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
006
007import java.io.IOException;
008import java.util.List;
009import java.util.regex.Pattern;
010
011public class SortformTokenFilter extends TokenFilter {
012
013    protected final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
014
015    private final List<Pattern> patterns;
016
017    protected SortformTokenFilter(TokenStream input, List<Pattern> patterns) {
018        super(input);
019        this.patterns = patterns;
020    }
021
022    @Override
023    public final boolean incrementToken() throws IOException {
024        if (!input.incrementToken()) {
025            return false;
026        } else {
027            String s = termAtt.toString();
028            for (Pattern pattern : patterns) {
029                s = pattern.matcher(s).replaceAll("");
030            }
031            termAtt.setEmpty().append(s);
032            return true;
033        }
034    }
035}
