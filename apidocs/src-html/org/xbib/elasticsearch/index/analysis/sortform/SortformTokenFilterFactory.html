001package org.xbib.elasticsearch.index.analysis.sortform;
002
003import org.apache.lucene.analysis.TokenStream;
004import org.elasticsearch.common.inject.Inject;
005import org.elasticsearch.common.inject.assistedinject.Assisted;
006import org.elasticsearch.common.settings.Settings;
007import org.elasticsearch.index.Index;
008import org.elasticsearch.index.analysis.AbstractTokenFilterFactory;
009import org.elasticsearch.index.settings.IndexSettings;
010
011import java.util.List;
012import java.util.regex.Pattern;
013
014import static org.elasticsearch.common.collect.Lists.newLinkedList;
015
016public class SortformTokenFilterFactory extends AbstractTokenFilterFactory {
017
018    private final List<Pattern> patterns;
019
020    @Inject
021    public SortformTokenFilterFactory(Index index,
022                                      @IndexSettings Settings indexSettings,
023                                      @Assisted String name, @Assisted Settings settings) {
024        super(index, indexSettings, name, settings);
025        patterns = newLinkedList();
026        String[] patternStrs = settings.getAsArray("pattern");
027        if (patternStrs == null || patternStrs.length == 0) {
028            // default patterns
029            patternStrs = new String[]{
030                    "\\s*<<.*?>>\\s*",
031                    "\\s*<.*?>\\s*",
032                    "\\s*\u0098.*?\u009C\\s*",
033                    "\\s*\u02BE.*?\u02BB\\s*",
034                    "\\s*\u00AC.*?\u00AC\\s*"
035            };
036        }
037        for (String patternStr : patternStrs) {
038            patterns.add(Pattern.compile(patternStr));
039        }
040    }
041
042    @Override
043    public TokenStream create(TokenStream tokenStream) {
044        return new SortformTokenFilter(tokenStream, patterns);
045    }
046
047}
