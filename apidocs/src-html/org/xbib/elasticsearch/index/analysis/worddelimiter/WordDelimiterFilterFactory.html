039public class WordDelimiterFilterFactory extends AbstractTokenFilterFactory {
040
041    private Set<String> protectedWords = null;
042    private int flags;
043    private byte[] typeTable = null;
044
045    @Inject
046    public WordDelimiterFilterFactory(Index index, @IndexSettings Settings indexSettings,
047                                      Environment env,
048                                      @Assisted String name,
049                                      @Assisted Settings settings) {
050        super(index, indexSettings, name, settings);
051
052        // Sample Format for the type table:
053        // $ => DIGIT
054        // % => DIGIT
055        // . => DIGIT
056        // \u002C => DIGIT
057        // \u200D => ALPHANUM
058        List<String> charTypeTableValues = Analysis.getWordList(env, settings, "type_table");
059        if (charTypeTableValues == null) {
060            this.typeTable = WordDelimiterIterator.DEFAULT_WORD_DELIM_TABLE;
061        } else {
062            this.typeTable = parseTypes(charTypeTableValues);
063        }
064
065        // If 1, causes parts of words to be generated: "PowerShot" => "Power" "Shot"
066        flags |= getFlag(GENERATE_WORD_PARTS, settings, "generate_word_parts", true);
067        // If 1, causes number subwords to be generated: "500-42" => "500" "42"
068        flags |= getFlag(GENERATE_NUMBER_PARTS, settings, "generate_number_parts", true);
069        // If 1, causes maximum runs of word parts to be catenated: "wi-fi" => "wifi"
070        flags |= getFlag(CATENATE_WORDS, settings, "catenate_words", false);
071        // If 1, causes maximum runs of number parts to be catenated: "500-42" => "50042"
072        flags |= getFlag(CATENATE_NUMBERS, settings, "catenate_numbers", false);
073        // If 1, causes all subword parts to be catenated: "wi-fi-4000" => "wifi4000"
074        flags |= getFlag(CATENATE_ALL, settings, "catenate_all", false);
075        // If 1, causes "PowerShot" to be two tokens; ("Power-Shot" remains two parts regards)
076        flags |= getFlag(SPLIT_ON_CASE_CHANGE, settings, "split_on_case_change", true);
077        // If 1, causes "j2se" to be three tokens; "j" "2" "se"
078        flags |= getFlag(SPLIT_ON_NUMERICS, settings, "split_on_numerics", true);
079        // If 1, includes original words in subwords: "500-42" => "500" "42" "500-42"
080        flags |= getFlag(PRESERVE_ORIGINAL, settings, "preserve_original", false);
081        // If 1, causes trailing "'s" to be removed for each subword: "O'Neil's" => "O", "Neil"
082        flags |= getFlag(STEM_ENGLISH_POSSESSIVE, settings, "stem_english_possessive", true);
083        // If not null is the set of tokens to protect from being delimited
084        List<String> protoWords = Analysis.getWordList(env, settings, "protected_words");
085        protectedWords = protoWords == null ? null : newHashSet(protoWords);
086    }
087
088    public WordDelimiterFilter2 create(TokenStream input) {
089        return new WordDelimiterFilter2(input, typeTable, flags, protectedWords);
090    }
091
092    public int getFlag(int flag, Settings settings, String key, boolean defaultValue) {
093        if (settings.getAsBoolean(key, defaultValue)) {
094            return flag;
095        }
096        return 0;
097    }
098
099    // source => type
100    private static Pattern typePattern = Pattern.compile("(.*)\\s*=>\\s*(.*)\\s*$");
101
102    // parses a list of MappingCharFilter style rules into a custom byte[] type table
103    private byte[] parseTypes(List<String> rules) {
104        SortedMap<Character, Byte> typeMap = newTreeMap();
105        for (String rule : rules) {
106            Matcher m = typePattern.matcher(rule);
107            if (!m.find()) {
108                throw new ElasticsearchIllegalArgumentException("invalid rule : [" + rule + "]");
109            }
110            String lhs = parseString(m.group(1).trim());
111            Byte rhs = parseType(m.group(2).trim());
112            if (lhs.length() != 1) {
113                throw new ElasticsearchIllegalArgumentException("invalid rule : [" + rule + "]: only single character allowed");
114            }
115            if (rhs == null) {
116                throw new ElasticsearchIllegalArgumentException("invalid rule : [" + rule + "]: type illegal");
117            }
118            typeMap.put(lhs.charAt(0), rhs);
119        }
120
121        // ensure the table is always at least as big as DEFAULT_WORD_DELIM_TABLE for performance
122        byte types[] = new byte[Math.max(typeMap.lastKey() + 1, WordDelimiterIterator.DEFAULT_WORD_DELIM_TABLE.length)];
123        for (int i = 0; i < types.length; i++) {
124            types[i] = WordDelimiterIterator.getType(i);
125        }
126        for (Map.Entry<Character, Byte> mapping : typeMap.entrySet()) {
127            types[mapping.getKey()] = mapping.getValue();
128        }
129        return types;
130    }
131
132    private Byte parseType(String s) {
133        if (s.equals("LOWER")) {
134            return LOWER;
135        } else if (s.equals("UPPER")) {
136            return UPPER;
137        } else if (s.equals("ALPHA")) {
138            return ALPHA;
139        } else if (s.equals("DIGIT")) {
140            return DIGIT;
141        } else if (s.equals("ALPHANUM")) {
142            return ALPHANUM;
143        } else if (s.equals("SUBWORD_DELIM")) {
144            return SUBWORD_DELIM;
145        } else {
146            return null;
147        }
148    }
149
150    private final char[] out = new char[256];
151
152    private String parseString(String s) {
153        int readPos = 0;
154        int len = s.length();
155        int writePos = 0;
156        while (readPos < len) {
157            char c = s.charAt(readPos++);
158            if (c == '\\') {
159                if (readPos >= len) {
160                    throw new ElasticsearchIllegalArgumentException("invalid escaped char in [" + s + "]");
161                }
162                c = s.charAt(readPos++);
163                switch (c) {
164                    case '\\':
165                        c = '\\';
166                        break;
167                    case 'n':
168                        c = '\n';
169                        break;
170                    case 't':
171                        c = '\t';
172                        break;
173                    case 'r':
174                        c = '\r';
175                        break;
176                    case 'b':
177                        c = '\b';
178                        break;
179                    case 'f':
180                        c = '\f';
181                        break;
182                    case 'u':
183                        if (readPos + 3 >= len) {
184                            throw new ElasticsearchIllegalArgumentException("invalid escaped char in [" + s + "]");
185                        }
186                        c = (char) Integer.parseInt(s.substring(readPos, readPos + 4), 16);
187                        readPos += 4;
188                        break;
189                }
190            }
191            out[writePos++] = c;
192        }
193        return new String(out, 0, writePos);
194    }
