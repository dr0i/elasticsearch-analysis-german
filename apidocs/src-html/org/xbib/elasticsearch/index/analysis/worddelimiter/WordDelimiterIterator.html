001package org.xbib.elasticsearch.index.analysis.worddelimiter;
002
003import static org.xbib.elasticsearch.index.analysis.worddelimiter.WordDelimiterFilter.ALPHA;
004import static org.xbib.elasticsearch.index.analysis.worddelimiter.WordDelimiterFilter.DIGIT;
005import static org.xbib.elasticsearch.index.analysis.worddelimiter.WordDelimiterFilter.LOWER;
006import static org.xbib.elasticsearch.index.analysis.worddelimiter.WordDelimiterFilter.SUBWORD_DELIM;
007import static org.xbib.elasticsearch.index.analysis.worddelimiter.WordDelimiterFilter.UPPER;
008import static org.xbib.elasticsearch.index.analysis.worddelimiter.WordDelimiterFilter.isAlpha;
009import static org.xbib.elasticsearch.index.analysis.worddelimiter.WordDelimiterFilter.isDigit;
010import static org.xbib.elasticsearch.index.analysis.worddelimiter.WordDelimiterFilter.isSubwordDelim;
011import static org.xbib.elasticsearch.index.analysis.worddelimiter.WordDelimiterFilter.isUpper;
012
013/**
014 * A BreakIterator-like API for iterating over subwords in text, according to WordDelimiterFilter rules.
015 */
016public final class WordDelimiterIterator {
017
018    /**
019     * Indicates the end of iteration
020     */
021    public static final int DONE = -1;
022
023    public static final byte[] DEFAULT_WORD_DELIM_TABLE;
024
025    char text[];
026    int length;
027
028    /**
029     * start position of text, excluding leading delimiters
030     */
031    int startBounds;
032    /**
033     * end position of text, excluding trailing delimiters
034     */
035    int endBounds;
036
037    /**
038     * Beginning of subword
039     */
040    int current;
041    /**
042     * End of subword
043     */
044    int end;
045
046    /* does this string end with a possessive such as 's */
047    private boolean hasFinalPossessive = false;
048
049    /**
050     * If false, causes case changes to be ignored (subwords will only be generated
051     * given SUBWORD_DELIM tokens). (Defaults to true)
052     */
053    final boolean splitOnCaseChange;
054
055    /**
056     * If false, causes numeric changes to be ignored (subwords will only be generated
057     * given SUBWORD_DELIM tokens). (Defaults to true)
058     */
059    final boolean splitOnNumerics;
060
061    /**
062     * If true, causes trailing "'s" to be removed for each subword. (Defaults to true)
063     * <p/>
064     * "O'Neil's" => "O", "Neil"
065     */
066    final boolean stemEnglishPossessive;
067
068    private final byte[] charTypeTable;
069
070    /**
071     * if true, need to skip over a possessive found in the last call to next()
072     */
073    private boolean skipPossessive = false;
074
075    // TODO: should there be a WORD_DELIM category for chars that only separate words (no catenation of subwords will be
076    // done if separated by these chars?) "," would be an obvious candidate...
077    static {
078        byte[] tab = new byte[256];
079        for (int i = 0; i < 256; i++) {
080            byte code = 0;
081            if (Character.isLowerCase(i)) {
082                code |= LOWER;
083            } else if (Character.isUpperCase(i)) {
084                code |= UPPER;
085            } else if (Character.isDigit(i)) {
086                code |= DIGIT;
087            }
088            if (code == 0) {
089                code = SUBWORD_DELIM;
090            }
091            tab[i] = code;
092        }
093        DEFAULT_WORD_DELIM_TABLE = tab;
094    }
095
096    /**
097     * Create a new WordDelimiterIterator operating with the supplied rules.
098     *
099     * @param charTypeTable         table containing character types
100     * @param splitOnCaseChange     if true, causes "PowerShot" to be two tokens; ("Power-Shot" remains two parts regards)
101     * @param splitOnNumerics       if true, causes "j2se" to be three tokens; "j" "2" "se"
102     * @param stemEnglishPossessive if true, causes trailing "'s" to be removed for each subword: "O'Neil's" => "O", "Neil"
103     */
104    WordDelimiterIterator(byte[] charTypeTable, boolean splitOnCaseChange, boolean splitOnNumerics, boolean stemEnglishPossessive) {
105        this.charTypeTable = charTypeTable;
106        this.splitOnCaseChange = splitOnCaseChange;
107        this.splitOnNumerics = splitOnNumerics;
108        this.stemEnglishPossessive = stemEnglishPossessive;
109    }
110
111    /**
112     * Advance to the next subword in the string.
113     *
114     * @return index of the next subword, or {@link #DONE} if all subwords have been returned
115     */
116    int next() {
117        current = end;
118        if (current == DONE) {
119            return DONE;
120        }
121
122        if (skipPossessive) {
123            current += 2;
124            skipPossessive = false;
125        }
126
127        int lastType = 0;
128
129        while (current < endBounds && (isSubwordDelim(lastType = charType(text[current])))) {
130            current++;
131        }
132
133        if (current >= endBounds) {
134            return end = DONE;
135        }
136
137        for (end = current + 1; end < endBounds; end++) {
138            int type = charType(text[end]);
139            if (isBreak(lastType, type)) {
140                break;
141            }
142            lastType = type;
143        }
144
145        if (end < endBounds - 1 && endsWithPossessive(end + 2)) {
146            skipPossessive = true;
147        }
148
149        return end;
150    }
151
152
153    /**
154     * Return the type of the current subword.
155     * This currently uses the type of the first character in the subword.
156     *
157     * @return type of the current word
158     */
159    int type() {
160        if (end == DONE) {
161            return 0;
162        }
163
164        int type = charType(text[current]);
165        switch (type) {
166            // return ALPHA word type for both lower and upper
167            case LOWER:
168            case UPPER:
169                return ALPHA;
170            default:
171                return type;
172        }
173    }
174
175    /**
176     * Reset the text to a new value, and reset all state
177     *
178     * @param text   New text
179     * @param length length of the text
180     */
181    void setText(char text[], int length) {
182        this.text = text;
183        this.length = this.endBounds = length;
184        current = startBounds = end = 0;
185        skipPossessive = hasFinalPossessive = false;
186        setBounds();
187    }
188
189    // ================================================= Helper Methods ================================================
190
191    /**
192     * Determines whether the transition from lastType to type indicates a break
193     *
194     * @param lastType Last subword type
195     * @param type     Current subword type
196     * @return {@code true} if the transition indicates a break, {@code false} otherwise
197     */
198    private boolean isBreak(int lastType, int type) {
199        if ((type & lastType) != 0) {
200            return false;
201        }
202
203        if (!splitOnCaseChange && isAlpha(lastType) && isAlpha(type)) {
204            // ALPHA->ALPHA: always ignore if case isn't considered.
205            return false;
206        } else if (isUpper(lastType) && isAlpha(type)) {
207            // UPPER->letter: Don't split
208            return false;
209        } else if (!splitOnNumerics && ((isAlpha(lastType) && isDigit(type)) || (isDigit(lastType) && isAlpha(type)))) {
210            // ALPHA->NUMERIC, NUMERIC->ALPHA :Don't split
211            return false;
212        }
213
214        return true;
215    }
216
217    /**
218     * Determines if the current word contains only one subword.  Note, it could be potentially surrounded by delimiters
219     *
220     * @return {@code true} if the current word contains only one subword, {@code false} otherwise
221     */
222    boolean isSingleWord() {
223        if (hasFinalPossessive) {
224            return current == startBounds && end == endBounds - 2;
225        } else {
226            return current == startBounds && end == endBounds;
227        }
228    }
229
230    /**
231     * Set the internal word bounds (remove leading and trailing delimiters). Note, if a possessive is found, don't remove
232     * it yet, simply note it.
233     */
234    private void setBounds() {
235        while (startBounds < length && (isSubwordDelim(charType(text[startBounds])))) {
236            startBounds++;
237        }
238
239        while (endBounds > startBounds && (isSubwordDelim(charType(text[endBounds - 1])))) {
240            endBounds--;
241        }
242        if (endsWithPossessive(endBounds)) {
243            hasFinalPossessive = true;
244        }
245        current = startBounds;
246    }
247
248    /**
249     * Determines if the text at the given position indicates an English possessive which should be removed
250     *
251     * @param pos Position in the text to check if it indicates an English possessive
252     * @return {@code true} if the text at the position indicates an English posessive, {@code false} otherwise
253     */
254    private boolean endsWithPossessive(int pos) {
255        return (stemEnglishPossessive &&
256                pos > 2 &&
257                text[pos - 2] == '\'' &&
258                (text[pos - 1] == 's' || text[pos - 1] == 'S') &&
259                isAlpha(charType(text[pos - 3])) &&
260                (pos == endBounds || isSubwordDelim(charType(text[pos]))));
261    }
262
263    /**
264     * Determines the type of the given character
265     *
266     * @param ch Character whose type is to be determined
267     * @return Type of the character
268     */
269    private int charType(int ch) {
270        if (ch < charTypeTable.length) {
271            return charTypeTable[ch];
272        }
273        return getType(ch);
274    }
275
276    /**
277     * Computes the type of the given character
278     *
279     * @param ch Character whose type is to be determined
280     * @return Type of the character
281     */
282    public static byte getType(int ch) {
283        switch (Character.getType(ch)) {
284            case Character.UPPERCASE_LETTER:
285                return UPPER;
286            case Character.LOWERCASE_LETTER:
287                return LOWER;
288
289            case Character.TITLECASE_LETTER:
290            case Character.MODIFIER_LETTER:
291            case Character.OTHER_LETTER:
292            case Character.NON_SPACING_MARK:
293            case Character.ENCLOSING_MARK:  // depends what it encloses?
294            case Character.COMBINING_SPACING_MARK:
295                return ALPHA;
296
297            case Character.DECIMAL_DIGIT_NUMBER:
298            case Character.LETTER_NUMBER:
299            case Character.OTHER_NUMBER:
300                return DIGIT;
301
302            // case Character.SPACE_SEPARATOR:
303            // case Character.LINE_SEPARATOR:
304            // case Character.PARAGRAPH_SEPARATOR:
305            // case Character.CONTROL:
306            // case Character.FORMAT:
307            // case Character.PRIVATE_USE:
308
309            case Character.SURROGATE:  // prevent splitting
310                return ALPHA | DIGIT;
311
312            // case Character.DASH_PUNCTUATION:
313            // case Character.START_PUNCTUATION:
314            // case Character.END_PUNCTUATION:
315            // case Character.CONNECTOR_PUNCTUATION:
316            // case Character.OTHER_PUNCTUATION:
317            // case Character.MATH_SYMBOL:
318            // case Character.CURRENCY_SYMBOL:
319            // case Character.MODIFIER_SYMBOL:
320            // case Character.OTHER_SYMBOL:
321            // case Character.INITIAL_QUOTE_PUNCTUATION:
322            // case Character.FINAL_QUOTE_PUNCTUATION:
323
324            default:
325                return SUBWORD_DELIM;
326        }
327    }
328}
