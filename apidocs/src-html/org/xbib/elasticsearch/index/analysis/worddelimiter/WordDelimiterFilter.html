017/**
018 * Splits words into subwords and performs optional transformations on subword
019 * groups. Words are split into subwords with the following rules:
020 * <ul>
021 * <li>split on intra-word delimiters (by default, all non alpha-numeric
022 * characters): <code>"Wi-Fi"</code> &#8594; <code>"Wi", "Fi"</code></li>
023 * <li>split on case transitions: <code>"PowerShot"</code> &#8594;
024 * <code>"Power", "Shot"</code></li>
025 * <li>split on letter-number transitions: <code>"SD500"</code> &#8594;
026 * <code>"SD", "500"</code></li>
027 * <li>leading and trailing intra-word delimiters on each subword are ignored:
028 * <code>"//hello---there, 'dude'"</code> &#8594;
029 * <code>"hello", "there", "dude"</code></li>
030 * <li>trailing "'s" are removed for each subword: <code>"O'Neil's"</code>
031 * &#8594; <code>"O", "Neil"</code>
032 * <ul>
033 * <li>Note: this step isn't performed in a separate filter because of possible
034 * subword combinations.</li>
035 * </ul>
036 * </li>
037 * </ul>
038 * <p/>
039 * The <b>combinations</b> parameter affects how subwords are combined:
040 * <ul>
041 * <li>combinations="0" causes no subword combinations: <code>"PowerShot"</code>
042 * &#8594; <code>0:"Power", 1:"Shot"</code> (0 and 1 are the token positions)</li>
043 * <li>combinations="1" means that in addition to the subwords, maximum runs of
044 * non-numeric subwords are catenated and produced at the same position of the
045 * last subword in the run:
046 * <ul>
047 * <li><code>"PowerShot"</code> &#8594;
048 * <code>0:"Power", 1:"Shot" 1:"PowerShot"</code></li>
049 * <li><code>"A's+B's&C's"</code> -gt; <code>0:"A", 1:"B", 2:"C", 2:"ABC"</code>
050 * </li>
051 * <li><code>"Super-Duper-XL500-42-AutoCoder!"</code> &#8594;
052 * <code>0:"Super", 1:"Duper", 2:"XL", 2:"SuperDuperXL", 3:"500" 4:"42", 5:"Auto", 6:"Coder", 6:"AutoCoder"</code>
053 * </li>
054 * </ul>
055 * </li>
056 * </ul>
057 * One use for {@link WordDelimiterFilter} is to help match words with different
058 * subword delimiters. For example, if the source text contained "wi-fi" one may
059 * want "wifi" "WiFi" "wi-fi" "wi+fi" queries to all match. One way of doing so
060 * is to specify combinations="1" in the analyzer used for indexing, and
061 * combinations="0" (the default) in the analyzer used for querying. Given that
062 * the current {@link StandardTokenizer} immediately removes many intra-word
063 * delimiters, it is recommended that this filter be used after a tokenizer that
064 * does not do this (such as {@link WhitespaceTokenizer}).
065 */
066public final class WordDelimiterFilter extends TokenFilter {
067
068    public static final int LOWER = 0x01;
069
070    public static final int UPPER = 0x02;
071
072    public static final int DIGIT = 0x04;
073
074    public static final int SUBWORD_DELIM = 0x08;
075
076    public static final int ALPHA = 0x03;
077
078    public static final int ALPHANUM = 0x07;
079
080    /**
081     * Causes parts of words to be generated:
082     * <p/>
083     * "PowerShot" => "Power" "Shot"
084     */
085    public static final int GENERATE_WORD_PARTS = 1;
086
087    /**
088     * Causes number subwords to be generated:
089     * <p/>
090     * "500-42" => "500" "42"
091     */
092    public static final int GENERATE_NUMBER_PARTS = 2;
093
094    /**
095     * Causes maximum runs of word parts to be catenated:
096     * <p/>
097     * "wi-fi" => "wifi"
098     */
099    public static final int CATENATE_WORDS = 4;
100
101    /**
102     * Causes maximum runs of word parts to be catenated:
103     * <p/>
104     * "wi-fi" => "wifi"
105     */
106    public static final int CATENATE_NUMBERS = 8;
107
108    /**
109     * Causes all subword parts to be catenated:
110     * <p/>
111     * "wi-fi-4000" => "wifi4000"
112     */
113    public static final int CATENATE_ALL = 16;
114
115    /**
116     * Causes original words are preserved and added to the subword list (Defaults to false)
117     * <p/>
118     * "500-42" => "500" "42" "500-42"
119     */
120    public static final int PRESERVE_ORIGINAL = 32;
121
122    /**
123     * If not set, causes case changes to be ignored (subwords will only be generated
124     * given SUBWORD_DELIM tokens)
125     */
126    public static final int SPLIT_ON_CASE_CHANGE = 64;
127
128    /**
129     * If not set, causes numeric changes to be ignored (subwords will only be generated
130     * given SUBWORD_DELIM tokens).
131     */
132    public static final int SPLIT_ON_NUMERICS = 128;
133
134    /**
135     * Causes trailing "'s" to be removed for each subword
136     * <p/>
137     * "O'Neil's" => "O", "Neil"
138     */
139    public static final int STEM_ENGLISH_POSSESSIVE = 256;

141    /**
142     * If not null is the set of tokens to protect from being delimited
143     */
144    private final CharArraySet protWords;
145
146    private final int flags;
147
148    private final CharTermAttribute termAttribute = addAttribute(CharTermAttribute.class);
149    private final OffsetAttribute offsetAttribute = addAttribute(OffsetAttribute.class);
150    private final PositionIncrementAttribute posIncAttribute = addAttribute(PositionIncrementAttribute.class);
151    private final TypeAttribute typeAttribute = addAttribute(TypeAttribute.class);
152
153    // used for iterating word delimiter breaks
154    private final WordDelimiterIterator iterator;
155
156    // used for concatenating runs of similar typed subwords (word,number)
157    private final WordDelimiterConcatenation concat = new WordDelimiterConcatenation();
158    // number of subwords last output by concat.
159    private int lastConcatCount = 0;
160
161    // used for catenate all
162    private final WordDelimiterConcatenation concatAll = new WordDelimiterConcatenation();
163
164    // used for accumulating position increment gaps
165    private int accumPosInc = 0;
166
167    private char savedBuffer[] = new char[1024];
168    private int savedStartOffset;
169    private int savedEndOffset;
170    private String savedType;
171    private boolean hasSavedState = false;
172    // if length by start + end offsets doesn't match the term text then assume
173    // this is a synonym and don't adjust the offsets.
174    private boolean hasIllegalOffsets = false;
175
176    // for a run of the same subword type within a word, have we output anything?
177    private boolean hasOutputToken = false;
178    // when preserve original is on, have we output any token following it?
179    // this token must have posInc=0!
180    private boolean hasOutputFollowingOriginal = false;
181
182    /**
183     * Creates a new WordDelimiterFilter
184     *
185     * @param in                 TokenStream to be filtered
186     * @param charTypeTable      table containing character types
187     * @param configurationFlags Flags configuring the filter
188     * @param protWords          If not null is the set of tokens to protect from being delimited
189     */
190    public WordDelimiterFilter(TokenStream in, byte[] charTypeTable, int configurationFlags, CharArraySet protWords) {
191        super(in);
192        this.flags = configurationFlags;
193        this.protWords = protWords;
194        this.iterator = new WordDelimiterIterator(
195                charTypeTable, has(SPLIT_ON_CASE_CHANGE), has(SPLIT_ON_NUMERICS), has(STEM_ENGLISH_POSSESSIVE));
196    }
197
198    /**
199     * Creates a new WordDelimiterFilter using {@link WordDelimiterIterator#DEFAULT_WORD_DELIM_TABLE}
200     * as its charTypeTable
201     *
202     * @param in                 TokenStream to be filtered
203     * @param configurationFlags Flags configuring the filter
204     * @param protWords          If not null is the set of tokens to protect from being delimited
205     */
206    public WordDelimiterFilter(TokenStream in, int configurationFlags, CharArraySet protWords) {
207        this(in, WordDelimiterIterator.DEFAULT_WORD_DELIM_TABLE, configurationFlags, protWords);
208    }

210    @Override
211    public boolean incrementToken() throws IOException {
212        while (true) {
213            if (!hasSavedState) {
214                // process a new input word
215                if (!input.incrementToken()) {
216                    return false;
217                }
218
219                int termLength = termAttribute.length();
220                char[] termBuffer = termAttribute.buffer();
221
222                accumPosInc += posIncAttribute.getPositionIncrement();
223
224                iterator.setText(termBuffer, termLength);
225                iterator.next();
226
227                // word of no delimiters, or protected word: just return it
228                if ((iterator.current == 0 && iterator.end == termLength) ||
229                        (protWords != null && protWords.contains(termBuffer, 0, termLength))) {
230                    posIncAttribute.setPositionIncrement(accumPosInc);
231                    accumPosInc = 0;
232                    return true;
233                }
234
235                // word of simply delimiters
236                if (iterator.end == WordDelimiterIterator.DONE && !has(PRESERVE_ORIGINAL)) {
237                    // if the posInc is 1, simply ignore it in the accumulation
238                    if (posIncAttribute.getPositionIncrement() == 1) {
239                        accumPosInc--;
240                    }
241                    continue;
242                }
243
244                saveState();
245
246                hasOutputToken = false;
247                hasOutputFollowingOriginal = !has(PRESERVE_ORIGINAL);
248                lastConcatCount = 0;
249
250                if (has(PRESERVE_ORIGINAL)) {
251                    posIncAttribute.setPositionIncrement(accumPosInc);
252                    accumPosInc = 0;
253                    return true;
254                }
255            }
256
257            // at the end of the string, output any concatenations
258            if (iterator.end == WordDelimiterIterator.DONE) {
259                if (!concat.isEmpty()) {
260                    if (flushConcatenation(concat)) {
261                        return true;
262                    }
263                }
264
265                if (!concatAll.isEmpty()) {
266                    // only if we haven't output this same combo above!
267                    if (concatAll.subwordCount > lastConcatCount) {
268                        concatAll.writeAndClear();
269                        return true;
270                    }
271                    concatAll.clear();
272                }
273
274                // no saved concatenations, on to the next input word
275                hasSavedState = false;
276                continue;
277            }
278
279            // word surrounded by delimiters: always output
280            if (iterator.isSingleWord()) {
281                generatePart(true);
282                iterator.next();
283                return true;
284            }
285
286            int wordType = iterator.type();
287
288            // do we already have queued up incompatible concatenations?
289            if (!concat.isEmpty() && (concat.type & wordType) == 0) {
290                if (flushConcatenation(concat)) {
291                    hasOutputToken = false;
292                    return true;
293                }
294                hasOutputToken = false;
295            }
296
297            // add subwords depending upon options
298            if (shouldConcatenate(wordType)) {
299                if (concat.isEmpty()) {
300                    concat.type = wordType;
301                }
302                concatenate(concat);
303            }
304
305            // add all subwords (catenateAll)
306            if (has(CATENATE_ALL)) {
307                concatenate(concatAll);
308            }
309
310            // if we should output the word or number part
311            if (shouldGenerateParts(wordType)) {
312                generatePart(false);
313                iterator.next();
314                return true;
315            }
316
317            iterator.next();
318        }
319    }

321    /**
322     * {@inheritDoc}
323     */
324    @Override
325    public void reset() throws IOException {
326        super.reset();
327        hasSavedState = false;
328        concat.clear();
329        concatAll.clear();
330        accumPosInc = 0;
331    }

335    /**
336     * Saves the existing attribute states
337     */
338    private void saveState() {
339        // otherwise, we have delimiters, save state
340        savedStartOffset = offsetAttribute.startOffset();
341        savedEndOffset = offsetAttribute.endOffset();
342        // if length by start + end offsets doesn't match the term text then assume this is a synonym and don't adjust the offsets.
343        hasIllegalOffsets = (savedEndOffset - savedStartOffset != termAttribute.length());
344        savedType = typeAttribute.type();
345
346        if (savedBuffer.length < termAttribute.length()) {
347            savedBuffer = new char[ArrayUtil.oversize(termAttribute.length(), RamUsageEstimator.NUM_BYTES_CHAR)];
348        }
349
350        System.arraycopy(termAttribute.buffer(), 0, savedBuffer, 0, termAttribute.length());
351        iterator.text = savedBuffer;
352
353        hasSavedState = true;
354    }
355
356    /**
357     * Flushes the given WordDelimiterConcatenation by either writing its concat and then clearing, or just clearing.
358     *
359     * @param concatenation WordDelimiterConcatenation that will be flushed
360     * @return {@code true} if the concatenation was written before it was cleared, {@code false} otherwise
361     */
362    private boolean flushConcatenation(WordDelimiterConcatenation concatenation) {
363        lastConcatCount = concatenation.subwordCount;
364        if (concatenation.subwordCount != 1 || !shouldGenerateParts(concatenation.type)) {
365            concatenation.writeAndClear();
366            return true;
367        }
368        concatenation.clear();
369        return false;
370    }
371
372    /**
373     * Determines whether to concatenate a word or number if the current word is the given type
374     *
375     * @param wordType Type of the current word used to determine if it should be concatenated
376     * @return {@code true} if concatenation should occur, {@code false} otherwise
377     */
378    private boolean shouldConcatenate(int wordType) {
379        return (has(CATENATE_WORDS) && isAlpha(wordType)) || (has(CATENATE_NUMBERS) && isDigit(wordType));
380    }
381
382    /**
383     * Determines whether a word/number part should be generated for a word of the given type
384     *
385     * @param wordType Type of the word used to determine if a word/number part should be generated
386     * @return {@code true} if a word/number part should be generated, {@code false} otherwise
387     */
388    private boolean shouldGenerateParts(int wordType) {
389        return (has(GENERATE_WORD_PARTS) && isAlpha(wordType)) || (has(GENERATE_NUMBER_PARTS) && isDigit(wordType));
390    }
391
392    /**
393     * Concatenates the saved buffer to the given WordDelimiterConcatenation
394     *
395     * @param concatenation WordDelimiterConcatenation to concatenate the buffer to
396     */
397    private void concatenate(WordDelimiterConcatenation concatenation) {
398        if (concatenation.isEmpty()) {
399            concatenation.startOffset = savedStartOffset + iterator.current;
400        }
401        concatenation.append(savedBuffer, iterator.current, iterator.end - iterator.current);
402        concatenation.endOffset = savedStartOffset + iterator.end;
403    }
404
405    /**
406     * Generates a word/number part, updating the appropriate attributes
407     *
408     * @param isSingleWord {@code true} if the generation is occurring from a single word, {@code false} otherwise
409     */
410    private void generatePart(boolean isSingleWord) {
411        clearAttributes();
412        termAttribute.copyBuffer(savedBuffer, iterator.current, iterator.end - iterator.current);
413
414        int startOffset = savedStartOffset + iterator.current;
415        int endOffset = savedStartOffset + iterator.end;
416
417        if (hasIllegalOffsets) {
418            // historically this filter did this regardless for 'isSingleWord',
419            // but we must do a sanity check:
420            if (isSingleWord && startOffset <= savedEndOffset) {
421                offsetAttribute.setOffset(startOffset, savedEndOffset);
422            } else {
423                offsetAttribute.setOffset(savedStartOffset, savedEndOffset);
424            }
425        } else {
426            offsetAttribute.setOffset(startOffset, endOffset);
427        }
428        posIncAttribute.setPositionIncrement(position(false));
429        typeAttribute.setType(savedType);
430    }

432    /**
433     * Get the position increment gap for a subword or concatenation
434     *
435     * @param inject true if this token wants to be injected
436     * @return position increment gap
437     */
438    private int position(boolean inject) {
439        int posInc = accumPosInc;
440
441        if (hasOutputToken) {
442            accumPosInc = 0;
443            return inject ? 0 : Math.max(1, posInc);
444        }
445
446        hasOutputToken = true;
447
448        if (!hasOutputFollowingOriginal) {
449            // the first token following the original is 0 regardless
450            hasOutputFollowingOriginal = true;
451            return 0;
452        }
453        // clear the accumulated position increment
454        accumPosInc = 0;
455        return Math.max(1, posInc);
456    }

458    /**
459     * Checks if the given word type includes {@link #ALPHA}
460     *
461     * @param type Word type to check
462     * @return {@code true} if the type contains ALPHA, {@code false} otherwise
463     */
464    static boolean isAlpha(int type) {
465        return (type & ALPHA) != 0;
466    }
467
468    /**
469     * Checks if the given word type includes {@link #DIGIT}
470     *
471     * @param type Word type to check
472     * @return {@code true} if the type contains DIGIT, {@code false} otherwise
473     */
474    static boolean isDigit(int type) {
475        return (type & DIGIT) != 0;
476    }
477
478    /**
479     * Checks if the given word type includes {@link #SUBWORD_DELIM}
480     *
481     * @param type Word type to check
482     * @return {@code true} if the type contains SUBWORD_DELIM, {@code false} otherwise
483     */
484    static boolean isSubwordDelim(int type) {
485        return (type & SUBWORD_DELIM) != 0;
486    }
487
488    /**
489     * Checks if the given word type includes {@link #UPPER}
490     *
491     * @param type Word type to check
492     * @return {@code true} if the type contains UPPER, {@code false} otherwise
493     */
494    static boolean isUpper(int type) {

496    }

498    /**
499     * Determines whether the given flag is set
500     *
501     * @param flag Flag to see if set
502     * @return {@code true} if flag is set
503     */
504    private boolean has(int flag) {
505        return (flags & flag) != 0;
506    }

510    /**
511     * A WDF concatenated 'run'
512     */
513    final class WordDelimiterConcatenation {
514        final StringBuilder buffer = new StringBuilder();
515        int startOffset;
516        int endOffset;
517        int type;
518        int subwordCount;
519
520        /**
521         * Appends the given text of the given length, to the concetenation at the given offset
522         *
523         * @param text   Text to append
524         * @param offset Offset in the concetenation to add the text
525         * @param length Length of the text to append
526         */
527        void append(char text[], int offset, int length) {
528            buffer.append(text, offset, length);
529            subwordCount++;
530        }
531
532        /**
533         * Writes the concatenation to the attributes
534         */
535        void write() {
536            clearAttributes();
537            if (termAttribute.length() < buffer.length()) {
538                termAttribute.resizeBuffer(buffer.length());
539            }
540            char termbuffer[] = termAttribute.buffer();
541
542            buffer.getChars(0, buffer.length(), termbuffer, 0);
543            termAttribute.setLength(buffer.length());
544
545            if (hasIllegalOffsets) {
546                offsetAttribute.setOffset(savedStartOffset, savedEndOffset);
547            } else {
548                offsetAttribute.setOffset(startOffset, endOffset);
549            }
550            posIncAttribute.setPositionIncrement(position(true));
551            typeAttribute.setType(savedType);
552            accumPosInc = 0;
553        }
554
555        /**
556         * Determines if the concatenation is empty
557         *
558         * @return {@code true} if the concatenation is empty, {@code false} otherwise
559         */
560        boolean isEmpty() {
561            return buffer.length() == 0;
562        }
563
564        /**
565         * Clears the concatenation and resets its state
566         */
567        void clear() {
568            buffer.setLength(0);
569            startOffset = endOffset = type = subwordCount = 0;
570        }
571
572        /**
573         * Convenience method for the common scenario of having to write the concetenation and then clearing its state
574         */
575        void writeAndClear() {
576            write();
577            clear();
578        }
579    }
