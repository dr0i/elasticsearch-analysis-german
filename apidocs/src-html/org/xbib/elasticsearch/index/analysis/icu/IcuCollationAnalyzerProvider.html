019package org.xbib.elasticsearch.index.analysis.icu;
020
021import com.ibm.icu.text.Collator;
022import com.ibm.icu.text.RuleBasedCollator;
023import com.ibm.icu.util.ULocale;
024import org.apache.lucene.collation.ICUCollationKeyAnalyzer;
025import org.elasticsearch.ElasticSearchIllegalArgumentException;
026import org.elasticsearch.common.inject.Inject;
027import org.elasticsearch.common.inject.assistedinject.Assisted;
028import org.elasticsearch.common.settings.Settings;
029import org.elasticsearch.env.Environment;
030import org.elasticsearch.env.FailedToResolveConfigException;
031import org.elasticsearch.index.Index;
032import org.elasticsearch.index.analysis.AbstractIndexAnalyzerProvider;
033import org.elasticsearch.index.settings.IndexSettings;
034
035import java.io.IOException;
036
037/**
038 * An ICU collation analyzer provider.
039 * <p/>
040 * There are two ways to configure collation:
041 * <p/>
042 * <p>The first is simply specifying the locale (defaults to the default locale).
043 * The <tt>language</tt> parameter is the lowercase two-letter ISO-639 code.
044 * An additional <tt>country</tt> and <tt>variant</tt> can be provided.
045 * <p/>
046 * <p>The second option is to specify collation rules as defined in the
047 * <a href="http://www.icu-project.org/userguide/Collate_Customization.html">
048 * Collation customization</a> chapter in icu docs. The <tt>rules</tt> parameter can either
049 * embed the rules definition in the settings or refer to an external location
050 * (preferable located under the <tt>config</tt> location, relative to it).
051 */
052public class IcuCollationAnalyzerProvider extends AbstractIndexAnalyzerProvider<ICUCollationKeyAnalyzer> {
053
054    private final ICUCollationKeyAnalyzer analyzer;
055
056    @Inject
057    public IcuCollationAnalyzerProvider(Index index,
058                                        @IndexSettings Settings indexSettings,
059                                        Environment environment,
060                                        @Assisted String name,
061                                        @Assisted Settings settings) {
062        super(index, indexSettings, name, settings);
063
064        Collator collator;
065        String rules = settings.get("rules");
066        if (rules != null) {
067            FailedToResolveConfigException failureToResolve = null;
068            try {
069                rules = environment.resolveConfigAndLoadToString(rules);
070            } catch (FailedToResolveConfigException e) {
071                failureToResolve = e;
072            } catch (IOException e) {
073                throw new ElasticSearchIllegalArgumentException("Failed to load collation rules", e);
074            }
075            try {
076                collator = new RuleBasedCollator(rules);
077            } catch (Exception e) {
078                if (failureToResolve != null) {
079                    throw new ElasticSearchIllegalArgumentException("Failed to resolve collation rules location", failureToResolve);
080                } else {
081                    throw new ElasticSearchIllegalArgumentException("Failed to parse collation rules", e);
082                }
083            }
084        } else {
085            String language = settings.get("language");
086            if (language != null) {
087                ULocale locale;
088                String country = settings.get("country");
089                if (country != null) {
090                    String variant = settings.get("variant");
091                    if (variant != null) {
092                        locale = new ULocale(language, country, variant);
093                    } else {
094                        locale = new ULocale(language, country);
095                    }
096                } else {
097                    locale = new ULocale(language);
098                }
099                collator = Collator.getInstance(locale);
100            } else {
101                collator = Collator.getInstance();
102            }
103        }
104
105        // set the strength flag, otherwise it will be the default.
106        String strength = settings.get("strength");
107        if (strength != null) {
108            if (strength.equalsIgnoreCase("primary")) {
109                collator.setStrength(Collator.PRIMARY);
110            } else if (strength.equalsIgnoreCase("secondary")) {
111                collator.setStrength(Collator.SECONDARY);
112            } else if (strength.equalsIgnoreCase("tertiary")) {
113                collator.setStrength(Collator.TERTIARY);
114            } else if (strength.equalsIgnoreCase("quaternary")) {
115                collator.setStrength(Collator.QUATERNARY);
116            } else if (strength.equalsIgnoreCase("identical")) {
117                collator.setStrength(Collator.IDENTICAL);
118            } else {
119                throw new ElasticSearchIllegalArgumentException("Invalid strength: " + strength);
120            }
121        }
122
123        // set the decomposition flag, otherwise it will be the default.
124        String decomposition = settings.get("decomposition");
125        if (decomposition != null) {
126            if (decomposition.equalsIgnoreCase("no")) {
127                collator.setDecomposition(Collator.NO_DECOMPOSITION);
128            } else if (decomposition.equalsIgnoreCase("canonical")) {
129                collator.setDecomposition(Collator.CANONICAL_DECOMPOSITION);
130            } else {
131                throw new ElasticSearchIllegalArgumentException("Invalid decomposition: " + decomposition);
132            }
133        }
134
135        // expert options: concrete subclasses are always a RuleBasedCollator
136        RuleBasedCollator rbc = (RuleBasedCollator) collator;
137        String alternate = settings.get("alternate");
138        if (alternate != null) {
139            if (alternate.equalsIgnoreCase("shifted")) {
140                rbc.setAlternateHandlingShifted(true);
141            } else if (alternate.equalsIgnoreCase("non-ignorable")) {
142                rbc.setAlternateHandlingShifted(false);
143            } else {
144                throw new ElasticSearchIllegalArgumentException("Invalid alternate: " + alternate);
145            }
146        }
147
148        Boolean caseLevel = settings.getAsBoolean("caseLevel", null);
149        if (caseLevel != null) {
150            rbc.setCaseLevel(caseLevel);
151        }
152
153        String caseFirst = settings.get("caseFirst");
154        if (caseFirst != null) {
155            if (caseFirst.equalsIgnoreCase("lower")) {
156                rbc.setLowerCaseFirst(true);
157            } else if (caseFirst.equalsIgnoreCase("upper")) {
158                rbc.setUpperCaseFirst(true);
159            } else {
160                throw new ElasticSearchIllegalArgumentException("Invalid caseFirst: " + caseFirst);
161            }
162        }
163
164        Boolean numeric = settings.getAsBoolean("numeric", null);
165        if (numeric != null) {
166            rbc.setNumericCollation(numeric);
167        }
168
169        String variableTop = settings.get("variableTop");
170        if (variableTop != null) {
171            rbc.setVariableTop(variableTop);
172        }
173
174        Boolean hiraganaQuaternaryMode = settings.getAsBoolean("hiraganaQuaternaryMode", null);
175        if (hiraganaQuaternaryMode != null) {
176            rbc.setHiraganaQuaternary(hiraganaQuaternaryMode);
177        }
178
179        this.analyzer = new ICUCollationKeyAnalyzer(version, collator);
180    }
181
182    @Override
183    public ICUCollationKeyAnalyzer get() {
184        return analyzer;
185    }
186}
