001
002package org.xbib.elasticsearch.index.analysis.icu;
003
004import com.ibm.icu.text.Collator;
005import com.ibm.icu.text.RuleBasedCollator;
006import com.ibm.icu.util.ULocale;
007import org.apache.lucene.collation.ICUCollationKeyAnalyzer;
008import org.elasticsearch.ElasticsearchIllegalArgumentException;
009import org.elasticsearch.common.inject.Inject;
010import org.elasticsearch.common.inject.assistedinject.Assisted;
011import org.elasticsearch.common.settings.Settings;
012import org.elasticsearch.env.Environment;
013import org.elasticsearch.env.FailedToResolveConfigException;
014import org.elasticsearch.index.Index;
015import org.elasticsearch.index.analysis.AbstractIndexAnalyzerProvider;
016import org.elasticsearch.index.settings.IndexSettings;
017
018import java.io.IOException;
019
020/**
021 * An ICU collation analyzer provider.
022 * <p/>
023 * There are two ways to configure collation:
024 * <p/>
025 * <p>The first is simply specifying the locale (defaults to the default locale).
026 * The <tt>language</tt> parameter is the lowercase two-letter ISO-639 code.
027 * An additional <tt>country</tt> and <tt>variant</tt> can be provided.
028 * <p/>
029 * <p>The second option is to specify collation rules as defined in the
030 * <a href="http://www.icu-project.org/userguide/Collate_Customization.html">
031 * Collation customization</a> chapter in icu docs. The <tt>rules</tt> parameter can either
032 * embed the rules definition in the settings or refer to an external location
033 * (preferable located under the <tt>config</tt> location, relative to it).
034 */
035public class IcuCollationAnalyzerProvider extends AbstractIndexAnalyzerProvider<ICUCollationKeyAnalyzer> {
036
037    private final ICUCollationKeyAnalyzer analyzer;
038
039    @Inject
040    public IcuCollationAnalyzerProvider(Index index,
041                                        @IndexSettings Settings indexSettings,
042                                        Environment environment,
043                                        @Assisted String name,
044                                        @Assisted Settings settings) {
045        super(index, indexSettings, name, settings);
046
047        Collator collator;
048        String rules = settings.get("rules");
049        if (rules != null) {
050            FailedToResolveConfigException failureToResolve = null;
051            try {
052                rules = environment.resolveConfigAndLoadToString(rules);
053            } catch (FailedToResolveConfigException e) {
054                failureToResolve = e;
055            } catch (IOException e) {
056                throw new ElasticsearchIllegalArgumentException("Failed to load collation rules", e);
057            }
058            try {
059                collator = new RuleBasedCollator(rules);
060            } catch (Exception e) {
061                if (failureToResolve != null) {
062                    throw new ElasticsearchIllegalArgumentException("Failed to resolve collation rules location", failureToResolve);
063                } else {
064                    throw new ElasticsearchIllegalArgumentException("Failed to parse collation rules", e);
065                }
066            }
067        } else {
068            String language = settings.get("language");
069            if (language != null) {
070                ULocale locale;
071                String country = settings.get("country");
072                if (country != null) {
073                    String variant = settings.get("variant");
074                    if (variant != null) {
075                        locale = new ULocale(language, country, variant);
076                    } else {
077                        locale = new ULocale(language, country);
078                    }
079                } else {
080                    locale = new ULocale(language);
081                }
082                collator = Collator.getInstance(locale);
083            } else {
084                collator = Collator.getInstance();
085            }
086        }
087
088        // set the strength flag, otherwise it will be the default.
089        String strength = settings.get("strength");
090        if (strength != null) {
091            if (strength.equalsIgnoreCase("primary")) {
092                collator.setStrength(Collator.PRIMARY);
093            } else if (strength.equalsIgnoreCase("secondary")) {
094                collator.setStrength(Collator.SECONDARY);
095            } else if (strength.equalsIgnoreCase("tertiary")) {
096                collator.setStrength(Collator.TERTIARY);
097            } else if (strength.equalsIgnoreCase("quaternary")) {
098                collator.setStrength(Collator.QUATERNARY);
099            } else if (strength.equalsIgnoreCase("identical")) {
100                collator.setStrength(Collator.IDENTICAL);
101            } else {
102                throw new ElasticsearchIllegalArgumentException("Invalid strength: " + strength);
103            }
104        }
105
106        // set the decomposition flag, otherwise it will be the default.
107        String decomposition = settings.get("decomposition");
108        if (decomposition != null) {
109            if (decomposition.equalsIgnoreCase("no")) {
110                collator.setDecomposition(Collator.NO_DECOMPOSITION);
111            } else if (decomposition.equalsIgnoreCase("canonical")) {
112                collator.setDecomposition(Collator.CANONICAL_DECOMPOSITION);
113            } else {
114                throw new ElasticsearchIllegalArgumentException("Invalid decomposition: " + decomposition);
115            }
116        }
117
118        // expert options: concrete subclasses are always a RuleBasedCollator
119        RuleBasedCollator rbc = (RuleBasedCollator) collator;
120        String alternate = settings.get("alternate");
121        if (alternate != null) {
122            if (alternate.equalsIgnoreCase("shifted")) {
123                rbc.setAlternateHandlingShifted(true);
124            } else if (alternate.equalsIgnoreCase("non-ignorable")) {
125                rbc.setAlternateHandlingShifted(false);
126            } else {
127                throw new ElasticsearchIllegalArgumentException("Invalid alternate: " + alternate);
128            }
129        }
130
131        Boolean caseLevel = settings.getAsBoolean("caseLevel", null);
132        if (caseLevel != null) {
133            rbc.setCaseLevel(caseLevel);
134        }
135
136        String caseFirst = settings.get("caseFirst");
137        if (caseFirst != null) {
138            if (caseFirst.equalsIgnoreCase("lower")) {
139                rbc.setLowerCaseFirst(true);
140            } else if (caseFirst.equalsIgnoreCase("upper")) {
141                rbc.setUpperCaseFirst(true);
142            } else {
143                throw new ElasticsearchIllegalArgumentException("Invalid caseFirst: " + caseFirst);
144            }
145        }
146
147        Boolean numeric = settings.getAsBoolean("numeric", null);
148        if (numeric != null) {
149            rbc.setNumericCollation(numeric);
150        }
151
152        String variableTop = settings.get("variableTop");
153        if (variableTop != null) {
154            rbc.setVariableTop(variableTop);
155        }
156
157        Boolean hiraganaQuaternaryMode = settings.getAsBoolean("hiraganaQuaternaryMode", null);
158        if (hiraganaQuaternaryMode != null) {
159            rbc.setHiraganaQuaternary(hiraganaQuaternaryMode);
160        }
161
162        this.analyzer = new ICUCollationKeyAnalyzer(version, collator);
163    }
164
165    @Override
166    public ICUCollationKeyAnalyzer get() {
167        return analyzer;
168    }
169}
