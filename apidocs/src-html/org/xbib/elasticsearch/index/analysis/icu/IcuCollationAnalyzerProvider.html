001package org.xbib.elasticsearch.index.analysis.icu;
002
003import com.ibm.icu.text.Collator;
004import com.ibm.icu.text.RuleBasedCollator;
005import com.ibm.icu.util.ULocale;
006import org.apache.lucene.collation.ICUCollationKeyAnalyzer;
007import org.elasticsearch.ElasticsearchIllegalArgumentException;
008import org.elasticsearch.common.inject.Inject;
009import org.elasticsearch.common.inject.assistedinject.Assisted;
010import org.elasticsearch.common.settings.Settings;
011import org.elasticsearch.env.Environment;
012import org.elasticsearch.env.FailedToResolveConfigException;
013import org.elasticsearch.index.Index;
014import org.elasticsearch.index.analysis.AbstractIndexAnalyzerProvider;
015import org.elasticsearch.index.settings.IndexSettings;
016
017import java.io.IOException;
018
019/**
020 * An ICU collation analyzer provider.
021 * <p/>
022 * There are two ways to configure collation:
023 * <p/>
024 * <p>The first is simply specifying the locale (defaults to the default locale).
025 * The <tt>language</tt> parameter is the lowercase two-letter ISO-639 code.
026 * An additional <tt>country</tt> and <tt>variant</tt> can be provided.
027 * <p/>
028 * <p>The second option is to specify collation rules as defined in the
029 * <a href="http://www.icu-project.org/userguide/Collate_Customization.html">
030 * Collation customization</a> chapter in icu docs. The <tt>rules</tt> parameter can either
031 * embed the rules definition in the settings or refer to an external location
032 * (preferable located under the <tt>config</tt> location, relative to it).
033 */
034public class IcuCollationAnalyzerProvider extends AbstractIndexAnalyzerProvider<ICUCollationKeyAnalyzer> {
035
036    private final Collator collator;
037
038    @Inject
039    public IcuCollationAnalyzerProvider(Index index,
040                                        @IndexSettings Settings indexSettings,
041                                        Environment environment,
042                                        @Assisted String name,
043                                        @Assisted Settings settings) {
044        super(index, indexSettings, name, settings);
045
046        String rules = settings.get("rules");
047        if (rules != null) {
048            FailedToResolveConfigException failureToResolve = null;
049            try {
050                rules = environment.resolveConfigAndLoadToString(rules);
051            } catch (FailedToResolveConfigException e) {
052                failureToResolve = e;
053            } catch (IOException e) {
054                throw new ElasticsearchIllegalArgumentException("Failed to load collation rules", e);
055            }
056            try {
057                collator = new RuleBasedCollator(rules);
058            } catch (Exception e) {
059                if (failureToResolve != null) {
060                    throw new ElasticsearchIllegalArgumentException("Failed to resolve collation rules location", failureToResolve);
061                } else {
062                    throw new ElasticsearchIllegalArgumentException("Failed to parse collation rules", e);
063                }
064            }
065        } else {
066            String language = settings.get("language");
067            if (language != null) {
068                ULocale locale;
069                String country = settings.get("country");
070                if (country != null) {
071                    String variant = settings.get("variant");
072                    if (variant != null) {
073                        locale = new ULocale(language, country, variant);
074                    } else {
075                        locale = new ULocale(language, country);
076                    }
077                } else {
078                    locale = new ULocale(language);
079                }
080                collator = Collator.getInstance(locale);
081            } else {
082                collator = Collator.getInstance();
083            }
084        }
085
086        // set the strength flag, otherwise it will be the default.
087        String strength = settings.get("strength");
088        if (strength != null) {
089            if (strength.equalsIgnoreCase("primary")) {
090                collator.setStrength(Collator.PRIMARY);
091            } else if (strength.equalsIgnoreCase("secondary")) {
092                collator.setStrength(Collator.SECONDARY);
093            } else if (strength.equalsIgnoreCase("tertiary")) {
094                collator.setStrength(Collator.TERTIARY);
095            } else if (strength.equalsIgnoreCase("quaternary")) {
096                collator.setStrength(Collator.QUATERNARY);
097            } else if (strength.equalsIgnoreCase("identical")) {
098                collator.setStrength(Collator.IDENTICAL);
099            } else {
100                throw new ElasticsearchIllegalArgumentException("Invalid strength: " + strength);
101            }
102        }
103
104        // set the decomposition flag, otherwise it will be the default.
105        String decomposition = settings.get("decomposition");
106        if (decomposition != null) {
107            if (decomposition.equalsIgnoreCase("no")) {
108                collator.setDecomposition(Collator.NO_DECOMPOSITION);
109            } else if (decomposition.equalsIgnoreCase("canonical")) {
110                collator.setDecomposition(Collator.CANONICAL_DECOMPOSITION);
111            } else {
112                throw new ElasticsearchIllegalArgumentException("Invalid decomposition: " + decomposition);
113            }
114        }
115
116        // expert options: concrete subclasses are always a RuleBasedCollator
117        RuleBasedCollator rbc = (RuleBasedCollator) collator;
118        String alternate = settings.get("alternate");
119        if (alternate != null) {
120            if (alternate.equalsIgnoreCase("shifted")) {
121                rbc.setAlternateHandlingShifted(true);
122            } else if (alternate.equalsIgnoreCase("non-ignorable")) {
123                rbc.setAlternateHandlingShifted(false);
124            } else {
125                throw new ElasticsearchIllegalArgumentException("Invalid alternate: " + alternate);
126            }
127        }
128
129        Boolean caseLevel = settings.getAsBoolean("caseLevel", null);
130        if (caseLevel != null) {
131            rbc.setCaseLevel(caseLevel);
132        }
133
134        String caseFirst = settings.get("caseFirst");
135        if (caseFirst != null) {
136            if (caseFirst.equalsIgnoreCase("lower")) {
137                rbc.setLowerCaseFirst(true);
138            } else if (caseFirst.equalsIgnoreCase("upper")) {
139                rbc.setUpperCaseFirst(true);
140            } else {
141                throw new ElasticsearchIllegalArgumentException("Invalid caseFirst: " + caseFirst);
142            }
143        }
144
145        Boolean numeric = settings.getAsBoolean("numeric", null);
146        if (numeric != null) {
147            rbc.setNumericCollation(numeric);
148        }
149
150        int maxVariable = settings.getAsInt("variableTop", Collator.ReorderCodes.DEFAULT);
151        rbc.setMaxVariable(maxVariable);
152    }
153
154    @Override
155    public ICUCollationKeyAnalyzer get() {
156        return new ICUCollationKeyAnalyzer(version, collator);
157    }
158}
