001package org.xbib.elasticsearch.index.analysis.icu;
002
003import com.ibm.icu.text.FilteredNormalizer2;
004import com.ibm.icu.text.Normalizer2;
005import com.ibm.icu.text.UnicodeSet;
006import org.apache.lucene.analysis.TokenStream;
007import org.apache.lucene.analysis.icu.ICUFoldingFilter;
008import org.elasticsearch.common.inject.Inject;
009import org.elasticsearch.common.inject.assistedinject.Assisted;
010import org.elasticsearch.common.settings.Settings;
011import org.elasticsearch.index.Index;
012import org.elasticsearch.index.analysis.AbstractTokenFilterFactory;
013import org.elasticsearch.index.settings.IndexSettings;
014
015
016/**
017 * Uses the {@link org.apache.lucene.analysis.icu.ICUFoldingFilter}.
018 * Applies foldings from UTR#30 Character Foldings.
019 * <p/>
020 * Can be filtered to handle certain characters in a specified way (see http://icu-project.org/apiref/icu4j/com/ibm/icu/text/UnicodeSet.html)
021 * E.g national chars that should be retained (filter : "[^åäöÅÄÖ]").
022 * <p/>
023 * <p>The <tt>unicodeSetFilter</tt> attribute can be used to provide the UniCodeSet for filtering.
024 */
025public class IcuFoldingTokenFilterFactory extends AbstractTokenFilterFactory {
026    private final String unicodeSetFilter;
027
028    @Inject
029    public IcuFoldingTokenFilterFactory(Index index, @IndexSettings Settings indexSettings, @Assisted String name, @Assisted Settings settings) {
030        super(index, indexSettings, name, settings);
031        this.unicodeSetFilter = settings.get("unicodeSetFilter");
032    }
033
034    @Override
035    public TokenStream create(TokenStream tokenStream) {
036        // The ICUFoldingFilter is in fact implemented as a ICUNormalizer2Filter.
037        // ICUFoldingFilter lacks a constructor for adding filtering so we implemement it here
038        if (unicodeSetFilter != null) {
039            Normalizer2 base = Normalizer2.getInstance(
040                    ICUFoldingFilter.class.getResourceAsStream("utr30.nrm"),
041                    "utr30", Normalizer2.Mode.COMPOSE);
042            UnicodeSet unicodeSet = new UnicodeSet(unicodeSetFilter);
043            unicodeSet.freeze();
044            Normalizer2 filtered = new FilteredNormalizer2(base, unicodeSet);
045            return new org.apache.lucene.analysis.icu.ICUNormalizer2Filter(tokenStream, filtered);
046        } else {
047            return new ICUFoldingFilter(tokenStream);
048        }
049    }
050}
