001package org.xbib.elasticsearch.index.analysis.icu;
002
003import org.elasticsearch.index.analysis.AnalysisModule;
004
005public class IcuAnalysisBinderProcessor extends AnalysisModule.AnalysisBinderProcessor {
006
007    @Override
008    public void processTokenizers(TokenizersBindings tokenizersBindings) {
009        tokenizersBindings.processTokenizer("icu_tokenizer", IcuTokenizerFactory.class);
010        tokenizersBindings.processTokenizer("icu_collation", IcuCollationTokenizerFactory.class);
011    }
012
013    @Override
014    public void processTokenFilters(TokenFiltersBindings tokenFiltersBindings) {
015        tokenFiltersBindings.processTokenFilter("icu_normalizer", IcuNormalizerTokenFilterFactory.class);
016        tokenFiltersBindings.processTokenFilter("icu_folding", IcuFoldingTokenFilterFactory.class);
017        tokenFiltersBindings.processTokenFilter("icu_transform", IcuTransformTokenFilterFactory.class);
018    }
019
020    @Override
021    public void processAnalyzers(AnalyzersBindings analyzersBindings) {
022        analyzersBindings.processAnalyzer("icu_collation", IcuCollationAnalyzerProvider.class);
023    }
024}
