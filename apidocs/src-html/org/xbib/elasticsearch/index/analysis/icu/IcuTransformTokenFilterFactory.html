020package org.xbib.elasticsearch.index.analysis.icu;
021
022import com.ibm.icu.text.Transliterator;
023import org.apache.lucene.analysis.TokenStream;
024import org.apache.lucene.analysis.icu.ICUTransformFilter;
025import org.elasticsearch.common.inject.Inject;
026import org.elasticsearch.common.inject.assistedinject.Assisted;
027import org.elasticsearch.common.settings.Settings;
028import org.elasticsearch.index.Index;
029import org.elasticsearch.index.analysis.AbstractTokenFilterFactory;
030import org.elasticsearch.index.settings.IndexSettings;
031
032
033/**
034 */
035public class IcuTransformTokenFilterFactory extends AbstractTokenFilterFactory {
036
037    private final String id;
038    private final int dir;
039    private final Transliterator transliterator;
040
041    @Inject
042    public IcuTransformTokenFilterFactory(Index index, @IndexSettings Settings indexSettings, @Assisted String name, @Assisted Settings settings) {
043        super(index, indexSettings, name, settings);
044        this.id = settings.get("id", "Null");
045        String s = settings.get("dir", "forward");
046        this.dir = "forward".equals(s) ? Transliterator.FORWARD : Transliterator.REVERSE;
047        this.transliterator = Transliterator.getInstance(id, dir);
048    }
049
050    @Override
051    public TokenStream create(TokenStream tokenStream) {
052        return new ICUTransformFilter(tokenStream, transliterator);
053    }
054}
