001package org.xbib.elasticsearch.index.analysis.icu;
002
003import com.ibm.icu.text.Transliterator;
004import org.apache.lucene.analysis.TokenStream;
005import org.apache.lucene.analysis.icu.ICUTransformFilter;
006import org.elasticsearch.common.inject.Inject;
007import org.elasticsearch.common.inject.assistedinject.Assisted;
008import org.elasticsearch.common.settings.Settings;
009import org.elasticsearch.index.Index;
010import org.elasticsearch.index.analysis.AbstractTokenFilterFactory;
011import org.elasticsearch.index.settings.IndexSettings;
012
013/**
014 */
015public class IcuTransformTokenFilterFactory extends AbstractTokenFilterFactory {
016
017    private final Transliterator transliterator;
018
019    @Inject
020    public IcuTransformTokenFilterFactory(Index index, @IndexSettings Settings indexSettings, @Assisted String name, @Assisted Settings settings) {
021        super(index, indexSettings, name, settings);
022        String id = settings.get("id", "Null");
023        String s = settings.get("dir", "forward");
024        int dir = "forward".equals(s) ? Transliterator.FORWARD : Transliterator.REVERSE;
025        this.transliterator = Transliterator.getInstance(id, dir);
026    }
027
028    @Override
029    public TokenStream create(TokenStream tokenStream) {
030        return new ICUTransformFilter(tokenStream, transliterator);
031    }
032}
