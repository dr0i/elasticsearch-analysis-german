020package org.xbib.elasticsearch.index.analysis.icu;
021
022import com.ibm.icu.text.Collator;
023import com.ibm.icu.text.RuleBasedCollator;
024import com.ibm.icu.util.ULocale;
025import org.apache.lucene.analysis.TokenStream;
026import org.apache.lucene.collation.ICUCollationKeyFilter;
027import org.elasticsearch.ElasticSearchIllegalArgumentException;
028import org.elasticsearch.common.inject.Inject;
029import org.elasticsearch.common.inject.assistedinject.Assisted;
030import org.elasticsearch.common.settings.Settings;
031import org.elasticsearch.env.Environment;
032import org.elasticsearch.env.FailedToResolveConfigException;
033import org.elasticsearch.index.Index;
034import org.elasticsearch.index.analysis.AbstractTokenFilterFactory;
035import org.elasticsearch.index.settings.IndexSettings;
036
037import java.io.IOException;
038
039/**
040 * An ICU based collation token filter. There are two ways to configure collation:
041 * <p/>
042 * <p>The first is simply specifying the locale (defaults to the default locale). The <tt>language</tt>
043 * parameter is the lowercase two-letter ISO-639 code. An additional <tt>country</tt> and <tt>variant</tt>
044 * can be provided.
045 * <p/>
046 * <p>The second option is to specify collation rules as defined in the <a href="http://www.icu-project.org/userguide/Collate_Customization.html">
047 * Collation customization</a> chapter in icu docs. The <tt>rules</tt> parameter can either embed the rules definition
048 * in the settings or refer to an external location (preferable located under the <tt>config</tt> location, relative to it).
049 */
050public class IcuCollationTokenFilterFactory extends AbstractTokenFilterFactory {
051
052    private final Collator collator;
053
054    @Inject
055    public IcuCollationTokenFilterFactory(Index index, @IndexSettings Settings indexSettings, Environment environment, @Assisted String name, @Assisted Settings settings) {
056        super(index, indexSettings, name, settings);
057
058        Collator collator;
059        String rules = settings.get("rules");
060        if (rules != null) {
061            FailedToResolveConfigException failureToResolve = null;
062            try {
063                rules = environment.resolveConfigAndLoadToString(rules);
064            } catch (FailedToResolveConfigException e) {
065                failureToResolve = e;
066            } catch (IOException e) {
067                throw new ElasticSearchIllegalArgumentException("Failed to load collation rules", e);
068            }
069            try {
070                collator = new RuleBasedCollator(rules);
071            } catch (Exception e) {
072                if (failureToResolve != null) {
073                    throw new ElasticSearchIllegalArgumentException("Failed to resolve collation rules location", failureToResolve);
074                } else {
075                    throw new ElasticSearchIllegalArgumentException("Failed to parse collation rules", e);
076                }
077            }
078        } else {
079            String language = settings.get("language");
080            if (language != null) {
081                ULocale locale;
082                String country = settings.get("country");
083                if (country != null) {
084                    String variant = settings.get("variant");
085                    if (variant != null) {
086                        locale = new ULocale(language, country, variant);
087                    } else {
088                        locale = new ULocale(language, country);
089                    }
090                } else {
091                    locale = new ULocale(language);
092                }
093                collator = Collator.getInstance(locale);
094            } else {
095                collator = Collator.getInstance();
096            }
097        }
098
099        // set the strength flag, otherwise it will be the default.
100        String strength = settings.get("strength");
101        if (strength != null) {
102            if (strength.equalsIgnoreCase("primary")) {
103                collator.setStrength(Collator.PRIMARY);
104            } else if (strength.equalsIgnoreCase("secondary")) {
105                collator.setStrength(Collator.SECONDARY);
106            } else if (strength.equalsIgnoreCase("tertiary")) {
107                collator.setStrength(Collator.TERTIARY);
108            } else if (strength.equalsIgnoreCase("quaternary")) {
109                collator.setStrength(Collator.QUATERNARY);
110            } else if (strength.equalsIgnoreCase("identical")) {
111                collator.setStrength(Collator.IDENTICAL);
112            } else {
113                throw new ElasticSearchIllegalArgumentException("Invalid strength: " + strength);
114            }
115        }
116
117        // set the decomposition flag, otherwise it will be the default.
118        String decomposition = settings.get("decomposition");
119        if (decomposition != null) {
120            if (decomposition.equalsIgnoreCase("no")) {
121                collator.setDecomposition(Collator.NO_DECOMPOSITION);
122            } else if (decomposition.equalsIgnoreCase("canonical")) {
123                collator.setDecomposition(Collator.CANONICAL_DECOMPOSITION);
124            } else {
125                throw new ElasticSearchIllegalArgumentException("Invalid decomposition: " + decomposition);
126            }
127        }
128
129        // expert options: concrete subclasses are always a RuleBasedCollator
130        RuleBasedCollator rbc = (RuleBasedCollator) collator;
131        String alternate = settings.get("alternate");
132        if (alternate != null) {
133            if (alternate.equalsIgnoreCase("shifted")) {
134                rbc.setAlternateHandlingShifted(true);
135            } else if (alternate.equalsIgnoreCase("non-ignorable")) {
136                rbc.setAlternateHandlingShifted(false);
137            } else {
138                throw new ElasticSearchIllegalArgumentException("Invalid alternate: " + alternate);
139            }
140        }
141
142        Boolean caseLevel = settings.getAsBoolean("caseLevel", null);
143        if (caseLevel != null) {
144            rbc.setCaseLevel(caseLevel);
145        }
146
147        String caseFirst = settings.get("caseFirst");
148        if (caseFirst != null) {
149            if (caseFirst.equalsIgnoreCase("lower")) {
150                rbc.setLowerCaseFirst(true);
151            } else if (caseFirst.equalsIgnoreCase("upper")) {
152                rbc.setUpperCaseFirst(true);
153            } else {
154                throw new ElasticSearchIllegalArgumentException("Invalid caseFirst: " + caseFirst);
155            }
156        }
157
158        Boolean numeric = settings.getAsBoolean("numeric", null);
159        if (numeric != null) {
160            rbc.setNumericCollation(numeric);
161        }
162
163        String variableTop = settings.get("variableTop");
164        if (variableTop != null) {
165            rbc.setVariableTop(variableTop);
166        }
167
168        Boolean hiraganaQuaternaryMode = settings.getAsBoolean("hiraganaQuaternaryMode", null);
169        if (hiraganaQuaternaryMode != null) {
170            rbc.setHiraganaQuaternary(hiraganaQuaternaryMode);
171        }
172
173        this.collator = collator;
174    }
175
176    @Override
177    public TokenStream create(TokenStream tokenStream) {
178        return new ICUCollationKeyFilter(tokenStream, collator);
179    }
180}
