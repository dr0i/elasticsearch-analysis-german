002package org.xbib.elasticsearch.index.analysis.icu;
003
004import com.ibm.icu.text.Collator;
005import com.ibm.icu.text.RuleBasedCollator;
006import com.ibm.icu.util.ULocale;
007import org.apache.lucene.analysis.TokenStream;
008import org.apache.lucene.collation.ICUCollationKeyFilter;
009import org.elasticsearch.ElasticsearchIllegalArgumentException;
010import org.elasticsearch.common.inject.Inject;
011import org.elasticsearch.common.inject.assistedinject.Assisted;
012import org.elasticsearch.common.settings.Settings;
013import org.elasticsearch.env.Environment;
014import org.elasticsearch.env.FailedToResolveConfigException;
015import org.elasticsearch.index.Index;
016import org.elasticsearch.index.analysis.AbstractTokenFilterFactory;
017import org.elasticsearch.index.settings.IndexSettings;
018
019import java.io.IOException;
020
021/**
022 * An ICU based collation token filter. There are two ways to configure collation:
023 * <p/>
024 * <p>The first is simply specifying the locale (defaults to the default locale). The <tt>language</tt>
025 * parameter is the lowercase two-letter ISO-639 code. An additional <tt>country</tt> and <tt>variant</tt>
026 * can be provided.
027 * <p/>
028 * <p>The second option is to specify collation rules as defined in the <a href="http://www.icu-project.org/userguide/Collate_Customization.html">
029 * Collation customization</a> chapter in icu docs. The <tt>rules</tt> parameter can either embed the rules definition
030 * in the settings or refer to an external location (preferable located under the <tt>config</tt> location, relative to it).
031 */
032public class IcuCollationTokenFilterFactory extends AbstractTokenFilterFactory {
033
034    private final Collator collator;
035
036    @Inject
037    public IcuCollationTokenFilterFactory(Index index, @IndexSettings Settings indexSettings, Environment environment, @Assisted String name, @Assisted Settings settings) {
038        super(index, indexSettings, name, settings);
039
040        Collator collator;
041        String rules = settings.get("rules");
042        if (rules != null) {
043            FailedToResolveConfigException failureToResolve = null;
044            try {
045                rules = environment.resolveConfigAndLoadToString(rules);
046            } catch (FailedToResolveConfigException e) {
047                failureToResolve = e;
048            } catch (IOException e) {
049                throw new ElasticsearchIllegalArgumentException("Failed to load collation rules", e);
050            }
051            try {
052                collator = new RuleBasedCollator(rules);
053            } catch (Exception e) {
054                if (failureToResolve != null) {
055                    throw new ElasticsearchIllegalArgumentException("Failed to resolve collation rules location", failureToResolve);
056                } else {
057                    throw new ElasticsearchIllegalArgumentException("Failed to parse collation rules", e);
058                }
059            }
060        } else {
061            String language = settings.get("language");
062            if (language != null) {
063                ULocale locale;
064                String country = settings.get("country");
065                if (country != null) {
066                    String variant = settings.get("variant");
067                    if (variant != null) {
068                        locale = new ULocale(language, country, variant);
069                    } else {
070                        locale = new ULocale(language, country);
071                    }
072                } else {
073                    locale = new ULocale(language);
074                }
075                collator = Collator.getInstance(locale);
076            } else {
077                collator = Collator.getInstance();
078            }
079        }
080
081        // set the strength flag, otherwise it will be the default.
082        String strength = settings.get("strength");
083        if (strength != null) {
084            if (strength.equalsIgnoreCase("primary")) {
085                collator.setStrength(Collator.PRIMARY);
086            } else if (strength.equalsIgnoreCase("secondary")) {
087                collator.setStrength(Collator.SECONDARY);
088            } else if (strength.equalsIgnoreCase("tertiary")) {
089                collator.setStrength(Collator.TERTIARY);
090            } else if (strength.equalsIgnoreCase("quaternary")) {
091                collator.setStrength(Collator.QUATERNARY);
092            } else if (strength.equalsIgnoreCase("identical")) {
093                collator.setStrength(Collator.IDENTICAL);
094            } else {
095                throw new ElasticsearchIllegalArgumentException("Invalid strength: " + strength);
096            }
097        }
098
099        // set the decomposition flag, otherwise it will be the default.
100        String decomposition = settings.get("decomposition");
101        if (decomposition != null) {
102            if (decomposition.equalsIgnoreCase("no")) {
103                collator.setDecomposition(Collator.NO_DECOMPOSITION);
104            } else if (decomposition.equalsIgnoreCase("canonical")) {
105                collator.setDecomposition(Collator.CANONICAL_DECOMPOSITION);
106            } else {
107                throw new ElasticsearchIllegalArgumentException("Invalid decomposition: " + decomposition);
108            }
109        }
110
111        // expert options: concrete subclasses are always a RuleBasedCollator
112        RuleBasedCollator rbc = (RuleBasedCollator) collator;
113        String alternate = settings.get("alternate");
114        if (alternate != null) {
115            if (alternate.equalsIgnoreCase("shifted")) {
116                rbc.setAlternateHandlingShifted(true);
117            } else if (alternate.equalsIgnoreCase("non-ignorable")) {
118                rbc.setAlternateHandlingShifted(false);
119            } else {
120                throw new ElasticsearchIllegalArgumentException("Invalid alternate: " + alternate);
121            }
122        }
123
124        Boolean caseLevel = settings.getAsBoolean("caseLevel", null);
125        if (caseLevel != null) {
126            rbc.setCaseLevel(caseLevel);
127        }
128
129        String caseFirst = settings.get("caseFirst");
130        if (caseFirst != null) {
131            if (caseFirst.equalsIgnoreCase("lower")) {
132                rbc.setLowerCaseFirst(true);
133            } else if (caseFirst.equalsIgnoreCase("upper")) {
134                rbc.setUpperCaseFirst(true);
135            } else {
136                throw new ElasticsearchIllegalArgumentException("Invalid caseFirst: " + caseFirst);
137            }
138        }
139
140        Boolean numeric = settings.getAsBoolean("numeric", null);
141        if (numeric != null) {
142            rbc.setNumericCollation(numeric);
143        }
144
145        String variableTop = settings.get("variableTop");
146        if (variableTop != null) {
147            rbc.setVariableTop(variableTop);
148        }
149
150        Boolean hiraganaQuaternaryMode = settings.getAsBoolean("hiraganaQuaternaryMode", null);
151        if (hiraganaQuaternaryMode != null) {
152            rbc.setHiraganaQuaternary(hiraganaQuaternaryMode);
153        }
154
155        this.collator = collator;
156    }
157
158    @Override
159    public TokenStream create(TokenStream tokenStream) {
160        // TODO use analyzer instead
161        return new ICUCollationKeyFilter(tokenStream, collator);
162    }
163}
