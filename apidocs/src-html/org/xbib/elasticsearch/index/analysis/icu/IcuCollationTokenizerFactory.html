001package org.xbib.elasticsearch.index.analysis.icu;
002
003import org.apache.lucene.analysis.Tokenizer;
004import org.elasticsearch.common.inject.Inject;
005import org.elasticsearch.common.inject.assistedinject.Assisted;
006import org.elasticsearch.common.settings.Settings;
007import org.elasticsearch.env.Environment;
008import org.elasticsearch.index.Index;
009import org.elasticsearch.index.analysis.AbstractTokenizerFactory;
010import org.elasticsearch.index.settings.IndexSettings;
011
012import java.io.Reader;
013
014public class IcuCollationTokenizerFactory extends AbstractTokenizerFactory {
015
016    private final Environment environment;
017
018    private final Settings settings;
019
020    @Inject
021    public IcuCollationTokenizerFactory(Environment environment,
022                                        Index index,
023                                        @IndexSettings Settings indexSettings,
024                                        @Assisted String name,
025                                        @Assisted Settings settings) {
026        super(index, indexSettings, name, settings);
027        this.environment = environment;
028        this.settings = settings;
029    }
030
031    @Override
032    public Tokenizer create(Reader reader) {
033        System.err.println("creating tokenizer from reader");
034        return new IcuCollationTokenizer(environment, settings, reader);
035    }
036
037}
