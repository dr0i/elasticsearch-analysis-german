001package org.xbib.elasticsearch.index.analysis.combo;
002
003import java.io.IOException;
004import java.io.Reader;
005import java.io.StringReader;
006
007/**
008 * A ReaderCloner specialized in duplicating Lucene's {@link org.apache.lucene.analysis.ReusableStringReader}.
009 * <p/>
010 * As this class is package private, this cloner has an additional function
011 * to perform an {@code instanceof} check for you.
012 * <p/>
013 * The implementation exploits the fact that ReusableStringReader has a package
014 * private field {@code String s}, storing the original content.
015 * It is therefore sensitive to Lucene implementation changes.
016 */
017public class ReusableStringReaderCloner implements ReaderCloneFactory.ReaderCloner<ReusableStringReader> {
018
019    private static java.lang.reflect.Field internalField;
020
021    private ReusableStringReader original;
022    private String originalContent;
023
024    static {
025        try {
026            internalField = ReusableStringReader.class.getDeclaredField("s");
027            internalField.setAccessible(true);
028        } catch (Exception ex) {
029            throw new IllegalArgumentException("Could not give accessibility to private \"str\" field of the given StringReader", ex);
030        }
031    }
032
033    /**
034     * Binds this ReaderCloner with the package-private {@link org.apache.lucene.analysis.ReusableStringReader} class
035     * into the {@link ReaderCloneFactory}, without giving access to the hidden class.
036     */
037    public static void registerCloner() {
038        ReaderCloneFactory.bindCloner(ReusableStringReader.class, ReusableStringReaderCloner.class);
039    }
040
041    /**
042     * @param originalReader Must pass the canHandleReader(Reader) test, otherwise an IllegalArgumentException will be thrown.
043     */
044    public void init(ReusableStringReader originalReader) throws IOException {
045        this.original = originalReader;
046        this.originalContent = null;
047        try {
048            this.originalContent = (String) internalField.get(original);
049        } catch (Exception ex) {
050            throw new IllegalArgumentException("Could not access private \"s\" field of the given org.apache.lucene.document.ReusableStringReader (actual class: " + original.getClass().getCanonicalName() + ")", ex);
051        }
052    }
053
054    /**
055     * First call will return the original Reader provided.
056     */
057    @Override
058    public Reader giveAClone() {
059        if (original != null) {
060            Reader rtn = original;
061            original = null; // no longer hold a reference
062            return rtn;
063        }
064        return new StringReader(originalContent);
065    }
066
067}
