002package org.xbib.elasticsearch.index.analysis.combo;
003
004import java.io.FilterReader;
005import java.io.Reader;
006import java.lang.reflect.Field;
007
008/**
009 * A {@link java.io.FilterReader} ReaderUnwrapper that
010 * returns the Reader wrapped inside the FilterReader
011 * (and all its subclasses)
012 */
013public class FilterReaderUnwrapper implements ReaderCloneFactory.ReaderUnwrapper<FilterReader> {
014
015    private static Field internalField;
016
017    static {
018        try {
019            internalField = FilterReader.class.getDeclaredField("in");
020            internalField.setAccessible(true);
021        } catch (Exception ex) {
022            throw new IllegalArgumentException("Could not give accessibility to private \"in\" field of the given FilterReader", ex);
023        }
024    }
025
026    public Reader unwrap(FilterReader originalReader) throws IllegalArgumentException {
027        try {
028            return (Reader) internalField.get(originalReader);
029        } catch (Exception ex) {
030            throw new IllegalArgumentException("Could not access private \"in\" field of the given FilterReader (actual class: " + originalReader.getClass().getCanonicalName() + ")", ex);
031        }
032    }
033
034}
