002package org.xbib.elasticsearch.index.analysis.combo;
003
004import org.elasticsearch.common.inject.Inject;
005import org.elasticsearch.common.inject.Injector;
006import org.elasticsearch.common.inject.assistedinject.Assisted;
007import org.elasticsearch.common.settings.Settings;
008import org.elasticsearch.index.Index;
009import org.elasticsearch.index.analysis.AbstractIndexAnalyzerProvider;
010import org.elasticsearch.index.settings.IndexSettings;
011
012public class ComboAnalyzerProvider extends AbstractIndexAnalyzerProvider<ComboAnalyzerWrapper> {
013
014    private final Injector injector;
015
016    private final Settings settings;
017
018    private final String name;
019
020    @Inject
021    ComboAnalyzerProvider(Index index, @IndexSettings Settings indexSettings, @Assisted String name, @Assisted Settings settings, Injector injector) {
022        super(index, indexSettings, name, settings);
023        this.injector = injector;
024        this.settings = settings;
025        this.name = name;
026    }
027
028    @Override
029    public ComboAnalyzerWrapper get() {
030        return new ComboAnalyzerWrapper(version, name, settings, injector);
031    }
032
033}
