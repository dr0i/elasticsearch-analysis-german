002package org.xbib.elasticsearch.index.analysis.combo;
003
004import java.io.BufferedReader;
005import java.io.Reader;
006import java.lang.reflect.Field;
007
008/**
009 * A {@link java.io.BufferedReader} ReaderUnwrapper that
010 * returns the Reader wrapped inside the BufferReader.
011 */
012public class BufferedReaderUnwrapper implements ReaderCloneFactory.ReaderUnwrapper<BufferedReader> {
013
014    private static Field internalField;
015
016    static {
017        try {
018            internalField = BufferedReader.class.getDeclaredField("in");
019            internalField.setAccessible(true);
020        } catch (Exception ex) {
021            throw new IllegalArgumentException("Could not give accessibility to private \"in\" field of the given BufferedReader", ex);
022        }
023    }
024
025    public Reader unwrap(BufferedReader originalReader) throws IllegalArgumentException {
026        try {
027            return (Reader) internalField.get(originalReader);
028        } catch (Exception ex) {
029            throw new IllegalArgumentException("Could not access private \"in\" field of the given BufferedReader (actual class: " + originalReader.getClass().getCanonicalName() + ")", ex);
030        }
031    }
032
033}
