001package org.xbib.elasticsearch.index.analysis.combo;
002
003import java.io.CharArrayReader;
004import java.io.IOException;
005import java.io.Reader;
006import java.lang.reflect.Field;
007
008/**
009 * A ReaderCloner specialized for CharArrayReader.
010 * <p/>
011 * The only efficient mean of retrieving the original content
012 * from a CharArrayReader is to use introspection and access the
013 * {@code private String str} field.
014 * <p/>
015 * Apart from being efficient, this code is also very sensitive
016 * to the used JVM implementation.
017 * If the introspection does not work, an {@link IllegalArgumentException}
018 * is thrown.
019 */
020public class CharArrayReaderCloner implements ReaderCloneFactory.ReaderCloner<CharArrayReader> {
021
022    private static Field internalField;
023
024    private CharArrayReader original;
025    private char[] originalContent;
026
027    static {
028        try {
029            internalField = CharArrayReader.class.getDeclaredField("buf");
030            internalField.setAccessible(true);
031        } catch (Exception ex) {
032            throw new IllegalArgumentException("Could not give accessibility to private \"buf\" field of the given CharArrayReader", ex);
033        }
034    }
035
036    public void init(CharArrayReader originalReader) throws IOException {
037        this.original = originalReader;
038        this.originalContent = null;
039        try {
040            this.originalContent = (char[]) internalField.get(original);
041        } catch (Exception ex) {
042            throw new IllegalArgumentException("Could not access private \"buf\" field of the given CharArrayReader (actual class: " + original.getClass().getCanonicalName() + ")", ex);
043        }
044    }
045
046    /**
047     * First call will return the original Reader provided.
048     */
049    public Reader giveAClone() {
050        if (original != null) {
051            Reader rtn = original;
052            original = null; // no longer hold a reference
053            return rtn;
054        }
055        return new CharArrayReader(originalContent);
056    }
057
058}
