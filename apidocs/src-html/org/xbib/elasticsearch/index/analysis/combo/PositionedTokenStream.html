002package org.xbib.elasticsearch.index.analysis.combo;
003
004import org.apache.lucene.analysis.TokenFilter;
005import org.apache.lucene.analysis.TokenStream;
006import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
007import org.apache.lucene.analysis.tokenattributes.PositionIncrementAttribute;
008
009import java.io.IOException;
010
011/**
012 * A {@link org.apache.lucene.analysis.TokenStream} wrapper that keeps track of
013 * the current term position of the given TokenStream,
014 * and defines a comparison order.
015 */
016public class PositionedTokenStream extends TokenFilter implements Comparable<PositionedTokenStream> {
017
018    // Attributes to track
019    private final OffsetAttribute offsetAttr;
020    private final PositionIncrementAttribute posAttr;
021    /**
022     * Position tracker.
023     */
024    private int position;
025
026    public PositionedTokenStream(TokenStream input) {
027        super(input);
028
029        // Force loading/adding these attributes
030        // won't do much bad if they're not read/written
031        offsetAttr = input.addAttribute(OffsetAttribute.class);
032        posAttr = input.addAttribute(PositionIncrementAttribute.class);
033
034        this.position = 0;
035    }
036
037    /**
038     * Returns the tracked current token position.
039     *
040     * @return The accumulated position increment attribute values.
041     */
042    public int getPosition() {
043        return position;
044    }
045
046    /*
047     * "TokenStream interface"
048     */
049
050    public final boolean incrementToken() throws IOException {
051        boolean rtn = input.incrementToken();
052        if (!rtn) {
053            position = Integer.MAX_VALUE;
054        }
055        // Track accumulated position
056        position += posAttr.getPositionIncrement();
057        return rtn;
058    }
059
060    public void end() throws IOException {
061        input.end();
062        position = 0;
063    }
064
065    public void reset() throws IOException {
066        input.reset();
067        position = 0;
068    }
069
070    public void close() throws IOException {
071        input.close();
072        position = 0;
073    }
074
075    /**
076     * Permit ordering by reading order: term position, then term offsets (start, then end).
077     */
078    @Override
079    public int compareTo(PositionedTokenStream that) {
080        // Nullity checks
081        if (that == null) {
082            return 1;
083        }
084        // Position checks
085        if (this.position != that.position) {
086            return this.position - that.position;
087        }
088        // TokenStream nullity checks
089        if (that.input == null) {
090            if (this.input == null) {
091                return 0;
092            } else {
093                return 1;
094            }
095        } else if (this.input == null) {
096            return -1;
097        }
098        // Order by reading order, using offsets
099        if (this.offsetAttr != null && that.offsetAttr != null) {
100            int a = this.offsetAttr.startOffset();
101            int b = that.offsetAttr.startOffset();
102            if (a != b) {
103                return a - b;
104            }
105            a = this.offsetAttr.endOffset();
106            b = that.offsetAttr.endOffset();
107            return a - b;
108        } else if (that.offsetAttr == null) {
109            if (this.offsetAttr == null) {
110                return 0;
111            }
112            return 1;
113        } else {
114            return -1;
115        }
116    }
117
118}
