002package org.xbib.elasticsearch.index.analysis.combo;
003
004import java.io.IOException;
005import java.io.Reader;
006import java.io.StringReader;
007
008/**
009 * Default, memory costly but generic implementation of a {@link java.io.Reader} duplicator.
010 * <p/>
011 * This implementation makes no assumption on the initial Reader.
012 * Therefore, only the read() functions are available to figure out
013 * what was the original content provided to the initial Reader.
014 * <p/>
015 * After having read and filled a buffer with the whole content,
016 * a String-based Reader implementation will be used and returned.
017 * <p/>
018 * This implementation is memory costly because the initial content is
019 * forcefully duplicated once. Moreover, buffer size growth may cost
020 * some more memory too.
021 *
022 * @author ofavre
023 */
024public class ReaderClonerDefaultImpl implements ReaderCloneFactory.ReaderCloner<Reader> {
025
026    public static final int DEFAULT_INITIAL_CAPACITY = 64 * 1024;
027    public static final int DEFAULT_READ_BUFFER_SIZE = 16 * 1024;
028
029    protected int initialCapacity;
030    protected int readBufferSize;
031
032    private String originalContent;
033
034    public ReaderClonerDefaultImpl() {
035        this(DEFAULT_INITIAL_CAPACITY, DEFAULT_READ_BUFFER_SIZE);
036    }
037
038    public ReaderClonerDefaultImpl(int initialCapacity) {
039        this(initialCapacity, DEFAULT_READ_BUFFER_SIZE);
040    }
041
042    /**
043     * Extracts the original content from a generic Reader instance
044     * by repeatedly calling {@link java.io.Reader#read(char[])} on it,
045     * feeding a {@link StringBuilder}.
046     *
047     * @param initialCapacity Initial StringBuilder capacity
048     * @param readBufferSize  Size of the char[] read buffer at each read() call
049     */
050    public ReaderClonerDefaultImpl(int initialCapacity, int readBufferSize) {
051        this.initialCapacity = initialCapacity;
052        this.readBufferSize = readBufferSize;
053    }
054
055    public void init(Reader originalReader) throws IOException {
056        this.originalContent = null;
057        StringBuilder sb;
058        if (initialCapacity < 0) {
059            sb = new StringBuilder();
060        } else {
061            sb = new StringBuilder(initialCapacity);
062        }
063        char[] buffer = new char[readBufferSize];
064        int read;
065        while ((read = originalReader.read(buffer)) != -1) {
066            sb.append(buffer, 0, read);
067        }
068        this.originalContent = sb.toString();
069        originalReader.close();
070    }
071
072    /**
073     * Returns a new {@link java.io.StringReader} instance,
074     * directly based on the extracted original content.
075     *
076     * @return A {@link java.io.StringReader}
077     */
078    @Override
079    public Reader giveAClone() {
080        return new StringReader(originalContent);
081    }
082
083}
