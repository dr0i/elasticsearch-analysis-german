002package org.xbib.elasticsearch.index.analysis.combo;
003
004import org.apache.lucene.analysis.TokenStream;
005import org.apache.lucene.util.Attribute;
006
007import java.io.IOException;
008import java.util.Iterator;
009import java.util.LinkedList;
010import java.util.List;
011
012/**
013 * This class can be used if the token attributes of a TokenStream
014 * are intended to be consumed more than once. It caches
015 * all token attribute states locally in a List.
016 * <p/>
017 * <P>CachingTokenStream implements the optional method
018 * {@link org.apache.lucene.analysis.TokenStream#reset()}, which repositions the
019 * stream to the first Token.
020 * <p/>
021 * <P>It differs from Lucene's {@link org.apache.lucene.analysis.CachingTokenFilter} in that it
022 * creates a new AttributeSource, unrelated to the provided input.
023 * This permits free reuse of the TokenStream, and after
024 * {@link #fillCache()} has been called, the underlying input
025 * TokenStream is available for garbage collection.
026 * {@link org.apache.lucene.analysis.CachingTokenFilter} has no such feature because it keeps
027 * the same {@link org.apache.lucene.util.AttributeSource} and if the source analyzer
028 * reuses the {@link org.apache.lucene.analysis.TokenStream} in a call of
029 * {@link org.apache.lucene.analysis.Analyzer#tokenStream(String, java.io.Reader)}, then a
030 * call on {@link org.apache.lucene.analysis.TokenStream#incrementToken()}
031 * will affect the current attribute state of the "cached" instance.
032 */
033public final class CachingTokenStream extends TokenStream {
034    protected List<State> cache = null;
035    protected Iterator<State> iterator = null;
036    protected State finalState;
037    protected TokenStream input;
038
039    /**
040     * Create a new CachingTokenStream around <code>input</code>,
041     * caching its token attributes, which can be replayed again
042     * after a call to {@link #reset()}.
043     */
044    public CachingTokenStream(TokenStream input) {
045        this.input = input;
046        Iterator<Class<? extends Attribute>> attrIter = this.input.getAttributeClassesIterator();
047        while (attrIter.hasNext()) {
048            addAttribute(attrIter.next());
049        }
050    }
051
052    public void fillCache() throws IOException {
053        if (cache != null) {
054            return;
055        }
056        cache = new LinkedList<State>();
057
058        while (input.incrementToken()) {
059            cache.add(input.captureState());
060        }
061        // capture final state
062        input.end();
063        finalState = input.captureState();
064        input = null; // don't hold on the input
065
066        iterator = cache.iterator();
067    }
068
069    @Override
070    public final boolean incrementToken() throws IOException {
071        // fill cache lazily
072        fillCache();
073
074        if (!iterator.hasNext()) {
075            // the cache is exhausted, return false
076            return false;
077        }
078        // Since the TokenFilter can be reset, the tokens need to be preserved as immutable.
079        restoreState(iterator.next());
080        return true;
081    }
082
083    @Override
084    public final void end() {
085        if (finalState != null) {
086            restoreState(finalState);
087        }
088    }
089
090    /**
091     * Rewinds the iterator to the beginning of the cached list.
092     * <p/>
093     * Note that this does not call reset() on the wrapped tokenstream ever, even
094     * the first time. You should reset() the inner tokenstream before wrapping
095     * it with CachingTokenStream.
096     */
097    @Override
098    public void reset() {
099        if (cache != null) {
100            iterator = cache.iterator();
101        }
102    }
103
104
105}
