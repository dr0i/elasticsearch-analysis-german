002package org.xbib.elasticsearch.index.analysis.combo;
003
004import org.apache.lucene.analysis.TokenStream;
005import org.apache.lucene.analysis.tokenattributes.PositionIncrementAttribute;
006import org.apache.lucene.util.Attribute;
007
008import java.io.IOException;
009import java.util.AbstractQueue;
010import java.util.Iterator;
011import java.util.PriorityQueue;
012
013/**
014 * A TokenStream combining the output of multiple sub-TokenStreams.
015 *
016 * This class copies the attributes from the last sub-TokenStream that
017 * was read from. If attributes are not uniform between sub-TokenStreams,
018 * extraneous attributes will stay untouched.
019 */
020public class ComboTokenStream extends TokenStream {
021
022    /**
023     * Whether or not to continue with the current TokenStream
024     * if it has multiple terms at same position, minimizing
025     * queue moves, or to enforce strict order (position, offsets)
026     */
027    static final boolean KEEP_STREAM_IF_SAME_POSITION = false;
028
029    private int lastPosition;
030    // Position tracked sub-TokenStreams
031    private final PositionedTokenStream[] positionedTokenStreams;
032    // Reading queue, using the reading order from PositionedTokenStream
033    private final AbstractQueue<PositionedTokenStream> readQueue;
034    // Flag for lazy initialization and reset
035    private boolean readQueueResetted;
036
037    public ComboTokenStream(TokenStream... tokenStreams) {
038        // Load the TokenStreams, track their position, and register their attributes
039        this.positionedTokenStreams = new PositionedTokenStream[tokenStreams.length];
040        for (int i = tokenStreams.length - 1; i >= 0; --i) {
041            if (tokenStreams[i] == null) {
042                continue;
043            }
044            this.positionedTokenStreams[i] = new PositionedTokenStream(tokenStreams[i]);
045            // Add each and every token seen in the current sub AttributeSource
046            Iterator<Class<? extends Attribute>> iterator = this.positionedTokenStreams[i].getAttributeClassesIterator();
047            while (iterator.hasNext()) {
048                addAttribute(iterator.next());
049            }
050        }
051        this.lastPosition = 0;
052        // Create an initially empty queue.
053        // It will be filled at first incrementToken() call, because
054        // it needs to call the same function on each sub-TokenStreams.
055        this.readQueue = new PriorityQueue<PositionedTokenStream>(tokenStreams.length);
056        readQueueResetted = false;
057    }
058
059    /*
060     * TokenStream multiplexed methods
061     */
062
063    @Override
064    public final boolean incrementToken() throws IOException {
065        clearAttributes();
066
067        // Fill the queue on first call
068        if (!readQueueResetted) {
069            readQueueResetted = true;
070            readQueue.clear();
071            for (PositionedTokenStream pts : positionedTokenStreams) {
072                if (pts == null) {
073                    continue;
074                }
075                // Read first token
076                pts.clearAttributes();
077                if (pts.incrementToken()) {
078                    // PositionedTokenStream.incrementToken() initialized internal
079                    // variables to perform proper ordering.
080                    // Therefore we can only add it to the queue now!
081                    readQueue.add(pts);
082                } // no token left (no token at all)
083            }
084        }
085
086        // Read from the first token
087        PositionedTokenStream toRead = readQueue.peek();
088        if (toRead == null) {
089            return false; // end of streams
090        }
091        // Look position to see if it will be increased, see usage a bit below
092        int pos = toRead.getPosition();
093
094        // Copy the current token attributes from the sub-TokenStream to our AttributeSource
095        restoreState(toRead.captureState());
096        // Override the PositionIncrementAttribute
097        this.getAttribute(PositionIncrementAttribute.class).setPositionIncrement(Math.max(0, pos - lastPosition));
098
099        // Prepare next read
100        // We did not remove the TokenStream from the queue yet,
101        // because if we have another token available at the same position,
102        // we can save a queue movement.
103        toRead.clearAttributes();
104        if (!toRead.incrementToken()) {
105            // No more token to read, remove from the queue
106            readQueue.poll();
107        } else {
108            // Check if token position changed
109            if (readQueue.size() > 1) {
110                // If yes, re-enter in the priority queue
111                readQueue.add(readQueue.poll());
112            }   // Otherwise, next call will continue with the same TokenStream (less queue movements)
113        }
114
115        lastPosition = pos;
116
117        return true;
118    }
119
120    @Override
121    public void end() throws IOException {
122        super.end();
123        lastPosition = 0;
124        // Apply on each sub-TokenStream
125        for (PositionedTokenStream pts : positionedTokenStreams) {
126            if (pts == null) {
127                continue;
128            }
129            pts.end();
130        }
131        readQueueResetted = false;
132        readQueue.clear();
133    }
134
135    @Override
136    public void reset() throws IOException {
137        super.reset();
138        clearAttributes();
139        lastPosition = 0;
140        // Apply on each sub-TokenStream
141        for (PositionedTokenStream pts : positionedTokenStreams) {
142            if (pts == null) {
143                continue;
144            }
145            pts.reset();
146        }
147        readQueueResetted = false;
148        readQueue.clear();
149    }
150
151    @Override
152    public void close() throws IOException {
153        super.close();
154        lastPosition = 0;
155        // Apply on each sub-TokenStream
156        for (PositionedTokenStream pts : positionedTokenStreams) {
157            if (pts == null) {
158                continue;
159            }
160            pts.close();
161        }
162        readQueueResetted = false;
163        readQueue.clear();
164    }
165
166}
