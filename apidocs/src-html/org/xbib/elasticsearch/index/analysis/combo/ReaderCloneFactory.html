003import java.io.BufferedReader;
004import java.io.CharArrayReader;
005import java.io.FilterReader;
006import java.io.IOException;
007import java.io.Reader;
008import java.io.StringReader;
009import java.lang.ref.WeakReference;
010import java.util.WeakHashMap;

012/**
013 * Duplicates {@link java.io.Reader}s in order to feed multiple consumers.
014 * <p/>
015 * This class registers multiple implementations, and tries to resolve which one to use,
016 * looking at the actual class of the Reader to clone, and matching with the most bond
017 * handled classes for each {@link ReaderCloner} implementation.
018 * <p/>
019 * By default, a few {@link java.io.Reader} implementations are handled, including the
020 * most used inside Lucene ({@link java.io.StringReader}), and a default, fallback implementation
021 * that merely reads all the available content, and creates a String out of it.
022 * <p/>
023 * Therefore you should understand the importance of having a proper implementation for
024 * any optimizable {@link java.io.Reader}. For instance, {@link StringReaderCloner} gains access
025 * to the underlying String in order to avoid copies. A generic BufferedReader
026 */
027public class ReaderCloneFactory {
028
029    /**
030     * Interface for a utility class, able to unwrap a {@link java.io.Reader}
031     * inside another {@link java.io.Reader}.
032     *
033     * @param <T> The base class handled.
034     */
035    public static interface ReaderUnwrapper<T extends Reader> {
036        /**
037         * Unwraps a {@link java.io.Reader} from another, simplifying an eventual chain.
038         */
039        public Reader unwrap(T originalReader) throws IllegalArgumentException;
040    }
041
042    /**
043     * Interface for a utility class, able to clone the content of a {@link java.io.Reader},
044     * possibly in an optimized way (such as gaining access to a package private field,
045     * or through reflection using {@link java.lang.reflect.Field#setAccessible(boolean)}).
046     *
047     * @param <T> The base class handled.
048     */
049    public static interface ReaderCloner<T extends Reader> {
050        /**
051         * Initialize or reinitialize the cloner with the given reader.
052         * The implementing class should have a default no arguments constructor.
053         * <p/>
054         * <P><B>Remark:</B> The given Reader is now controlled by this ReaderCloner, it may
055         * be closed during a call to this method, or it may be returned
056         * at first call to {@link #giveAClone()}.
057         *
058         * @see #giveAClone()
059         */
060        public void init(T originalReader) throws IOException;
061
062        /**
063         * Returns a new {@link java.io.Reader}.
064         * <P><B>Remark:</B> The returned Reader should be closed.
065         * The original Reader, if not consumed by the {@link #init(java.io.Reader)} method,
066         * should be returned at first call. Therefore it is important to
067         * call this method at least once, or to be prepared to face possible
068         * exceptions when closing the original Reader.
069         */
070        public Reader giveAClone();
071    }
072
073    /**
074     * Map storing the mapping between a handled class and a handling class, for {@link ReaderCloner}s
075     */
076    private static final WeakHashMap<Class<? extends Reader>, WeakReference<Class<? extends ReaderCloner>>> typeMap =
077            new WeakHashMap<Class<? extends Reader>, WeakReference<Class<? extends ReaderCloner>>>();
078    /**
079     * Map storing the mapping between a handled class and a handling instance, for {@link ReaderUnwrapper}s
080     */
081    private static final WeakHashMap<Class<? extends Reader>, ReaderUnwrapper> unwrapperTypeMap =
082            new WeakHashMap<Class<? extends Reader>, ReaderUnwrapper>();
083
084    /**
085     * Add the association between a (handled) class and its handling {@link ReaderCloner}.
086     *
087     * @param handledClass    The base class that is handled by clonerImplClass.
088     *                        Using this parameter, you can further restrict the usage of a more generic cloner.
089     * @param clonerImplClass The class of the associated cloner.
090     * @param <T>             The base handled class of the ReaderCloner.
091     * @return The previously associated ReaderCloner for the handledClass.
092     */
093    public static <T extends Reader> WeakReference<Class<? extends ReaderCloner>> bindCloner(
094            Class<? extends T> handledClass, Class<? extends ReaderCloner<T>> clonerImplClass) {
095        return typeMap.put(handledClass, new WeakReference<Class<? extends ReaderCloner>>(clonerImplClass));
096    }
097
098    /**
099     * Add the association between a (handled) class and its handling {@link ReaderUnwrapper} instance.
100     *
101     * @param handledClass  The base class that is handled by clonerImplClass.
102     *                      Using this parameter, you can further restrict the usage of a more generic cloner.
103     * @param unwrapperImpl The instance of the associated unwrapper.
104     * @param <T>           The base handled class of the ReaderUnwrapper.
105     * @return The previously associated ReaderUnwrapper instance for the handledClass.
106     */
107    public static <T extends Reader> ReaderUnwrapper bindUnwrapper(
108            Class<? extends T> handledClass, ReaderUnwrapper<T> unwrapperImpl) {
109        return unwrapperTypeMap.put(handledClass, unwrapperImpl);
110    }
111
112    /**
113     * Static initialization registering default associations
114     */
115    static {
116        // General purpose Reader handling
117        bindCloner(Reader.class, ReaderClonerDefaultImpl.class);
118        bindUnwrapper(BufferedReader.class, new BufferedReaderUnwrapper());
119        bindUnwrapper(FilterReader.class, new FilterReaderUnwrapper());
120        // Often used Java Readers
121        bindCloner(StringReader.class, StringReaderCloner.class); // very, very used inside Lucene
122        bindCloner(CharArrayReader.class, CharArrayReaderCloner.class);
123        // Lucene specific handling
124        ReusableStringReaderCloner.registerCloner();
125    }

127    /**
128     * (Expert) Returns the ReaderUnwrapper associated with the exact given class.
129     *
130     * @param forClass The handled class bond to the ReaderUnwrapper to return.
131     * @param <T>      The base handled class of the ReaderUnwrapper to return.
132     * @return The bond ReaderUnwrapper, or null.
133     */
134    public static <T extends Reader> ReaderUnwrapper<T> getUnwrapperStrict(Class<? extends T> forClass) {
135        return unwrapperTypeMap.get(forClass);
136    }
137
138    /**
139     * Returns the ReaderCloner associated with the exact given class.
140     *
141     * @param forClass The handled class bond to the ReaderCloner to return.
142     * @param <T>      The base handled class of the ReaderCloner to return.
143     * @return The bond ReaderCloner, or null.
144     */
145    public static <T extends Reader> ReaderCloner<T> getClonerStrict(Class<? extends T> forClass) {
146        WeakReference<Class<? extends ReaderCloner>> refClonerClass = typeMap.get(forClass);
147        if (refClonerClass != null) {
148            Class<? extends ReaderCloner> clazz = refClonerClass.get();
149            if (clazz != null) {
150                try {
151                    return (ReaderCloner<T>) clazz.newInstance();
152                } catch (Throwable ignored) {
153                }
154            }
155        }
156        return null;
157    }
158
159    /**
160     * (Advanced) Returns an initialized ReaderCloner, associated with the exact class of the given Reader.
161     * If the initialization fails, this function returns null.
162     *
163     * @param forReader The handled class bond to the ReaderCloner to return.
164     * @param <T>       The base handled class of the ReaderCloner to return.
165     * @return The bond, initialized ReaderCloner, or null.
166     */
167    public static <T extends Reader> ReaderCloner<T> getClonerStrict(T forReader) {
168        ReaderCloner<T> rtn = ReaderCloneFactory.<T>getClonerStrict((Class<? extends T>) forReader.getClass());
169        if (rtn != null) {
170            try {
171                rtn.init(forReader);
172            } catch (Throwable fail) {
173                return null;
174            }
175        }
176        return rtn;
177    }

179    /**
180     * (Advanced) Returns an initialized ReaderCloner, associated with the given base class, for the given Reader.
181     * If the initialization fails, this function returns null.
182     * <p/>
183     * The function first tries to match the exact class of forReader, and initialize the ReaderCloner.
184     * If no ReaderCloner or (tested second) ReaderUnwrapper matches, the resolution continues with the super class,
185     * until the baseClass is reached, and tested.
186     * <p/>
187     * If this process is not successful, <code>null</code> is returned.
188     *
189     * @param baseClass The baseClass, above which the resolution will not try to continue with the super class.
190     * @param forClass  The class to start with, should be the class of forReader (but the latter can be null, hence this parameter)
191     * @param forReader The Reader instance to return and initialize a ReaderCloner for. Can be null.
192     * @param <T>       The base handled class of the ReaderCloner to return
193     * @param <S>       The class of the given Reader to handle
194     * @return An initialized ReaderCloner suitable for the givenReader, or null.
195     */
196    public static <T extends Reader, S extends T> ReaderCloner<T> getCloner(Class<T> baseClass, Class<S> forClass, final S forReader) {
197        final Class<S> originalForClass = forClass;
198        // Loop through each super class
199        while (forClass != null) {
200            // Try first a matching cloner
201            ReaderCloner<T> cloner = ReaderCloneFactory.<T>getClonerStrict(forClass);
202            if (cloner != null) {
203                if (forReader != null) {
204                    try {
205                        cloner.init(forReader);
206                    } catch (Exception e) {
207                        //logger.debug("Error while initializing [{}]", e, cloner.getClass().getCanonicalName());
208                        cloner = null;
209                    }
210                }
211                if (cloner != null) {
212                    return cloner;
213                }
214            }
215            // Try then a matching unwrapper, for better suitability of the used cloner
216            if (forReader != null) {
217                ReaderUnwrapper<T> unwrapper = ReaderCloneFactory.<T>getUnwrapperStrict(forClass);
218                if (unwrapper != null) {
219                    try {
220                        // Recursive resolution
221                        Reader unwrapped = unwrapper.unwrap(forReader);
222                        if (unwrapped != null) {
223                            return (ReaderCloner<T>) ReaderCloneFactory.getCloner(Reader.class, (Class<Reader>) unwrapped.getClass(), unwrapped);
224                        }
225                    } catch (Throwable ignore) {
226                        // in case of errors, simply continue the began process and forget about this failed attempt
227                        //logger.debug("Error while cloning [{}] with [{}]", ignore, unwrapper.getClass().getCanonicalName(), cloner.getClass().getCanonicalName());
228                    }
229                }
230            }
231            // Continue resolution with super class...
232            Class clazz = forClass.getSuperclass();
233            // ... checking ancestry with the given base class
234            if (baseClass.isAssignableFrom(clazz)) {
235                forClass = clazz;
236            } else {
237                forClass = null;
238            }
239        }
240        //if (forReader != null)
241        //logger.debug("Could not find a suitable ReaderCloner for [{}]", forReader.getClass().getCanonicalName());
242        //else
243        //logger.debug("Could not find a suitable ReaderCloner for class [{}]", originalForClass.getCanonicalName());
244        return null;
245    }

247    /**
248     * Returns a ReaderCloner suitable for handling general <code>S</code>s instances (inheriting <code>T</code>, itself
249     * inheriting {@link java.io.Reader}).
250     * <p/>
251     * Resolution starts on <code>forClass</code> (<code>S</code>), and does not go further than <code>baseClass</code>.
252     * <p/>
253     * Not all optimizations can be ran, like unwrapping and failing initialization fallback.
254     * However, for standard cases, when performance is really critical,
255     * using this function can reduce a possible resolution overhead
256     * because ReaderCloner are reusable.
257     *
258     * @param baseClass The baseClass, above which the resolution will not try to continue with the super class.
259     * @param forClass  The class to start with, should be the class of forReader (but the latter can be null, hence this parameter)
260     * @param <T>       The base handled class of the ReaderCloner to return
261     * @param <S>       The class of the given Reader to handle
262     * @return An uninitialized ReaderCloner suitable for any T Readers, or null.
263     */
264    public static <T extends Reader, S extends T> ReaderCloner<T> getCloner(Class<T> baseClass, Class<S> forClass) {
265        return ReaderCloneFactory.<T, S>getCloner(baseClass, forClass, null);
266    }

268    /**
269     * Returns a ReaderCloner suitable for handling general <code>S</code>s instances (inheriting {@link java.io.Reader}).
270     * <p/>
271     * Calls <code>ReaderCloneFactory.<Reader,S>getCloner(Reader.class, forClass, (S)null)</code>.
272     * <p/>
273     * Not all optimizations can be ran, like unwrapping and failing initialization fallback.
274     * However, for standard cases, when performance is really critical,
275     * using this function can reduce a possible resolution overhead
276     * because ReaderCloner are reusable.
277     *
278     * @param forClass The class to start with, should be the class of forReader (but the latter can be null, hence this parameter)
279     * @param <S>      The class of the given Reader to handle
280     * @return An uninitialized ReaderCloner suitable for any <code>S</code>, or null.
281     */
282    public static <S extends Reader> ReaderCloner<Reader> getCloner(Class<S> forClass) {
283        return ReaderCloneFactory.<Reader, S>getCloner(Reader.class, forClass, null);
284    }

286    /**
287     * Returns an initialized ReaderCloner, for the given Reader.
288     * <p/>
289     * Calls <code>ReaderCloneFactory.<Reader, S>getCloner(Reader.class, (Class<S>)forReader.getClass(), forReader)</code>.
290     * If <code>forReader</code> is <code>null</code>, works as {@link ReaderCloneFactory#getGenericCloner()}.
291     *
292     * @param forReader The Reader instance to return and initialize a ReaderCloner for. Can be null.
293     * @param <S>       The class of the given Reader
294     * @return An initialized ReaderCloner suitable for given Reader, or null.
295     */
296    public static <S extends Reader> ReaderCloner<Reader> getCloner(S forReader) {
297        if (forReader != null) {
298            return ReaderCloneFactory.<Reader, S>getCloner(Reader.class, (Class<S>) forReader.getClass(), forReader);
299        } else {
300            return ReaderCloneFactory.getGenericCloner();
301        }
302    }
303
304    /**
305     * Returns a {@link ReaderCloner} suitable for any {@link java.io.Reader} instance.
306     */
307    public static ReaderCloner<Reader> getGenericCloner() {
308        return ReaderCloneFactory.getCloner(Reader.class, Reader.class, null);
309    }
