001package org.xbib.elasticsearch.index.analysis.baseform;
002
003import java.util.Collections;
004import java.util.Set;
005
006/**
007 * An FSA with constant-size arc representation produced directly
008 * by {@link FSABuilder}.
009 *
010 * @see FSABuilder
011 */
012public final class ConstantArcSizeFSA extends FSA {
013    /**
014     * Size of the target address field (constant for the builder).
015     */
016    public final static int TARGET_ADDRESS_SIZE = 4;
017
018    /**
019     * Size of the flags field (constant for the builder).
020     */
021    public final static int FLAGS_SIZE = 1;
022
023    /**
024     * Size of the label field (constant for the builder).
025     */
026    public final static int LABEL_SIZE = 1;
027
028    /**
029     * Size of a single arc structure.
030     */
031    public final static int ARC_SIZE = FLAGS_SIZE + LABEL_SIZE + TARGET_ADDRESS_SIZE;
032
033    /**
034     * Offset of the flags field inside an arc.
035     */
036    public final static int FLAGS_OFFSET = 0;
037
038    /**
039     * Offset of the label field inside an arc.
040     */
041    public final static int LABEL_OFFSET = FLAGS_SIZE;
042
043    /**
044     * Offset of the address field inside an arc.
045     */
046    public final static int ADDRESS_OFFSET = LABEL_OFFSET + LABEL_SIZE;
047
048    /**
049     * A dummy address of the terminal state.
050     */
051    final static int TERMINAL_STATE = 0;
052
053    /**
054     * An arc flag indicating the target node of an arc corresponds to a final
055     * state.
056     */
057    public final static int BIT_ARC_FINAL = 1 << 1;
058
059    /**
060     * An arc flag indicating the arc is last within its state.
061     */
062    public final static int BIT_ARC_LAST = 1 << 0;
063
064    /**
065     * An epsilon state. The first and only arc of this state points either
066     * to the root or to the terminal state, indicating an empty automaton.
067     */
068    private final int epsilon;
069
070    /**
071     * FSA data, serialized as a byte array.
072     */
073    private final byte[] data;
074
075    /**
076     * @param data FSA data. There must be no trailing bytes after the last state.
077     */
078    ConstantArcSizeFSA(byte[] data, int epsilon) {
079        assert epsilon == 0 : "Epsilon is not zero?";
080
081        this.epsilon = epsilon;
082        this.data = data;
083    }
084
085    @Override
086    public int getRootNode() {
087        return getEndNode(getFirstArc(epsilon));
088    }
089
090    @Override
091    public int getFirstArc(int node) {
092        return node;
093    }
094
095    @Override
096    public int getArc(int node, byte label) {
097        for (int arc = getFirstArc(node); arc != 0; arc = getNextArc(arc)) {
098            if (getArcLabel(arc) == label) {
099                return arc;
100            }
101        }
102        return 0;
103    }
104
105    @Override
106    public int getNextArc(int arc) {
107        if (isArcLast(arc)) {
108            return 0;
109        }
110        return arc + ARC_SIZE;
111    }
112
113    @Override
114    public byte getArcLabel(int arc) {
115        return data[arc + LABEL_OFFSET];
116    }
117
118    /**
119     * Fills the target state address of an arc.
120     */
121    private int getArcTarget(int arc) {
122        arc += ADDRESS_OFFSET;
123        return (data[arc]) << 24 |
124                (data[arc + 1] & 0xff) << 16 |
125                (data[arc + 2] & 0xff) << 8 |
126                (data[arc + 3] & 0xff);
127    }
128
129    @Override
130    public boolean isArcFinal(int arc) {
131        return (data[arc + FLAGS_OFFSET] & BIT_ARC_FINAL) != 0;
132    }
133
134    @Override
135    public boolean isArcTerminal(int arc) {
136        return getArcTarget(arc) == 0;
137    }
138
139    private boolean isArcLast(int arc) {
140        return (data[arc + FLAGS_OFFSET] & BIT_ARC_LAST) != 0;
141    }
142
143    @Override
144    public int getEndNode(int arc) {
145        return getArcTarget(arc);
146    }
147
148    @Override
149    public Set<FSAFlags> getFlags() {
150        return Collections.emptySet();
151    }
152}
