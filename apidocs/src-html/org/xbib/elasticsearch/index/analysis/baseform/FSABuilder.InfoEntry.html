003import java.util.Arrays;
004import java.util.Comparator;
005import java.util.Map;

008import static org.xbib.elasticsearch.index.analysis.baseform.ConstantArcSizeFSA.ADDRESS_OFFSET;
009import static org.xbib.elasticsearch.index.analysis.baseform.ConstantArcSizeFSA.ARC_SIZE;
010import static org.xbib.elasticsearch.index.analysis.baseform.ConstantArcSizeFSA.BIT_ARC_FINAL;
011import static org.xbib.elasticsearch.index.analysis.baseform.ConstantArcSizeFSA.BIT_ARC_LAST;
012import static org.xbib.elasticsearch.index.analysis.baseform.ConstantArcSizeFSA.FLAGS_OFFSET;
013import static org.xbib.elasticsearch.index.analysis.baseform.ConstantArcSizeFSA.LABEL_OFFSET;
014import static org.xbib.elasticsearch.index.analysis.baseform.ConstantArcSizeFSA.TARGET_ADDRESS_SIZE;
015import static org.xbib.elasticsearch.index.analysis.baseform.ConstantArcSizeFSA.TERMINAL_STATE;

017/**
018 * Fast, memory-conservative finite state automaton builder, returning a
019 * byte-serialized {@link ConstantArcSizeFSA} (a tradeoff between construction
020 * speed and memory consumption).
021 */
022public final class FSABuilder {
023    /**
024     * Debug and information constants.
025     *
026     * @see FSABuilder#getInfo()
027     */
028    public enum InfoEntry {
029        SERIALIZATION_BUFFER_SIZE("Serialization buffer size"),
030        SERIALIZATION_BUFFER_REALLOCATIONS("Serialization buffer reallocs"),
031        CONSTANT_ARC_AUTOMATON_SIZE("Constant arc FSA size"),
032        MAX_ACTIVE_PATH_LENGTH("Max active path"),
033        STATE_REGISTRY_TABLE_SLOTS("Registry hash slots"),
034        STATE_REGISTRY_SIZE("Registry hash entries"),
035        ESTIMATED_MEMORY_CONSUMPTION_MB("Estimated mem consumption (MB)");
036
037        private final String stringified;
038
039        InfoEntry(String stringified) {
040            this.stringified = stringified;
041        }
042
043        @Override
044        public String toString() {
045            return stringified;
046        }
047    }
048
049    /**
050     * Comparator comparing full byte arrays consistently with
051     * {@link #compare(byte[], int, int, byte[], int, int)}.
052     */
053    public static final Comparator<byte[]> LEXICAL_ORDERING = new Comparator<byte[]>() {
054        public int compare(byte[] o1, byte[] o2) {
055            return FSABuilder.compare(o1, 0, o1.length, o2, 0, o2.length);
056        }
057    };
058
059    /**
060     * A megabyte.
061     */
062    private final static int MB = 1024 * 1024;
063
064    /**
065     * Internal serialized FSA buffer expand ratio.
066     */
067    private final static int BUFFER_GROWTH_SIZE = 5 * MB;
068
069    /**
070     * Maximum number of labels from a single state.
071     */
072    private final static int MAX_LABELS = 256;
073
074    /**
075     * Internal serialized FSA buffer expand ratio.
076     */
077    private final int bufferGrowthSize;
078
079    /**
080     * Holds serialized and mutable states. Each state is a sequential list of
081     * arcs, the last arc is marked with {@link ConstantArcSizeFSA#BIT_ARC_LAST}.
082     */
083    private byte[] serialized = new byte[0];
084
085    /**
086     * Number of bytes already taken in {@link #serialized}. Start from 1 to
087     * keep 0 a sentinel value (for the hash set and final state).
088     */
089    private int size;
090
091    /**
092     * States on the "active path" (still mutable). Values are addresses of each
093     * state's first arc.
094     */
095    private int[] activePath = new int[0];
096
097    /**
098     * Current length of the active path.
099     */
100    private int activePathLen;
101
102    /**
103     * The next offset at which an arc will be added to the given state on
104     * {@link #activePath}.
105     */
106    private int[] nextArcOffset = new int[0];
107
108    /**
109     * Root state. If negative, the automaton has been built already and cannot be extended.
110     */
111    private int root;
112
113    /**
114     * An epsilon state. The first and only arc of this state points either
115     * to the root or to the terminal state, indicating an empty automaton.
116     */
117    private int epsilon;
118
119    /**
120     * Hash set of state addresses in {@link #serialized}, hashed by
121     * {@link #hash(int, int)}. Zero reserved for an unoccupied slot.
122     */
123    private int[] hashSet = new int[2];
124
125    /**
126     * Number of entries currently stored in {@link #hashSet}.
127     */
128    private int hashSize = 0;
129
130    /**
131     * Previous sequence added to the automaton in {@link #add(byte[], int, int)}. Used in assertions only.
132     */
133    private byte[] previous;
134
135    /**
136     * Information about the automaton and its compilation.
137     */
138    private Map<InfoEntry, Object> info;
139
140    /**
141     * {@link #previous} sequence's length, used in assertions only.
142     */
143    private int previousLength;
144
145    /** */
146    public FSABuilder() {
147        this(BUFFER_GROWTH_SIZE);
148    }
149
150    /** */
151    public FSABuilder(int bufferGrowthSize) {
152        this.bufferGrowthSize = Math.max(bufferGrowthSize, ARC_SIZE * MAX_LABELS);
153        // Allocate epsilon state.
154        epsilon = allocateState(1);
155        serialized[epsilon + FLAGS_OFFSET] |= BIT_ARC_LAST;
156        // Allocate root, with an initial empty set of output arcs.
157        expandActivePath(1);
158        root = activePath[0];
159    }
160
161    /**
162     * Add a single sequence of bytes to the FSA. The input must be lexicographically greater
163     * than any previously added sequence.
164     */
165    public void add(byte[] sequence, int start, int len) {
166        assert serialized != null : "Automaton already built.";
167        assert previous == null || len == 0 || compare(previous, 0, previousLength, sequence, start, len) <= 0 :
168                "Input must be sorted: "
169                        + Arrays.toString(previous) + " >= "
170                        + Arrays.toString(sequence);
171        assert setPrevious(sequence, start, len);
172        // Determine common prefix length.
173        final int commonPrefix = commonPrefix(sequence, start, len);
174        // Make room for extra states on active path, if needed.
175        expandActivePath(len);
176        // Freeze all the states after the common prefix.
177        for (int i = activePathLen - 1; i > commonPrefix; i--) {
178            final int frozenState = freezeState(i);
179            setArcTarget(nextArcOffset[i - 1] - ARC_SIZE, frozenState);
180            nextArcOffset[i] = activePath[i];
181        }
182        // Create arcs to new suffix states.
183        for (int i = commonPrefix + 1, j = start + commonPrefix; i <= len; i++) {
184            final int p = nextArcOffset[i - 1];
185            serialized[p + FLAGS_OFFSET] = (byte) (i == len ? BIT_ARC_FINAL : 0);
186            serialized[p + LABEL_OFFSET] = sequence[j++];
187            setArcTarget(p, i == len ? TERMINAL_STATE : activePath[i]);
188            nextArcOffset[i - 1] = p + ARC_SIZE;
189        }
190        // Save last sequence's length so that we don't need to calculate it again.
191        this.activePathLen = len;
192    }
193
194    /**
195     * Number of serialization buffer reallocations.
196     */
197    private int serializationBufferReallocations;
198
199    /**
200     * Complete the automaton.
201     */
202    public FSA complete() {
203        add(new byte[0], 0, 0);
204        if (nextArcOffset[0] - activePath[0] == 0) {
205            // An empty FSA.
206            setArcTarget(epsilon, TERMINAL_STATE);
207        } else {
208            // An automaton with at least a single arc from root.
209            root = freezeState(0);
210            setArcTarget(epsilon, root);
211        }
212        info = new TreeMap<InfoEntry, Object>();
213        info.put(InfoEntry.SERIALIZATION_BUFFER_SIZE, serialized.length);
214        info.put(InfoEntry.SERIALIZATION_BUFFER_REALLOCATIONS, serializationBufferReallocations);
215        info.put(InfoEntry.CONSTANT_ARC_AUTOMATON_SIZE, size);
216        info.put(InfoEntry.MAX_ACTIVE_PATH_LENGTH, activePath.length);
217        info.put(InfoEntry.STATE_REGISTRY_TABLE_SLOTS, hashSet.length);
218        info.put(InfoEntry.STATE_REGISTRY_SIZE, hashSize);
219        info.put(InfoEntry.ESTIMATED_MEMORY_CONSUMPTION_MB,
220                (this.serialized.length + this.hashSet.length * 4) / (double) MB);
221        final FSA fsa = new ConstantArcSizeFSA(Arrays.copyOf(this.serialized, this.size), epsilon);
222        this.serialized = null;
223        this.hashSet = null;
224        return fsa;
225    }

227    /**
228     * Build a minimal, deterministic automaton from a sorted list of byte sequences.
229     */
230    public static FSA build(byte[][] input) {
231        final FSABuilder builder = new FSABuilder();
232        for (byte[] chs : input) {
233            builder.add(chs, 0, chs.length);
234        }
235        return builder.complete();
236    }
237
238    /**
239     * Build a minimal, deterministic automaton from an iterable list of byte sequences.
240     */
241    public static FSA build(Iterable<byte[]> input) {
242        final FSABuilder builder = new FSABuilder();
243        for (byte[] chs : input) {
244            builder.add(chs, 0, chs.length);
245        }
246        return builder.complete();
247    }

249    /**
250     * Return various statistics concerning the FSA and its compilation.
251     */
252    public Map<InfoEntry, Object> getInfo() {
253        return info;
254    }

256    /**
257     * Is this arc the state's last?
258     */
259    private boolean isArcLast(int arc) {
260        return (serialized[arc + FLAGS_OFFSET] & BIT_ARC_LAST) != 0;
261    }
262
263    /**
264     * Is this arc final?
265     */
266    private boolean isArcFinal(int arc) {
267        return (serialized[arc + FLAGS_OFFSET] & BIT_ARC_FINAL) != 0;
268    }
269
270    /**
271     * Get label's arc.
272     */
273    private byte getArcLabel(int arc) {
274        return serialized[arc + LABEL_OFFSET];
275    }
276
277    /**
278     * Fills the target state address of an arc.
279     */
280    private void setArcTarget(int arc, int state) {
281        arc += ADDRESS_OFFSET + TARGET_ADDRESS_SIZE;
282        for (int i = 0; i < TARGET_ADDRESS_SIZE; i++) {
283            serialized[--arc] = (byte) state;
284            state >>>= 8;
285        }
286    }
287
288    /**
289     * Returns the address of an arc.
290     */
291    private int getArcTarget(int arc) {
292        arc += ADDRESS_OFFSET;
293        return (serialized[arc]) << 24 |
294                (serialized[arc + 1] & 0xff) << 16 |
295                (serialized[arc + 2] & 0xff) << 8 |
296                (serialized[arc + 3] & 0xff);
297    }
298
299    /**
300     * @return The number of common prefix characters with the previous
301     * sequence.
302     */
303    private int commonPrefix(byte[] sequence, int start, int len) {
304        // Empty root state case.
305        final int max = Math.min(len, activePathLen);
306        int i;
307        for (i = 0; i < max; i++) {
308            final int lastArc = nextArcOffset[i] - ARC_SIZE;
309            if (sequence[start++] != getArcLabel(lastArc)) {
310                break;
311            }
312        }
313        return i;
314    }
315
316    /**
317     * Freeze a state: try to find an equivalent state in the interned states
318     * dictionary first, if found, return it, otherwise, write the mutable
319     * state at <code>activePathIndex</code> and return it.
320     */
321    private int freezeState(final int activePathIndex) {
322        final int start = activePath[activePathIndex];
323        final int end = nextArcOffset[activePathIndex];
324        final int len = end - start;
325        // Set the last arc flag on the current active path's state.
326        serialized[end - ARC_SIZE + FLAGS_OFFSET] |= BIT_ARC_LAST;
327        // Try to locate a state with an identical content in the hash set.
328        final int bucketMask = (hashSet.length - 1);
329        int slot = hash(start, len) & bucketMask;
330        for (int i = 0; ; ) {
331            int state = hashSet[slot];
332            if (state == 0) {
333                state = hashSet[slot] = serialize(activePathIndex);
334                if (++hashSize > hashSet.length / 2) {
335                    expandAndRehash();
336                }
337                return state;
338            } else if (equivalent(state, start, len)) {
339                return state;
340            }
341
342            slot = (slot + (++i)) & bucketMask;
343        }
344    }
345
346    /**
347     * Reallocate and rehash the hash set.
348     */
349    private void expandAndRehash() {
350        final int[] newHashSet = new int[hashSet.length * 2];
351        final int bucketMask = (newHashSet.length - 1);
352        for (int j = 0; j < hashSet.length; j++) {
353            final int state = hashSet[j];
354            if (state > 0) {
355                int slot = hash(state, stateLength(state)) & bucketMask;
356                for (int i = 0; newHashSet[slot] > 0; ) {
357                    slot = (slot + (++i)) & bucketMask;
358                }
359                newHashSet[slot] = state;
360            }
361        }
362        this.hashSet = newHashSet;
363    }
364
365    /**
366     * The total length of the serialized state data (all arcs).
367     */
368    private int stateLength(int state) {
369        int arc = state;
370        while (!isArcLast(arc)) {
371            arc += ARC_SIZE;
372        }
373        return arc - state + ARC_SIZE;
374    }
375
376    /**
377     * Return <code>true</code> if two regions in {@link #serialized} are identical.
378     */
379    private boolean equivalent(int start1, int start2, int len) {
380        if (start1 + len > size || start2 + len > size) {
381            return false;
382        }
383        while (len-- > 0) {
384            if (serialized[start1++] != serialized[start2++]) {
385                return false;
386            }
387        }
388        return true;
389    }
390
391    /**
392     * Serialize a given state on the active path.
393     */
394    private int serialize(final int activePathIndex) {
395        expandBuffers();
396        final int newState = size;
397        final int start = activePath[activePathIndex];
398        final int len = nextArcOffset[activePathIndex] - start;
399        System.arraycopy(serialized, start, serialized, newState, len);
400        size += len;
401        return newState;
402    }
403
404    /**
405     * Hash code of a fragment of {@link #serialized} array.
406     */
407    private int hash(int start, int byteCount) {
408        assert byteCount % ARC_SIZE == 0 : "not an arc multiply?";
409        int h = 0;
410        for (int arcs = byteCount / ARC_SIZE; --arcs >= 0; start += ARC_SIZE) {
411            h = 17 * h + getArcLabel(start);
412            h = 17 * h + getArcTarget(start);
413            if (isArcFinal(start)) {
414                h += 17;
415            }
416        }
417        return h;
418    }
419
420    /**
421     * Append a new mutable state to the active path.
422     */
423    private void expandActivePath(int size) {
424        if (activePath.length < size) {
425            final int p = activePath.length;
426            activePath = Arrays.copyOf(activePath, size);
427            nextArcOffset = Arrays.copyOf(nextArcOffset, size);
428            for (int i = p; i < size; i++) {
429                nextArcOffset[i] = activePath[i] =
430                        allocateState(/* assume max labels count */ MAX_LABELS);
431            }
432        }
433    }
434
435    /**
436     * Expand internal buffers for the next state.
437     */
438    private void expandBuffers() {
439        if (this.serialized.length < size + ARC_SIZE * MAX_LABELS) {
440            serialized = Arrays.copyOf(serialized, serialized.length + bufferGrowthSize);
441            serializationBufferReallocations++;
442        }
443    }
444
445    /**
446     * Allocate space for a state with the given number of outgoing labels.
447     *
448     * @return state offset
449     */
450    private int allocateState(int labels) {
451        expandBuffers();
452        final int state = size;
453        size += labels * ARC_SIZE;
454        return state;
455    }
456
457    /**
458     * Copy <code>current</code> into an internal buffer.
459     */
460    private boolean setPrevious(byte[] sequence, int start, int length) {
461        if (previous == null || previous.length < length) {
462            previous = new byte[length];
463        }
464        System.arraycopy(sequence, start, previous, 0, length);
465        previousLength = length;
466        return true;
467    }

469    /**
470     * Lexicographic order of input sequences. By default, consistent with the "C" sort
471     * (absolute value of bytes, 0-255).
472     */
473    public static int compare(byte[] s1, int start1, int lens1,
474                              byte[] s2, int start2, int lens2) {
475        final int max = Math.min(lens1, lens2);
476        for (int i = 0; i < max; i++) {
477            final byte c1 = s1[start1++];
478            final byte c2 = s2[start2++];
479            if (c1 != c2) {
480                return (c1 & 0xff) - (c2 & 0xff);
481            }
482        }
483        return lens1 - lens2;
484    }
