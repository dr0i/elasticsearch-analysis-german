001package org.xbib.elasticsearch.index.analysis.baseform;
002
003/**
004 * A matching result returned from {@link FSATraversal}.
005 *
006 * @see FSATraversal
007 */
008public final class MatchResult {
009    /**
010     * The automaton has exactly one match for the input sequence.
011     */
012    public static final int EXACT_MATCH = 0;
013
014    /**
015     * The automaton has no match for the input sequence.
016     */
017    public static final int NO_MATCH = -1;
018
019    /**
020     * The automaton contains a prefix of the input sequence. That is:
021     * one of the input sequences used to build the automaton is a
022     * prefix of the input sequence that is shorter than the sequence.
023     * <p/>
024     * <p>{@link MatchResult#index} will contain an index of the
025     * first character of the input sequence not present in the
026     * dictionary.</p>
027     */
028    public static final int AUTOMATON_HAS_PREFIX = -3;
029
030    /**
031     * The sequence is a prefix of at least one sequence in the automaton.
032     * {@link MatchResult#node} returns the node from which all sequences
033     * with the given prefix start in the automaton.
034     */
035    public static final int SEQUENCE_IS_A_PREFIX = -4;
036
037    /**
038     * One of the match kind constants defined in this class.
039     *
040     * @see #NO_MATCH
041     * @see #EXACT_MATCH
042     * @see #AUTOMATON_HAS_PREFIX
043     * @see #SEQUENCE_IS_A_PREFIX
044     */
045    public int kind;
046
047    /**
048     * Input sequence's index, interpretation depends on {@link #kind}.
049     */
050    public int index;
051
052    /**
053     * Automaton node, interpretation depends on the {@link #kind}.
054     */
055    public int node;
056
057    /*
058     *
059     */
060    public MatchResult() {
061        reset(NO_MATCH, 0, 0);
062    }
063
064    /*
065     * 
066     */
067    final void reset(int kind, int index, int node) {
068        this.kind = kind;
069        this.index = index;
070        this.node = node;
071    }
072}
