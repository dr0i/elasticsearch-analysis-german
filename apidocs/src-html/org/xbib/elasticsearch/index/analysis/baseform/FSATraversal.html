001package org.xbib.elasticsearch.index.analysis.baseform;
002
003import static org.xbib.elasticsearch.index.analysis.baseform.MatchResult.AUTOMATON_HAS_PREFIX;
004import static org.xbib.elasticsearch.index.analysis.baseform.MatchResult.EXACT_MATCH;
005import static org.xbib.elasticsearch.index.analysis.baseform.MatchResult.NO_MATCH;
006import static org.xbib.elasticsearch.index.analysis.baseform.MatchResult.SEQUENCE_IS_A_PREFIX;
007
008/**
009 * This class implements some common matching and scanning operations on a
010 * generic FSA.
011 */
012public final class FSATraversal {
013    /**
014     * Target automaton.
015     */
016    private final FSA fsa;
017
018    /**
019     * Traversals of the given FSA.
020     */
021    public FSATraversal(FSA fsa) {
022        this.fsa = fsa;
023    }
024
025    /**
026     * Calculate perfect hash for a given input sequence of bytes. The perfect hash requires
027     * that {@link FSA} is built with {@link FSAFlags#NUMBERS} and corresponds to the sequential
028     * order of input sequences used at automaton construction time.
029     *
030     * @param start  Start index in the sequence array.
031     * @param length Length of the byte sequence, must be at least 1.
032     * @return Returns a unique integer assigned to the input sequence in the automaton (reflecting
033     * the number of that sequence in the input used to build the automaton). Returns a negative
034     * integer if the input sequence was not part of the input from which the automaton was created.
035     * The type of mismatch is a constant defined in {@link MatchResult}.
036     */
037    public int perfectHash(byte[] sequence, int start, int length, int node) {
038        assert fsa.getFlags().contains(FSAFlags.NUMBERS) : "FSA not built with NUMBERS option.";
039        assert length > 0 : "must be a non-empty sequence";
040        int hash = 0;
041        final int end = start + length - 1;
042        int seqIndex = start;
043        byte label = sequence[seqIndex];
044
045        // Seek through the current node's labels, looking for 'label', update hash.
046        for (int arc = fsa.getFirstArc(node); arc != 0; ) {
047            if (fsa.getArcLabel(arc) == label) {
048                if (fsa.isArcFinal(arc)) {
049                    if (seqIndex == end) {
050                        return hash;
051                    }
052
053                    hash++;
054                }
055
056                if (fsa.isArcTerminal(arc)) {
057                    /* The automaton contains a prefix of the input sequence. */
058                    return AUTOMATON_HAS_PREFIX;
059                }
060
061                // The sequence is a prefix of one of the sequences stored in the automaton.
062                if (seqIndex == end) {
063                    return SEQUENCE_IS_A_PREFIX;
064                }
065
066                // Make a transition along the arc, go the target node's first arc.
067                arc = fsa.getFirstArc(fsa.getEndNode(arc));
068                label = sequence[++seqIndex];
069                continue;
070            } else {
071                if (fsa.isArcFinal(arc)) {
072                    hash++;
073                }
074                if (!fsa.isArcTerminal(arc)) {
075                    hash += fsa.getRightLanguageCount(fsa.getEndNode(arc));
076                }
077            }
078
079            arc = fsa.getNextArc(arc);
080        }
081
082        // Labels of this node ended without a match on the sequence.
083        // Perfect hash does not exist.
084        return NO_MATCH;
085    }
086
087    /**
088     * @see #perfectHash(byte[], int, int, int)
089     */
090    public int perfectHash(byte[] sequence) {
091        return perfectHash(sequence, 0, sequence.length, fsa.getRootNode());
092    }
093
094    /**
095     * Same as {@link #match(byte[], int, int, int)}, but allows passing
096     * a reusable {@link MatchResult} object so that no intermediate garbage is
097     * produced.
098     *
099     * @return The same object as <code>result</code>, but with reset internal
100     * type and other fields.
101     */
102    public MatchResult match(MatchResult result,
103                             byte[] sequence, int start, int length, int node) {
104        if (node == 0) {
105            result.reset(NO_MATCH, start, node);
106            return result;
107        }
108        final FSA fsa = this.fsa;
109        final int end = start + length;
110        for (int i = start; i < end; i++) {
111            final int arc = fsa.getArc(node, sequence[i]);
112            if (arc != 0) {
113                if (fsa.isArcFinal(arc) && i + 1 == end) {
114                    /* The automaton has an exact match of the input sequence. */
115                    result.reset(EXACT_MATCH, i, node);
116                    return result;
117                }
118                if (fsa.isArcTerminal(arc)) {
119                                        /* The automaton contains a prefix of the input sequence. */
120                    result.reset(AUTOMATON_HAS_PREFIX, i + 1, 0);
121                    return result;
122                }
123                // Make a transition along the arc.
124                node = fsa.getEndNode(arc);
125            } else {
126                result.reset(NO_MATCH, i, node);
127                return result;
128            }
129        }
130                /* The sequence is a prefix of at least one sequence in the automaton. */
131        result.reset(SEQUENCE_IS_A_PREFIX, 0, node);
132        return result;
133    }
134
135    /**
136     * Finds a matching path in the dictionary for a given sequence of labels
137     * from <code>sequence</code> and starting at node <code>node</code>.
138     *
139     * @param sequence An array of labels to follow in the FSA.
140     * @param start    Starting index in <code>sequence</code>.
141     * @param length   How many symbols to consider from <code>sequence</code>?
142     * @param node     Start node identifier in the FSA.
143     */
144    public MatchResult match(byte[] sequence, int start, int length, int node) {
145        return match(new MatchResult(), sequence, start, length, node);
146    }
147
148    /**
149     * @see #match(byte[], int, int, int)
150     */
151    public MatchResult match(byte[] sequence, int node) {
152        return match(sequence, 0, sequence.length, node);
153    }
154
155    /**
156     * @see #match(byte[], int, int, int)
157     */
158    public MatchResult match(byte[] sequence) {
159        return match(sequence, fsa.getRootNode());
160    }
161}
