001package org.xbib.elasticsearch.index.analysis.baseform;
002
003import java.io.BufferedReader;
004import java.io.IOException;
005import java.io.Reader;
006import java.nio.ByteBuffer;
007import java.nio.CharBuffer;
008import java.nio.charset.CharacterCodingException;
009import java.nio.charset.Charset;
010import java.util.ArrayList;
011import java.util.Collections;
012import java.util.List;
013
014import static org.xbib.elasticsearch.index.analysis.baseform.MatchResult.EXACT_MATCH;
015import static org.xbib.elasticsearch.index.analysis.baseform.MatchResult.NO_MATCH;
016import static org.xbib.elasticsearch.index.analysis.baseform.MatchResult.SEQUENCE_IS_A_PREFIX;
017
018public class Dictionary {
019
020    private final Charset UTF8 = Charset.forName("UTF-8");
021
022    private FSA fsa;
023
024    private FSATraversal matcher;
025
026    public Dictionary load(Reader in) throws IOException {
027        BufferedReader reader = new BufferedReader(in);
028        List<byte[]> lines = new ArrayList<byte[]>();
029        String line;
030        while ((line = reader.readLine()) != null) {
031            lines.add(line.replace('\t', '+').getBytes(UTF8));
032        }
033        reader.close();
034        Collections.sort(lines, FSABuilder.LEXICAL_ORDERING);
035        FSABuilder builder = new FSABuilder();
036        for (byte[] b : lines) {
037            builder.add(b, 0, b.length);
038        }
039        this.fsa = builder.complete();
040        this.matcher = new FSATraversal(fsa);
041        return this;
042    }
043
044    public String lookup(CharSequence prefix) throws CharacterCodingException {
045        return lookup(UTF8.newEncoder().encode(CharBuffer.wrap(prefix)), prefix.toString());
046    }
047
048    public String lookup(ByteBuffer buf, String result) {
049        MatchResult match = matcher.match(buf.array(), buf.position(), buf.remaining(), fsa.getRootNode());
050        switch (match.kind) {
051            case SEQUENCE_IS_A_PREFIX: {
052                final int arc = fsa.getArc(match.node, (byte) '+');
053                if (arc != 0 && !fsa.isArcFinal(arc)) {
054                    FSAFinalStatesIterator finalStatesIterator = new FSAFinalStatesIterator(fsa, fsa.getRootNode());
055                    finalStatesIterator.restartFrom(fsa.getEndNode(arc));
056                    if (finalStatesIterator.hasNext()) {
057                        buf = finalStatesIterator.next();
058                        String s = new String(buf.array(), buf.position(), buf.remaining(), UTF8);
059                        return s.isEmpty() || s.equals(result) ? s : lookup(buf, s);
060                    }
061                }
062                break;
063            }
064            case EXACT_MATCH: {
065                break;
066            }
067            case NO_MATCH: {
068                break;
069            }
070        }
071        return result;
072    }
073
074}
