001package org.xbib.elasticsearch.index.analysis.baseform;
002
003import org.apache.lucene.analysis.TokenStream;
004import org.elasticsearch.ElasticsearchIllegalArgumentException;
005import org.elasticsearch.common.inject.Inject;
006import org.elasticsearch.common.inject.assistedinject.Assisted;
007import org.elasticsearch.common.settings.Settings;
008import org.elasticsearch.env.Environment;
009import org.elasticsearch.index.Index;
010import org.elasticsearch.index.analysis.AbstractTokenFilterFactory;
011import org.elasticsearch.index.settings.IndexSettings;
012
013import java.io.IOException;
014import java.io.InputStreamReader;
015
016public class BaseformTokenFilterFactory extends AbstractTokenFilterFactory {
017
018    private final Dictionary dictionary;
019
020    @Inject
021    public BaseformTokenFilterFactory(Index index,
022                                      @IndexSettings Settings indexSettings, Environment env,
023                                      @Assisted String name, @Assisted Settings settings) {
024        super(index, indexSettings, name, settings);
025        this.dictionary = createDictionary(env, settings);
026    }
027
028    @Override
029    public TokenStream create(TokenStream tokenStream) {
030        return new BaseformTokenFilter(tokenStream, dictionary);
031    }
032
033    private Dictionary createDictionary(Environment env, Settings settings) {
034        try {
035            String lang = settings.get("language", "de");
036            String path = "/baseform/" + lang + "-lemma-utf8.txt";
037            return new Dictionary().load(new InputStreamReader(env.resolveConfig(path).openStream(), "UTF-8"));
038        } catch (IOException e) {
039            throw new ElasticsearchIllegalArgumentException("resources in settings not found: " + settings, e);
040        }
041    }
042}
