001package org.xbib.elasticsearch.index.analysis.langdetect;
002
003import java.util.HashMap;
004import java.util.Iterator;
005import java.util.Map;
006import java.util.Set;
007
008public class LangProfile {
009
010    private static final int MINIMUM_FREQ = 2;
011
012    private static final int LESS_FREQ_RATIO = 100000;
013
014    public String name = null;
015
016    public Map<String, Integer> freq = new HashMap<String, Integer>();
017
018    public int[] n_words = new int[NGram.N_GRAM];
019
020    public LangProfile() {
021    }
022
023    public LangProfile(String name) {
024        this.name = name;
025    }
026
027    public void add(String gram) {
028        if (name == null || gram == null) {
029            return;
030        }
031        int len = gram.length();
032        if (len < 1 || len > NGram.N_GRAM) {
033            return;
034        }
035        ++n_words[len - 1];
036        if (freq.containsKey(gram)) {
037            freq.put(gram, freq.get(gram) + 1);
038        } else {
039            freq.put(gram, 1);
040        }
041    }
042
043    public void omitLessFreq() {
044        if (name == null) {
045            return;
046        }
047        int threshold = n_words[0] / LESS_FREQ_RATIO;
048        if (threshold < MINIMUM_FREQ) {
049            threshold = MINIMUM_FREQ;
050        }
051        Set<String> keys = freq.keySet();
052        int roman = 0;
053        for (Iterator<String> i = keys.iterator(); i.hasNext(); ) {
054            String key = i.next();
055            int count = freq.get(key);
056            if (count <= threshold) {
057                n_words[key.length() - 1] -= count;
058                i.remove();
059            } else {
060                if (key.matches("^[A-Za-z]$")) {
061                    roman += count;
062                }
063            }
064        }
065        if (roman < n_words[0] / 3) {
066            Set<String> keys2 = freq.keySet();
067            for (Iterator<String> i = keys2.iterator(); i.hasNext(); ) {
068                String key = i.next();
069                if (key.matches(".*[A-Za-z].*")) {
070                    n_words[key.length() - 1] -= freq.get(key);
071                    i.remove();
072                }
073            }
074        }
075    }
076}
