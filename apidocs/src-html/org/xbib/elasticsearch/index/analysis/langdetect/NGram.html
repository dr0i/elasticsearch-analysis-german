019    public NGram() {
020        grams = new StringBuilder(" ");
021        capitalword = false;
022    }
023
024    public void addChar(char ch) {
025        ch = normalize(ch);
026        char lastchar = grams.charAt(grams.length() - 1);
027        if (lastchar == ' ') {
028            grams = new StringBuilder(" ");
029            capitalword = false;
030            if (ch == ' ') {
031                return;
032            }
033        } else if (grams.length() >= N_GRAM) {
034            grams.deleteCharAt(0);
035        }
036        grams.append(ch);
037        if (Character.isUpperCase(ch)) {
038            if (Character.isUpperCase(lastchar)) {
039                capitalword = true;
040            }
041        } else {
042            capitalword = false;
043        }
044    }
045
046    public String get(int n) {
047        if (capitalword) {
048            return null;
049        }
050        int len = grams.length();
051        if (n < 1 || n > 3 || len < n) {
052            return null;
053        }
054        if (n == 1) {
055            char ch = grams.charAt(len - 1);
056            if (ch == ' ') {
057                return null;
058            }
059            return Character.toString(ch);
060        } else {
061            return grams.substring(len - n, len);
062        }
063    }
064
065    public static char normalize(char ch) {
066        Character.UnicodeBlock block = Character.UnicodeBlock.of(ch);
067        if (block == UnicodeBlock.BASIC_LATIN) {
068            if (ch < 'A' || (ch < 'a' && ch > 'Z') || ch > 'z') {
069                ch = ' ';
070            }
071        } else if (block == UnicodeBlock.LATIN_1_SUPPLEMENT) {
072            if (LATIN1_EXCLUDED.indexOf(ch) >= 0) {
073                ch = ' ';
074            }
075        } else if (block == UnicodeBlock.GENERAL_PUNCTUATION) {
076            ch = ' ';
077        } else if (block == UnicodeBlock.ARABIC) {
078            if (ch == '\u06cc') {
079                ch = '\u064a';
080            }
081        } else if (block == UnicodeBlock.LATIN_EXTENDED_ADDITIONAL) {
082            if (ch >= '\u1ea0') {
083                ch = '\u1ec3';
084            }
085        } else if (block == UnicodeBlock.HIRAGANA) {
086            ch = '\u3042';
087        } else if (block == UnicodeBlock.KATAKANA) {
088            ch = '\u30a2';
089        } else if (block == UnicodeBlock.BOPOMOFO || block == UnicodeBlock.BOPOMOFO_EXTENDED) {
090            ch = '\u3105';
091        } else if (block == UnicodeBlock.CJK_UNIFIED_IDEOGRAPHS) {
092            if (cjk_map.containsKey(ch)) {
093                ch = cjk_map.get(ch);
094            }
095        } else if (block == UnicodeBlock.HANGUL_SYLLABLES) {
096            ch = '\uac00';
097        }
098        return ch;
099    }
