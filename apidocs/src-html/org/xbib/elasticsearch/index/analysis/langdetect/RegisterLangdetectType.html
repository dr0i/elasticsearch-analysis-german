001package org.xbib.elasticsearch.index.analysis.langdetect;
002
003import org.elasticsearch.common.inject.Inject;
004import org.elasticsearch.common.settings.Settings;
005import org.elasticsearch.index.AbstractIndexComponent;
006import org.elasticsearch.index.Index;
007import org.elasticsearch.index.mapper.MapperService;
008import org.elasticsearch.index.settings.IndexSettings;
009
010public class RegisterLangdetectType extends AbstractIndexComponent {
011
012    @Inject
013    public RegisterLangdetectType(Index index, @IndexSettings Settings indexSettings,
014                                  MapperService mapperService, LangdetectService detector) {
015        super(index, indexSettings);
016        mapperService.documentMapperParser().putTypeParser("langdetect",
017                new LangdetectMapper.TypeParser(detector));
018    }
019}
