001package org.xbib.elasticsearch.index.analysis.langdetect;
002
003import org.elasticsearch.common.io.stream.StreamInput;
004import org.elasticsearch.common.io.stream.StreamOutput;
005import org.elasticsearch.common.io.stream.Streamable;
006
007import java.io.IOException;
008
009public class Language implements Streamable {
010
011    private String lang;
012    private double prob;
013
014    public Language(String lang, double prob) {
015        this.lang = lang;
016        this.prob = prob;
017    }
018
019    public String getLanguage() {
020        return lang;
021    }
022
023    public double getProbability() {
024        return prob;
025    }
026
027    @Override
028    public void readFrom(StreamInput in) throws IOException {
029        this.lang = in.readString();
030        this.prob = in.readDouble();
031    }
032
033    @Override
034    public void writeTo(StreamOutput out) throws IOException {
035        out.writeString(lang);
036        out.writeDouble(prob);
037    }
038}
