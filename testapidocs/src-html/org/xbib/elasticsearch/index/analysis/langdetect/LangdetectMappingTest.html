027import static org.elasticsearch.common.io.Streams.copyToStringFromClasspath;

030
031public class LangdetectMappingTest extends Assert {
032
033    private DocumentMapperParser mapperParser;
034
035    @BeforeClass
036    public void setupMapperParser() throws IOException {
037        Index index = new Index("test");
038        Map<String, AnalyzerProviderFactory> analyzerFactoryFactories = Maps.newHashMap();
039        analyzerFactoryFactories.put("keyword",
040                new PreBuiltAnalyzerProviderFactory("keyword", AnalyzerScope.INDEX, new KeywordAnalyzer()));
041        AnalysisService analysisService = new AnalysisService(index,
042                ImmutableSettings.Builder.EMPTY_SETTINGS, null, analyzerFactoryFactories, null, null, null);
043        mapperParser = new DocumentMapperParser(index,
044                analysisService,
045                new PostingsFormatService(index),
046                new DocValuesFormatService(index),
047                new SimilarityLookupService(index, ImmutableSettings.Builder.EMPTY_SETTINGS));
048        Settings settings = settingsBuilder()
049                .build();
050        LangdetectService detector = new LangdetectService(settings);
051        detector.start();
052        mapperParser.putTypeParser(LangdetectMapper.CONTENT_TYPE,
053                new LangdetectMapper.TypeParser(detector));
054    }
055
056    @Test
057    public void testSimpleMappings() throws Exception {
058        String mapping = copyToStringFromClasspath("/simple-mapping.json");
059        DocumentMapper docMapper = mapperParser.parse(mapping);
060
061        String sampleText = copyToStringFromClasspath("/english.txt");
062        BytesReference json = jsonBuilder().startObject().field("_id", 1).field("someField", sampleText).endObject().bytes();
063        ParseContext.Document doc = docMapper.parse(json).rootDoc();
064
065        assertEquals(doc.get(docMapper.mappers().smartName("someField").mapper().names().indexName()), sampleText);
066        assertEquals(doc.getFields("someField.lang").length, 1);
067        assertEquals(doc.getFields("someField.lang")[0].stringValue(), "en");
068
069        // re-parse it
070        String builtMapping = docMapper.mappingSource().string();
071        docMapper = mapperParser.parse(builtMapping);
072
073        json = jsonBuilder().startObject().field("_id", 1).field("someField", sampleText).endObject().bytes();
074        doc = docMapper.parse(json).rootDoc();
075
076        assertEquals(doc.get(docMapper.mappers().smartName("someField").mapper().names().indexName()), sampleText);
077        assertEquals(doc.getFields("someField.lang").length, 1);
078        assertEquals(doc.getFields("someField.lang")[0].stringValue(), "en");
079    }
080
