001package org.xbib.elasticsearch.index.analysis.icu;
002
003import com.ibm.icu.text.Collator;
004import org.apache.lucene.analysis.Analyzer;
005import org.apache.lucene.collation.ICUCollationKeyAnalyzer;
006import org.apache.lucene.util.BytesRef;
007import org.apache.lucene.util.Version;
008import org.junit.Test;
009
010import java.util.Locale;
011import java.util.Random;
012
013public class ICUCollationKeyAnalyzerTests extends CollationTestBase {
014
015    private Collator collator = Collator.getInstance(new Locale("fa"));
016
017    private Analyzer analyzer = new ICUCollationKeyAnalyzer(Version.LUCENE_4_9, collator);
018
019    private BytesRef firstRangeBeginning = new BytesRef(collator.getCollationKey(firstRangeBeginningOriginal).toByteArray());
020    private BytesRef firstRangeEnd = new BytesRef(collator.getCollationKey(firstRangeEndOriginal).toByteArray());
021    private BytesRef secondRangeBeginning = new BytesRef(collator.getCollationKey(secondRangeBeginningOriginal).toByteArray());
022    private BytesRef secondRangeEnd = new BytesRef(collator.getCollationKey(secondRangeEndOriginal).toByteArray());
023
024    @Test
025    public void testFarsiRangeFilterCollating() throws Exception {
026        testFarsiRangeFilterCollating(analyzer, firstRangeBeginning, firstRangeEnd,
027                secondRangeBeginning, secondRangeEnd);
028    }
029
030    @Test
031    public void testFarsiRangeQueryCollating() throws Exception {
032        testFarsiRangeQueryCollating(analyzer, firstRangeBeginning, firstRangeEnd,
033                secondRangeBeginning, secondRangeEnd);
034    }
035
036    @Test
037    public void testFarsiTermRangeQuery() throws Exception {
038        testFarsiTermRangeQuery
039                (analyzer, firstRangeBeginning, firstRangeEnd,
040                        secondRangeBeginning, secondRangeEnd);
041    }
042
043    @Test
044    public void testCollationKeySort() throws Exception {
045        Analyzer usAnalyzer = new ICUCollationKeyAnalyzer(Version.LUCENE_4_9, Collator.getInstance(Locale.ROOT));
046        Analyzer franceAnalyzer = new ICUCollationKeyAnalyzer(Version.LUCENE_4_9, Collator.getInstance(Locale.FRANCE));
047        Analyzer swedenAnalyzer = new ICUCollationKeyAnalyzer(Version.LUCENE_4_9, Collator.getInstance(new Locale("sv", "se")));
048        Analyzer denmarkAnalyzer = new ICUCollationKeyAnalyzer(Version.LUCENE_4_9, Collator.getInstance(new Locale("da", "dk")));
049        testCollationKeySort(usAnalyzer, franceAnalyzer, swedenAnalyzer, denmarkAnalyzer,
050                "BFJHD", "ECAGI", "BJDFH", "BJDHF");
051    }
052
053    @Test
054    public void testThreadSafe() throws Exception {
055        int iters = 20;
056        for (int i = 0; i < iters; i++) {
057            Locale locale = Locale.GERMAN;
058            Collator collator = Collator.getInstance(locale);
059            collator.setStrength(Collator.IDENTICAL);
060            assertThreadSafe(new Random(), new ICUCollationKeyAnalyzer(Version.LUCENE_4_9, collator));
061        }
062    }
063}
