003import org.apache.lucene.analysis.Tokenizer;
004import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
005import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
006import org.apache.lucene.util.AttributeFactory;
007import org.apache.lucene.util.automaton.CharacterRunAutomaton;
008import org.apache.lucene.util.automaton.RegExp;
009
010import java.io.IOException;
011import java.io.Reader;
012import java.nio.CharBuffer;
013import java.util.Random;

015/**
016 * Tokenizer for testing.
017 * <p/>
018 * This tokenizer is a replacement for {@link #WHITESPACE}, {@link #SIMPLE}, and {@link #KEYWORD}
019 * tokenizers. If you are writing a component such as a TokenFilter, its a great idea to test
020 * it wrapping this tokenizer instead for extra checks. This tokenizer has the following behavior:
021 * <ul>
022 * <li>An internal state-machine is used for checking consumer consistency. These checks can
023 * be disabled with {@link #setEnableChecks(boolean)}.
024 * <li>For convenience, optionally lowercases terms that it outputs.
025 * </ul>
026 */
027public class MockTokenizer extends Tokenizer {
028    /**
029     * Acts Similar to WhitespaceTokenizer
030     */
031    public static final CharacterRunAutomaton WHITESPACE =
032            new CharacterRunAutomaton(new RegExp("[^ \t\r\n]+").toAutomaton());
033    /**
034     * Acts Similar to KeywordTokenizer.
035     * TODO: Keyword returns an "empty" token for an empty reader...
036     */
037    public static final CharacterRunAutomaton KEYWORD =
038            new CharacterRunAutomaton(new RegExp(".*").toAutomaton());
039    /**
040     * Acts like LetterTokenizer.
041     */
042    // the ugly regex below is incomplete Unicode 5.2 [:Letter:]
043    public static final CharacterRunAutomaton SIMPLE =
044            new CharacterRunAutomaton(new RegExp("[A-Za-zªµºÀ-ÖØ-öø-ˁ一-鿌]+").toAutomaton());
045
046    private final CharacterRunAutomaton runAutomaton;
047    private final boolean lowerCase;
048    private final int maxTokenLength;
049    public static final int DEFAULT_MAX_TOKEN_LENGTH = Integer.MAX_VALUE;
050    private int state;
051
052    private final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
053    private final OffsetAttribute offsetAtt = addAttribute(OffsetAttribute.class);
054    int off = 0;
055
056    // buffered state (previous codepoint and offset). we replay this once we
057    // hit a reject state in case its permissible as the start of a new term.
058    int bufferedCodePoint = -1; // -1 indicates empty buffer
059    int bufferedOff = -1;
060
061    // TODO: "register" with LuceneTestCase to ensure all streams are closed() ?
062    // currently, we can only check that the lifecycle is correct if someone is reusing,
063    // but not for "one-offs".
064    private static enum State {
065        SETREADER,       // consumer set a reader input either via ctor or via reset(Reader)
066        RESET,           // consumer has called reset()
067        INCREMENT,       // consumer is consuming, has called incrementToken() == true
068        INCREMENT_FALSE, // consumer has called incrementToken() which returned false
069        END,             // consumer has called end() to perform end of stream operations
070        CLOSE            // consumer has called close() to release any resources
071    }
072
073    ;
074
075    private State streamState = State.CLOSE;
076    private int lastOffset = 0; // only for asserting
077    private boolean enableChecks = true;
078
079    // evil: but we don't change the behavior with this random, we only switch up how we read
080    private final Random random = new Random();
081
082    public MockTokenizer(AttributeFactory factory, Reader input, CharacterRunAutomaton runAutomaton, boolean lowerCase, int maxTokenLength) {
083        super(factory, input);
084        this.runAutomaton = runAutomaton;
085        this.lowerCase = lowerCase;
086        this.state = runAutomaton.getInitialState();
087        this.streamState = State.SETREADER;
088        this.maxTokenLength = maxTokenLength;
089    }
090
091    public MockTokenizer(Reader input, CharacterRunAutomaton runAutomaton, boolean lowerCase, int maxTokenLength) {
092        this(AttributeFactory.DEFAULT_ATTRIBUTE_FACTORY, input, runAutomaton, lowerCase, maxTokenLength);
093    }
094
095    public MockTokenizer(Reader input, CharacterRunAutomaton runAutomaton, boolean lowerCase) {
096        this(input, runAutomaton, lowerCase, DEFAULT_MAX_TOKEN_LENGTH);
097    }
098
099    /**
100     * Calls {@link #MockTokenizer(Reader, CharacterRunAutomaton, boolean) MockTokenizer(Reader, WHITESPACE, true)}
101     */
102    public MockTokenizer(Reader input) {
103        this(input, WHITESPACE, true);
104    }
105
106    public MockTokenizer(AttributeFactory factory, Reader input, CharacterRunAutomaton runAutomaton, boolean lowerCase) {
107        this(factory, input, runAutomaton, lowerCase, DEFAULT_MAX_TOKEN_LENGTH);
108    }
109
110    public MockTokenizer(AttributeFactory factory, Reader input) {
111        this(input, WHITESPACE, true);
112    }
113
114    @Override
115    public final boolean incrementToken() throws IOException {
116        assert !enableChecks || (streamState == State.RESET || streamState == State.INCREMENT)
117                : "incrementToken() called while in wrong state: " + streamState;
118        clearAttributes();
119        for (; ; ) {
120            int startOffset;
121            int cp;
122            if (bufferedCodePoint >= 0) {
123                cp = bufferedCodePoint;
124                startOffset = bufferedOff;
125                bufferedCodePoint = -1;
126            } else {
127                startOffset = off;
128                cp = readCodePoint();
129            }
130            if (cp < 0) {
131                break;
132            } else if (isTokenChar(cp)) {
133                int endOffset;
134                do {
135                    char chars[] = Character.toChars(normalize(cp));
136                    for (int i = 0; i < chars.length; i++) {
137                        termAtt.append(chars[i]);
138                    }
139                    endOffset = off;
140                    if (termAtt.length() >= maxTokenLength) {
141                        break;
142                    }
143                    cp = readCodePoint();
144                } while (cp >= 0 && isTokenChar(cp));
145
146                if (termAtt.length() < maxTokenLength) {
147                    // buffer up, in case the "rejected" char can start a new word of its own
148                    bufferedCodePoint = cp;
149                    bufferedOff = endOffset;
150                } else {
151                    // otherwise, its because we hit term limit.
152                    bufferedCodePoint = -1;
153                }
154                int correctedStartOffset = correctOffset(startOffset);
155                int correctedEndOffset = correctOffset(endOffset);
156                assert correctedStartOffset >= 0;
157                assert correctedEndOffset >= 0;
158                assert correctedStartOffset >= lastOffset;
159                lastOffset = correctedStartOffset;
160                assert correctedEndOffset >= correctedStartOffset;
161                offsetAtt.setOffset(correctedStartOffset, correctedEndOffset);
162                if (state == -1 || runAutomaton.isAccept(state)) {
163                    // either we hit a reject state (longest match), or end-of-text, but in an accept state
164                    streamState = State.INCREMENT;
165                    return true;
166                }
167            }
168        }
169        streamState = State.INCREMENT_FALSE;
170        return false;
171    }
172
173    protected int readCodePoint() throws IOException {
174        int ch = readChar();
175        if (ch < 0) {
176            return ch;
177        } else {
178            assert !Character.isLowSurrogate((char) ch) : "unpaired low surrogate: " + Integer.toHexString(ch);
179            off++;
180            if (Character.isHighSurrogate((char) ch)) {
181                int ch2 = readChar();
182                if (ch2 >= 0) {
183                    off++;
184                    assert Character.isLowSurrogate((char) ch2) : "unpaired high surrogate: " + Integer.toHexString(ch) + ", followed by: " + Integer.toHexString(ch2);
185                    return Character.toCodePoint((char) ch, (char) ch2);
186                } else {
187                    assert false : "stream ends with unpaired high surrogate: " + Integer.toHexString(ch);
188                }
189            }
190            return ch;
191        }
192    }
193
194    protected int readChar() throws IOException {
195        switch (random.nextInt(10)) {
196            case 0: {
197                // read(char[])
198                char c[] = new char[1];
199                int ret = input.read(c);
200                return ret < 0 ? ret : c[0];
201            }
202            case 1: {
203                // read(char[], int, int)
204                char c[] = new char[2];
205                int ret = input.read(c, 1, 1);
206                return ret < 0 ? ret : c[1];
207            }
208            case 2: {
209                // read(CharBuffer)
210                char c[] = new char[1];
211                CharBuffer cb = CharBuffer.wrap(c);
212                int ret = input.read(cb);
213                return ret < 0 ? ret : c[0];
214            }
215            default:
216                // read()
217                return input.read();
218        }
219    }
220
221    protected boolean isTokenChar(int c) {
222        if (state < 0) {
223            state = runAutomaton.getInitialState();
224        }
225        state = runAutomaton.step(state, c);
226        if (state < 0) {
227            return false;
228        } else {
229            return true;
230        }
231    }
232
233    protected int normalize(int c) {
234        return lowerCase ? Character.toLowerCase(c) : c;
235    }
236
237    @Override
238    public void reset() throws IOException {
239        super.reset();
240        state = runAutomaton.getInitialState();
241        lastOffset = off = 0;
242        bufferedCodePoint = -1;
243        assert !enableChecks || streamState != State.RESET : "double reset()";
244        streamState = State.RESET;
245    }
246
247    @Override
248    public void close() throws IOException {
249        super.close();
250        // in some exceptional cases (e.g. TestIndexWriterExceptions) a test can prematurely close()
251        // these tests should disable this check, by default we check the normal workflow.
252        // TODO: investigate the CachingTokenFilter "double-close"... for now we ignore this
253        assert !enableChecks || streamState == State.END || streamState == State.CLOSE : "close() called in wrong state: " + streamState;
254        streamState = State.CLOSE;
255    }
256
257    @Override
258    public void end() throws IOException {
259        super.end();
260        int finalOffset = correctOffset(off);
261        offsetAtt.setOffset(finalOffset, finalOffset);
262        // some tokenizers, such as limiting tokenizers, call end() before incrementToken() returns false.
263        // these tests should disable this check (in general you should consume the entire stream)
264        try {
265            assert !enableChecks || streamState == State.INCREMENT_FALSE : "end() called before incrementToken() returned false!";
266        } finally {
267            streamState = State.END;
268        }
269    }
270
271    /**
272     * Toggle consumer workflow checking: if your test consumes tokenstreams normally you
273     * should leave this enabled.
274     */
275    public void setEnableChecks(boolean enableChecks) {
276        this.enableChecks = enableChecks;
277    }
278}
