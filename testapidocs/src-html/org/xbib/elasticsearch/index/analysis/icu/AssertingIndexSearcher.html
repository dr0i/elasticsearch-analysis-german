001package org.xbib.elasticsearch.index.analysis.icu;
002
003import org.apache.lucene.index.AtomicReaderContext;
004import org.apache.lucene.index.IndexReader;
005import org.apache.lucene.index.IndexReaderContext;
006import org.apache.lucene.search.Collector;
007import org.apache.lucene.search.Filter;
008import org.apache.lucene.search.FilteredQuery;
009import org.apache.lucene.search.IndexSearcher;
010import org.apache.lucene.search.Query;
011import org.apache.lucene.search.Weight;
012import org.apache.lucene.util.Bits;
013
014import java.io.IOException;
015import java.util.List;
016import java.util.Random;
017import java.util.concurrent.ExecutorService;
018
019
020public class AssertingIndexSearcher extends IndexSearcher {
021    final Random random;
022
023    public AssertingIndexSearcher(Random random, IndexReader r) {
024        super(r);
025        this.random = new Random(random.nextLong());
026    }
027
028    public AssertingIndexSearcher(Random random, IndexReaderContext context) {
029        super(context);
030        this.random = new Random(random.nextLong());
031    }
032
033    public AssertingIndexSearcher(Random random, IndexReader r, ExecutorService ex) {
034        super(r, ex);
035        this.random = new Random(random.nextLong());
036    }
037
038    public AssertingIndexSearcher(Random random, IndexReaderContext context, ExecutorService ex) {
039        super(context, ex);
040        this.random = new Random(random.nextLong());
041    }
042
043    /**
044     * Ensures, that the returned {@code Weight} is not normalized again, which may produce wrong scores.
045     */
046    @Override
047    public Weight createNormalizedWeight(Query query) throws IOException {
048        return super.createNormalizedWeight(query);
049    }
050
051    @Override
052    public Query rewrite(Query original) throws IOException {
053        return super.rewrite(original);
054    }
055
056    @Override
057    protected Query wrapFilter(Query query, Filter filter) {
058        if (random.nextBoolean()) {
059            return super.wrapFilter(query, filter);
060        }
061        return (filter == null) ? query : new FilteredQuery(query, filter, randomFilterStrategy(random));
062    }
063
064    @Override
065    protected void search(List<AtomicReaderContext> leaves, Weight weight, Collector collector) throws IOException {
066        super.search(leaves, weight, collector);
067    }
068
069    @Override
070    public String toString() {
071        return "AssertingIndexSearcher(" + super.toString() + ")";
072    }
073
074
075    public static FilteredQuery.FilterStrategy randomFilterStrategy(final Random random) {
076        switch (random.nextInt(6)) {
077            case 5:
078            case 4:
079                return new FilteredQuery.RandomAccessFilterStrategy() {
080                    @Override
081                    protected boolean useRandomAccess(Bits bits, int firstFilterDoc) {
082                        return random.nextBoolean();
083                    }
084                };
085            case 3:
086                return FilteredQuery.RANDOM_ACCESS_FILTER_STRATEGY;
087            case 2:
088                return FilteredQuery.LEAP_FROG_FILTER_FIRST_STRATEGY;
089            case 1:
090                return FilteredQuery.LEAP_FROG_QUERY_FIRST_STRATEGY;
091            case 0:
092                return FilteredQuery.QUERY_FIRST_FILTER_STRATEGY;
093            default:
094                return FilteredQuery.RANDOM_ACCESS_FILTER_STRATEGY;
095        }
096    }
097
098
099}
