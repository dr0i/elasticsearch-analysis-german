048    public void testFarsiRangeFilterCollating(Analyzer analyzer, BytesRef firstBeg,
049                                              BytesRef firstEnd, BytesRef secondBeg,
050                                              BytesRef secondEnd) throws Exception {
051        Directory dir = new RAMDirectory();
052        IndexWriter writer = new IndexWriter(dir, new IndexWriterConfig(
053                TEST_VERSION_CURRENT, analyzer));
054        Document doc = new Document();
055        doc.add(new TextField("content", "\u0633\u0627\u0628", Field.Store.YES));
056        doc.add(new StringField("body", "body", Field.Store.YES));
057        writer.addDocument(doc);
058        writer.close();
059        IndexReader reader = DirectoryReader.open(dir);
060        IndexSearcher searcher = new IndexSearcher(reader);
061        Query query = new TermQuery(new Term("body", "body"));
062
063        // Unicode order would include U+0633 in [ U+062F - U+0698 ], but Farsi
064        // orders the U+0698 character before the U+0633 character, so the single
065        // index Term below should NOT be returned by a TermRangeFilter with a Farsi
066        // Collator (or an Arabic one for the case when Farsi searcher not
067        // supported).
068        ScoreDoc[] result = searcher.search(query, new TermRangeFilter("content", firstBeg, firstEnd, true, true), 1).scoreDocs;
069        assertEquals("The index Term should not be included.", 0, result.length);
070
071        result = searcher.search(query, new TermRangeFilter("content", secondBeg, secondEnd, true, true), 1).scoreDocs;
072        assertEquals("The index Term should be included.", 1, result.length);
073
074        reader.close();
075        dir.close();
076    }

140    public void testCollationKeySort(Analyzer usAnalyzer,
141                                     Analyzer franceAnalyzer,
142                                     Analyzer swedenAnalyzer,
143                                     Analyzer denmarkAnalyzer,
144                                     String usResult,
145                                     String frResult,
146                                     String svResult,
147                                     String dkResult) throws Exception {
148        Directory indexStore = new RAMDirectory();
149        IndexWriter writer = new IndexWriter(indexStore, new IndexWriterConfig(
150                TEST_VERSION_CURRENT, new MockAnalyzer(MockTokenizer.WHITESPACE, false)));
151
152        String[][] sortData = new String[][]{
153                // tracer contents US                 France             Sweden (sv_SE)     Denmark (da_DK)
154                {"A", "x", "p\u00EAche", "p\u00EAche", "p\u00EAche", "p\u00EAche"},
155                {"B", "y", "HAT", "HAT", "HAT", "HAT"},
156                {"C", "x", "p\u00E9ch\u00E9", "p\u00E9ch\u00E9", "p\u00E9ch\u00E9", "p\u00E9ch\u00E9"},
157                {"D", "y", "HUT", "HUT", "HUT", "HUT"},
158                {"E", "x", "peach", "peach", "peach", "peach"},
159                {"F", "y", "H\u00C5T", "H\u00C5T", "H\u00C5T", "H\u00C5T"},
160                {"G", "x", "sin", "sin", "sin", "sin"},
161                {"H", "y", "H\u00D8T", "H\u00D8T", "H\u00D8T", "H\u00D8T"},
162                {"I", "x", "s\u00EDn", "s\u00EDn", "s\u00EDn", "s\u00EDn"},
163                {"J", "y", "HOT", "HOT", "HOT", "HOT"},
164        };
165
166        FieldType customType = new FieldType();
167        customType.setStored(true);
168
169        for (String[] s : sortData) {
170            Document doc = new Document();
171            doc.add(new Field("tracer", s[0], customType));
172            doc.add(new TextField("contents", s[1], Field.Store.NO));
173            if (s[2] != null) {
174                doc.add(new TextField("US", usAnalyzer.tokenStream("US", s[2])));
175            }
176            if (s[3] != null) {
177                doc.add(new TextField("France", franceAnalyzer.tokenStream("France", s[3])));
178            }
179            if (s[4] != null) {
180                doc.add(new TextField("Sweden", swedenAnalyzer.tokenStream("Sweden", s[4])));
181            }
182            if (s[5] != null) {
183                doc.add(new TextField("Denmark", denmarkAnalyzer.tokenStream("Denmark", s[5])));
184            }
185            writer.addDocument(doc);
186        }
187        writer.forceMerge(1);
188        writer.close();
189        IndexReader reader = DirectoryReader.open(indexStore);
190        IndexSearcher searcher = new IndexSearcher(reader);
191
192        Sort sort = new Sort();
193        Query queryX = new TermQuery(new Term("contents", "x"));
194        Query queryY = new TermQuery(new Term("contents", "y"));
195
196        sort.setSort(new SortField("US", SortField.Type.STRING));
197        assertMatches(searcher, queryY, sort, usResult);
198
199        sort.setSort(new SortField("France", SortField.Type.STRING));
200        assertMatches(searcher, queryX, sort, frResult);
201
202        sort.setSort(new SortField("Sweden", SortField.Type.STRING));
203        assertMatches(searcher, queryY, sort, svResult);
204
205        sort.setSort(new SortField("Denmark", SortField.Type.STRING));
206        assertMatches(searcher, queryY, sort, dkResult);
207        reader.close();
208        indexStore.close();
209    }

211    private void assertMatches(IndexSearcher searcher, Query query, Sort sort,
212                               String expectedResult) throws IOException {
213        ScoreDoc[] result = searcher.search(query, null, 1000, sort).scoreDocs;
214        StringBuilder buff = new StringBuilder(10);
215        int n = result.length;
216        for (ScoreDoc sd : result) {
217            Document doc = searcher.doc(sd.doc);
218            IndexableField[] v = doc.getFields("tracer");
219            for (IndexableField aV : v) {
220                buff.append(aV.stringValue());
221            }
222        }
223        assertEquals(expectedResult, buff.toString());
224    }

226    public void assertThreadSafe(final Random random, final Analyzer analyzer) throws Exception {
227        int numTestPoints = 100;
228        int numThreads = randomIntBetween(random, 3, 5);
229        final HashMap<String, BytesRef> map = new HashMap<>();
230        for (int i = 0; i < numTestPoints; i++) {
231            String term = randomSimpleString(random, 10);
232            try (TokenStream ts = analyzer.tokenStream("fake", term)) {
233                TermToBytesRefAttribute termAtt = ts.addAttribute(TermToBytesRefAttribute.class);
234                BytesRef bytes = termAtt.getBytesRef();
235                ts.reset();
236                assertTrue(ts.incrementToken());
237                termAtt.fillBytesRef();
238                // ensure we make a copy of the actual bytes too
239                map.put(term, BytesRef.deepCopyOf(bytes));
240                assertFalse(ts.incrementToken());
241                ts.end();
242            }
243        }
244
245        Thread threads[] = new Thread[numThreads];
246        for (int i = 0; i < numThreads; i++) {
247            threads[i] = new Thread() {
248                @Override
249                public void run() {
250                    try {
251                        for (Map.Entry<String, BytesRef> mapping : map.entrySet()) {
252                            String term = mapping.getKey();
253                            BytesRef expected = mapping.getValue();
254                            try (TokenStream ts = analyzer.tokenStream("fake", term)) {
255                                TermToBytesRefAttribute termAtt = ts.addAttribute(TermToBytesRefAttribute.class);
256                                BytesRef bytes = termAtt.getBytesRef();
257                                ts.reset();
258                                assertTrue(ts.incrementToken());
259                                termAtt.fillBytesRef();
260                                assertEquals(expected, bytes);
261                                assertFalse(ts.incrementToken());
262                                ts.end();
263                            }
264                        }
265                    } catch (IOException e) {
266                        throw new RuntimeException(e);
267                    }
268                }
269            };
270        }
271        for (int i = 0; i < numThreads; i++) {
272            threads[i].start();
273        }
274        for (int i = 0; i < numThreads; i++) {
275            threads[i].join();
276        }
277    }
