001package org.xbib.elasticsearch.index.analysis.icu;
002
003import org.apache.lucene.analysis.TokenFilter;
004import org.apache.lucene.analysis.TokenStream;
005import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
006import org.apache.lucene.analysis.tokenattributes.PositionIncrementAttribute;
007import org.apache.lucene.util.automaton.BasicOperations;
008import org.apache.lucene.util.automaton.CharacterRunAutomaton;
009
010import java.io.IOException;
011import java.util.Arrays;
012
013import static org.apache.lucene.util.automaton.BasicAutomata.makeEmpty;
014import static org.apache.lucene.util.automaton.BasicAutomata.makeString;
015
016/**
017 * A tokenfilter for testing that removes terms accepted by a DFA.
018 * <ul>
019 * <li>Union a list of singletons to act like a stopfilter.
020 * <li>Use the complement to act like a keepwordfilter
021 * <li>Use a regex like <code>.{12,}</code> to act like a lengthfilter
022 * </ul>
023 */
024public final class MockTokenFilter extends TokenFilter {
025    /**
026     * Empty set of stopwords
027     */
028    public static final CharacterRunAutomaton EMPTY_STOPSET =
029            new CharacterRunAutomaton(makeEmpty());
030
031    /**
032     * Set of common english stopwords
033     */
034    public static final CharacterRunAutomaton ENGLISH_STOPSET =
035            new CharacterRunAutomaton(BasicOperations.union(Arrays.asList(
036                    makeString("a"), makeString("an"), makeString("and"), makeString("are"),
037                    makeString("as"), makeString("at"), makeString("be"), makeString("but"),
038                    makeString("by"), makeString("for"), makeString("if"), makeString("in"),
039                    makeString("into"), makeString("is"), makeString("it"), makeString("no"),
040                    makeString("not"), makeString("of"), makeString("on"), makeString("or"),
041                    makeString("such"), makeString("that"), makeString("the"), makeString("their"),
042                    makeString("then"), makeString("there"), makeString("these"), makeString("they"),
043                    makeString("this"), makeString("to"), makeString("was"), makeString("will"),
044                    makeString("with"))));
045
046    private final CharacterRunAutomaton filter;
047
048    private final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
049    private final PositionIncrementAttribute posIncrAtt = addAttribute(PositionIncrementAttribute.class);
050    private int skippedPositions;
051
052    /**
053     * Create a new MockTokenFilter.
054     *
055     * @param input  TokenStream to filter
056     * @param filter DFA representing the terms that should be removed.
057     */
058    public MockTokenFilter(TokenStream input, CharacterRunAutomaton filter) {
059        super(input);
060        this.filter = filter;
061    }
062
063    @Override
064    public boolean incrementToken() throws IOException {
065        // TODO: fix me when posInc=false, to work like FilteringTokenFilter in that case and not return
066        // initial token with posInc=0 ever
067
068        // return the first non-stop word found
069        skippedPositions = 0;
070        while (input.incrementToken()) {
071            if (!filter.run(termAtt.buffer(), 0, termAtt.length())) {
072                posIncrAtt.setPositionIncrement(posIncrAtt.getPositionIncrement() + skippedPositions);
073                return true;
074            }
075            skippedPositions += posIncrAtt.getPositionIncrement();
076        }
077        // reached EOS -- return false
078        return false;
079    }
080
081    @Override
082    public void end() throws IOException {
083        super.end();
084        posIncrAtt.setPositionIncrement(posIncrAtt.getPositionIncrement() + skippedPositions);
085    }
086
087    @Override
088    public void reset() throws IOException {
089        super.reset();
090        skippedPositions = 0;
091    }
092}
