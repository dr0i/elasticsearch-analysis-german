034    /*
035    * Turkish has some funny casing.
036    * This test shows how you can solve this kind of thing easily with collation.
037    * Instead of using LowerCaseFilter, use a turkish collator with primary strength.
038    * Then things will sort and match correctly.
039    */
040    @Test
041    public void testBasicUsage() throws Exception {
042        Index index = new Index("test");
043        Settings settings = ImmutableSettings.settingsBuilder()
044                .put("index.analysis.analyzer.myAnalyzer.type", "icu_collation")
045                .put("index.analysis.analyzer.myAnalyzer.language", "tr")
046                .put("index.analysis.analyzer.myAnalyzer.strength", "primary")
047                .build();
048        AnalysisService analysisService = createAnalysisService(index, settings);
049
050        Analyzer analyzer = analysisService.analyzer("myAnalyzer");
051
052        String turkishUpperCase = "I WİLL USE TURKİSH CASING";
053        String turkishLowerCase = "ı will use turkish casıng";
054
055        assertAnalyzesTo(analyzer, turkishUpperCase, new String[]{"I WİLL USE TURKİSH CASING"});
056        assertAnalyzesTo(analyzer, turkishLowerCase, new String[]{"ı will use turkish casıng"});
057
058        TokenStream tsUpper = analyzer.tokenStream("test", new StringReader(turkishUpperCase));
059        assertTokenStreamContents(tsUpper, new String[]{"I WİLL USE TURKİSH CASING"});
060        //TokenStream tsLower = analyzer.tokenStream("test", new StringReader(turkishLowerCase));
061        //assertCollatesToSame(tsUpper, tsLower);
062    }

131    /*
132    * Setting alternate=shifted and variableTop to shift whitespace, but not
133    * punctuation or symbols, to quaternary level
134    */
135    public void testIgnoreWhitespace() throws IOException {
136        Index index = new Index("test");
137        Settings settings = ImmutableSettings.settingsBuilder()
138                .put("index.analysis.filter.myCollator.type", "icu_collation")
139                .put("index.analysis.filter.myCollator.language", "en")
140                .put("index.analysis.filter.myCollator.strength", "primary")
141                .put("index.analysis.filter.myCollator.alternate", "shifted")
142                .put("index.analysis.filter.myCollator.variableTop", " ")
143                .build();
144        AnalysisService analysisService = createAnalysisService(index, settings);
145
146        String withSpace = "foo bar";
147        String withoutSpace = "foobar";
148        String withPunctuation = "foo-bar";
149        TokenFilterFactory filterFactory = analysisService.tokenFilter("myCollator");
150
151        TokenStream tsWithSpace = filterFactory.create(new KeywordTokenizer(new StringReader(withSpace)));
152        TokenStream tsWithoutSpace = filterFactory.create(new KeywordTokenizer(new StringReader(withoutSpace)));
153        assertCollatesToSame(tsWithSpace, tsWithoutSpace);
154        // now assert that punctuation still matters: foo-bar < foo bar
155        tsWithSpace = filterFactory.create(new KeywordTokenizer(new StringReader(withSpace)));
156        TokenStream tsWithPunctuation = filterFactory.create(new KeywordTokenizer(new StringReader(withPunctuation)));
157        assertCollation(tsWithPunctuation, tsWithSpace, -1);
158    }

182    /*
183    * Setting caseLevel=true to create an additional case level between
184    * secondary and tertiary
185    */
186    public void testIgnoreAccentsButNotCase() throws IOException {
187        Index index = new Index("test");
188        Settings settings = ImmutableSettings.settingsBuilder()
189                .put("index.analysis.filter.myCollator.type", "icu_collation")
190                .put("index.analysis.filter.myCollator.language", "en")
191                .put("index.analysis.filter.myCollator.strength", "primary")
192                .put("index.analysis.filter.myCollator.caseLevel", "true")
193                .build();
194        AnalysisService analysisService = createAnalysisService(index, settings);
195
196        String withAccents = "résumé";
197        String withoutAccents = "resume";
198        String withAccentsUpperCase = "Résumé";
199        String withoutAccentsUpperCase = "Resume";
200        TokenFilterFactory filterFactory = analysisService.tokenFilter("myCollator");
201
202        TokenStream tsWithAccents = filterFactory.create(new KeywordTokenizer(new StringReader(withAccents)));
203        TokenStream tsWithoutAccents = filterFactory.create(new KeywordTokenizer(new StringReader(withoutAccents)));
204        assertCollatesToSame(tsWithAccents, tsWithoutAccents);
205
206        TokenStream tsWithAccentsUpperCase = filterFactory.create(new KeywordTokenizer(new StringReader(withAccentsUpperCase)));
207        TokenStream tsWithoutAccentsUpperCase = filterFactory.create(new KeywordTokenizer(new StringReader(withoutAccentsUpperCase)));
208        assertCollatesToSame(tsWithAccentsUpperCase, tsWithoutAccentsUpperCase);
209
210        // now assert that case still matters: resume < Resume
211        TokenStream tsLower = filterFactory.create(new KeywordTokenizer(new StringReader(withoutAccents)));
212        TokenStream tsUpper = filterFactory.create(new KeywordTokenizer(new StringReader(withoutAccentsUpperCase)));
213        assertCollation(tsLower, tsUpper, -1);
214    }

239    /*
240    * For german, you might want oe to sort and match with o umlaut.
241    * This is not the default, but you can make a customized ruleset to do this.
242    *
243    * The default is DIN 5007-1, this shows how to tailor a collator to get DIN 5007-2 behavior.
244    *  http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=4423383
245    */
246    public void testCustomRules() throws Exception {
247        RuleBasedCollator baseCollator = (RuleBasedCollator) Collator.getInstance(new ULocale("de_DE"));
248        String DIN5007_2_tailorings =
249                "& ae , a\u0308 & AE , A\u0308"+
250                        "& oe , o\u0308 & OE , O\u0308"+
251                        "& ue , u\u0308 & UE , u\u0308";
252
253        RuleBasedCollator tailoredCollator = new RuleBasedCollator(baseCollator.getRules() + DIN5007_2_tailorings);
254        String tailoredRules = tailoredCollator.getRules();
255
256        Index index = new Index("test");
257        Settings settings = ImmutableSettings.settingsBuilder()
258                .put("index.analysis.filter.myCollator.type", "icu_collation")
259                .put("index.analysis.filter.myCollator.rules", tailoredRules)
260                .put("index.analysis.filter.myCollator.strength", "primary")
261                .build();
262        AnalysisService analysisService = createAnalysisService(index, settings);
263
264        String germanUmlaut = "Töne";
265        String germanOE = "Toene";
266        TokenFilterFactory filterFactory = analysisService.tokenFilter("myCollator");
267        TokenStream tsUmlaut = filterFactory.create(new KeywordTokenizer(new StringReader(germanUmlaut)));
268        TokenStream tsOE = filterFactory.create(new KeywordTokenizer(new StringReader(germanOE)));
269        assertCollatesToSame(tsUmlaut, tsOE);
270    }

272    private AnalysisService createAnalysisService(Index index, Settings settings) {
273        Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings), new EnvironmentModule(new Environment(settings)), new IndicesAnalysisModule()).createInjector();
274        Injector injector = new ModulesBuilder().add(
275                new IndexSettingsModule(index, settings),
276                new IndexNameModule(index),
277                new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class)).addProcessor(new IcuAnalysisBinderProcessor()))
278                .createChildInjector(parentInjector);
279
280        return injector.getInstance(AnalysisService.class);
281    }
282
283    private void assertCollatesToSame(TokenStream stream1, TokenStream stream2) throws IOException {
284        assertCollation(stream1, stream2, 0);
285    }
286
287    private void assertCollation(TokenStream stream1, TokenStream stream2, int comparison) throws IOException {
288        stream1.reset();
289        stream2.reset();
290        CharTermAttribute term1 = stream1
291                .addAttribute(CharTermAttribute.class);
292        CharTermAttribute term2 = stream2
293                .addAttribute(CharTermAttribute.class);
294        assertTrue(stream1.incrementToken());
295        assertTrue(stream2.incrementToken());
296        assertEquals(term1.toString(), term2.toString());
297        assertFalse(stream1.incrementToken());
298        assertFalse(stream2.incrementToken());
299        stream1.end();
300        stream2.end();
301        stream1.close();
302        stream2.close();
303    }
