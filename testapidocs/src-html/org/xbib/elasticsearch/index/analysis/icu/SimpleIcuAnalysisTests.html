001package org.xbib.elasticsearch.index.analysis.icu;
002
003import org.apache.lucene.analysis.Analyzer;
004import org.elasticsearch.common.inject.Injector;
005import org.elasticsearch.common.inject.ModulesBuilder;
006import org.elasticsearch.common.settings.SettingsModule;
007import org.elasticsearch.env.Environment;
008import org.elasticsearch.env.EnvironmentModule;
009import org.elasticsearch.index.Index;
010import org.elasticsearch.index.IndexNameModule;
011import org.elasticsearch.index.analysis.AnalysisModule;
012import org.elasticsearch.index.analysis.AnalysisService;
013import org.elasticsearch.index.analysis.NamedAnalyzer;
014import org.elasticsearch.index.analysis.TokenFilterFactory;
015import org.elasticsearch.index.analysis.TokenizerFactory;
016import org.elasticsearch.index.settings.IndexSettingsModule;
017import org.elasticsearch.indices.analysis.IndicesAnalysisModule;
018import org.elasticsearch.indices.analysis.IndicesAnalysisService;
019import org.junit.Test;
020
021import static org.elasticsearch.common.settings.ImmutableSettings.Builder.EMPTY_SETTINGS;
022import static org.hamcrest.MatcherAssert.assertThat;
023import static org.hamcrest.Matchers.instanceOf;
024
025public class SimpleIcuAnalysisTests {
026
027    @Test
028    public void testDefaultsIcuAnalysis() {
029        Index index = new Index("test");
030
031        Injector parentInjector = new ModulesBuilder().add(new SettingsModule(EMPTY_SETTINGS), new EnvironmentModule(new Environment(EMPTY_SETTINGS)), new IndicesAnalysisModule()).createInjector();
032        Injector injector = new ModulesBuilder().add(
033                new IndexSettingsModule(index, EMPTY_SETTINGS),
034                new IndexNameModule(index),
035                new AnalysisModule(EMPTY_SETTINGS, parentInjector.getInstance(IndicesAnalysisService.class))
036                        .addProcessor(new IcuAnalysisBinderProcessor()))
037                .createChildInjector(parentInjector);
038
039        AnalysisService analysisService = injector.getInstance(AnalysisService.class);
040
041        TokenizerFactory tokenizerFactory = analysisService.tokenizer("icu_tokenizer");
042        assertThat(tokenizerFactory, instanceOf(IcuTokenizerFactory.class));
043
044        TokenFilterFactory filterFactory = analysisService.tokenFilter("icu_normalizer");
045        assertThat(filterFactory, instanceOf(IcuNormalizerTokenFilterFactory.class));
046
047        filterFactory = analysisService.tokenFilter("icu_folding");
048        assertThat(filterFactory, instanceOf(IcuFoldingTokenFilterFactory.class));
049
050        filterFactory = analysisService.tokenFilter("icu_transform");
051        assertThat(filterFactory, instanceOf(IcuTransformTokenFilterFactory.class));
052
053        Analyzer analyzer = analysisService.analyzer("icu_collation");
054        assertThat(analyzer, instanceOf(NamedAnalyzer.class));
055
056    }
057}
