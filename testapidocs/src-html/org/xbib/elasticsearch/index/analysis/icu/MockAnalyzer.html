001package org.xbib.elasticsearch.index.analysis.icu;
002
003import org.apache.lucene.analysis.Analyzer;
004import org.apache.lucene.analysis.TokenFilter;
005import org.apache.lucene.util.automaton.CharacterRunAutomaton;
006
007import java.io.Reader;
008import java.util.HashMap;
009import java.util.Map;
010
011public final class MockAnalyzer extends Analyzer {
012    private final CharacterRunAutomaton runAutomaton;
013    private final boolean lowerCase;
014    private final CharacterRunAutomaton filter;
015    private int positionIncrementGap;
016    private Integer offsetGap;
017    private Map<String, Integer> previousMappings = new HashMap<>();
018    private boolean enableChecks = true;
019    private int maxTokenLength = MockTokenizer.DEFAULT_MAX_TOKEN_LENGTH;
020
021    public MockAnalyzer(CharacterRunAutomaton runAutomaton, boolean lowerCase, CharacterRunAutomaton filter) {
022        super(PER_FIELD_REUSE_STRATEGY);
023        this.runAutomaton = runAutomaton;
024        this.lowerCase = lowerCase;
025        this.filter = filter;
026    }
027
028    public MockAnalyzer(CharacterRunAutomaton runAutomaton, boolean lowerCase) {
029        this(runAutomaton, lowerCase, MockTokenFilter.EMPTY_STOPSET);
030    }
031
032    @Override
033    public TokenStreamComponents createComponents(String fieldName, Reader reader) {
034        MockTokenizer tokenizer = new MockTokenizer(reader, runAutomaton, lowerCase, maxTokenLength);
035        tokenizer.setEnableChecks(enableChecks);
036        MockTokenFilter filt = new MockTokenFilter(tokenizer, filter);
037        return new TokenStreamComponents(tokenizer, maybePayload(filt, fieldName));
038    }
039
040    private synchronized TokenFilter maybePayload(TokenFilter stream, String fieldName) {
041        Integer val = previousMappings.get(fieldName);
042        if (val == null) {
043            val = -1; // no payloads
044            previousMappings.put(fieldName, val); // save it so we are consistent for this field
045        }
046        return stream;
047    }
048
049    public void setPositionIncrementGap(int positionIncrementGap) {
050        this.positionIncrementGap = positionIncrementGap;
051    }
052
053    @Override
054    public int getPositionIncrementGap(String fieldName) {
055        return positionIncrementGap;
056    }
057
058    /**
059     * Set a new offset gap which will then be added to the offset when several fields with the same name are indexed
060     *
061     * @param offsetGap The offset gap that should be used.
062     */
063    public void setOffsetGap(int offsetGap) {
064        this.offsetGap = offsetGap;
065    }
066
067    /**
068     * Get the offset gap between tokens in fields if several fields with the same name were added.
069     *
070     * @param fieldName Currently not used, the same offset gap is returned for each field.
071     */
072    @Override
073    public int getOffsetGap(String fieldName) {
074        return offsetGap == null ? super.getOffsetGap(fieldName) : offsetGap;
075    }
076
077    /**
078     * Toggle consumer workflow checking: if your test consumes tokenstreams normally you
079     * should leave this enabled.
080     */
081    public void setEnableChecks(boolean enableChecks) {
082        this.enableChecks = enableChecks;
083    }
084
085    /**
086     * Toggle maxTokenLength for MockTokenizer
087     */
088    public void setMaxTokenLength(int length) {
089        this.maxTokenLength = length;
090    }
091}
