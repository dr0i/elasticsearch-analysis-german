032    /*
033    * Turkish has some funny casing.
034    * This test shows how you can solve this kind of thing easily with collation.
035    * Instead of using LowerCaseFilter, use a turkish collator with primary strength.
036    * Then things will sort and match correctly.
037    */
038    @Test
039    public void testBasicUsage() throws Exception {
040        Index index = new Index("test");
041        Settings settings = ImmutableSettings.settingsBuilder()
042                .put("index.analysis.analyzer.myAnalyzer.type", "icu_collation")
043                .put("index.analysis.analyzer.myAnalyzer.language", "tr")
044                .put("index.analysis.analyzer.myAnalyzer.strength", "primary")
045                .put("index.analysis.analyzer.myAnalyzer.decomposition", "canonical")
046                .build();
047        AnalysisService analysisService = createAnalysisService(index, settings);
048        Analyzer analyzer = analysisService.analyzer("myAnalyzer").analyzer();
049        TokenStream tsUpper = analyzer.tokenStream(null, "I WİLL USE TURKİSH CASING");
050        BytesRef b1 = bytesFromTokenStream(tsUpper);
051        TokenStream tsLower = analyzer.tokenStream(null, "ı will use turkish casıng");
052        BytesRef b2 = bytesFromTokenStream(tsLower);
053        assertTrue(compare(b1.bytes, b2.bytes) == 0);
054    }

170    * Setting caseLevel=true to create an additional case level between
171    * secondary and tertiary
172    */
173    @Test
174    public void testIgnoreAccentsButNotCase() throws IOException {
175        Index index = new Index("test");
176        Settings settings = ImmutableSettings.settingsBuilder()
177                .put("index.analysis.analyzer.myAnalyzer.type", "icu_collation")
178                .put("index.analysis.analyzer.myAnalyzer.language", "en")
179                .put("index.analysis.analyzer.myAnalyzer.strength", "primary")
180                .put("index.analysis.analyzer.myAnalyzer.caseLevel", "true")
181                .build();
182        AnalysisService analysisService = createAnalysisService(index, settings);
183        Analyzer analyzer = analysisService.analyzer("myAnalyzer").analyzer();
184
185        String withAccents = "résumé";
186        String withoutAccents = "resume";
187        String withAccentsUpperCase = "Résumé";
188        String withoutAccentsUpperCase = "Resume";
189
190        TokenStream tsWithAccents = analyzer.tokenStream(null, withAccents);
191        BytesRef b1 = bytesFromTokenStream(tsWithAccents);
192        TokenStream tsWithoutAccents = analyzer.tokenStream(null, withoutAccents);
193        BytesRef b2 = bytesFromTokenStream(tsWithoutAccents);
194        assertTrue(compare(b1.bytes, b2.bytes) == 0);
195
196        TokenStream tsWithAccentsUpperCase = analyzer.tokenStream(null, withAccentsUpperCase);
197        BytesRef b3 = bytesFromTokenStream(tsWithAccentsUpperCase);
198        TokenStream tsWithoutAccentsUpperCase = analyzer.tokenStream(null, withoutAccentsUpperCase);
199        BytesRef b4 = bytesFromTokenStream(tsWithoutAccentsUpperCase);
200        assertTrue(compare(b3.bytes, b4.bytes) == 0);
201
202        // now assert that case still matters: resume < Resume
203        TokenStream tsLower = analyzer.tokenStream(null, withoutAccents);
204        BytesRef b5 = bytesFromTokenStream(tsLower);
205        TokenStream tsUpper = analyzer.tokenStream(null, withoutAccentsUpperCase);
206        BytesRef b6 = bytesFromTokenStream(tsUpper);
207        assertTrue(compare(b5.bytes, b6.bytes) < 0);
208    }

234    /*
235    * For german, you might want oe to sort and match with o umlaut.
236    * This is not the default, but you can make a customized ruleset to do this.
237    *
238    * The default is DIN 5007-1, this shows how to tailor a collator to get DIN 5007-2 behavior.
239    *  http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=4423383
240    */
241    @Test
242    public void testCustomRules() throws Exception {
243        RuleBasedCollator baseCollator = (RuleBasedCollator) Collator.getInstance(new ULocale("de_DE"));
244        String DIN5007_2_tailorings =
245                "& ae , a\u0308 & AE , A\u0308" +
246                        "& oe , o\u0308 & OE , O\u0308" +
247                        "& ue , u\u0308 & UE , u\u0308";
248
249        RuleBasedCollator tailoredCollator = new RuleBasedCollator(baseCollator.getRules() + DIN5007_2_tailorings);
250        String tailoredRules = tailoredCollator.getRules();
251
252        Index index = new Index("test");
253        Settings settings = ImmutableSettings.settingsBuilder()
254                .put("index.analysis.analyzer.myAnalyzer.type", "icu_collation")
255                .put("index.analysis.analyzer.myAnalyzer.rules", tailoredRules)
256                .put("index.analysis.analyzer.myAnalyzer.strength", "primary")
257                .build();
258        AnalysisService analysisService = createAnalysisService(index, settings);
259        Analyzer analyzer = analysisService.analyzer("myAnalyzer").analyzer();
260        String germanUmlaut = "Töne";
261        String germanOE = "Toene";
262        TokenStream tsUmlaut = analyzer.tokenStream(null, germanUmlaut);
263        BytesRef b1 = bytesFromTokenStream(tsUmlaut);
264        TokenStream tsOE = analyzer.tokenStream(null, germanOE);
265        BytesRef b2 = bytesFromTokenStream(tsOE);
266        assertTrue(compare(b1.bytes, b2.bytes) == 0);
267    }

269    private AnalysisService createAnalysisService(Index index, Settings settings) {
270        Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings), new EnvironmentModule(new Environment(settings)), new IndicesAnalysisModule()).createInjector();
271        Injector injector = new ModulesBuilder().add(
272                new IndexSettingsModule(index, settings),
273                new IndexNameModule(index),
274                new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class))
275                        .addProcessor(new IcuAnalysisBinderProcessor()))
276                .createChildInjector(parentInjector);
277
278        return injector.getInstance(AnalysisService.class);
279    }

281    private BytesRef bytesFromTokenStream(TokenStream stream) throws IOException {
282        TermToBytesRefAttribute termAttr = stream.getAttribute(TermToBytesRefAttribute.class);
283        BytesRef bytesRef = termAttr.getBytesRef();
284        stream.reset();
285        while (stream.incrementToken()) {
286            termAttr.fillBytesRef();
287        }
288        stream.close();
289        BytesRef copy = new BytesRef();
290        copy.copyBytes(bytesRef);
291        return copy;
292    }

294    private int compare(byte[] left, byte[] right) {
295        for (int i = 0, j = 0; i < left.length && j < right.length; i++, j++) {
296            int a = (left[i] & 0xff);
297            int b = (right[j] & 0xff);
298            if (a != b) {
299                return a - b;
300            }
301        }
302        return left.length - right.length;
303    }
