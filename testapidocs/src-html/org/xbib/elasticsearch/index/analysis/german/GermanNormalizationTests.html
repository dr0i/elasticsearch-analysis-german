001package org.xbib.elasticsearch.index.analysis.german;
002
003import org.apache.lucene.analysis.TokenStream;
004import org.apache.lucene.analysis.Tokenizer;
005import org.apache.lucene.analysis.standard.StandardTokenizer;
006import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
007import org.apache.lucene.util.Version;
008import org.elasticsearch.common.inject.Injector;
009import org.elasticsearch.common.inject.ModulesBuilder;
010import org.elasticsearch.common.settings.ImmutableSettings;
011import org.elasticsearch.common.settings.Settings;
012import org.elasticsearch.common.settings.SettingsModule;
013import org.elasticsearch.env.Environment;
014import org.elasticsearch.env.EnvironmentModule;
015import org.elasticsearch.index.Index;
016import org.elasticsearch.index.IndexNameModule;
017import org.elasticsearch.index.analysis.AnalysisModule;
018import org.elasticsearch.index.analysis.AnalysisService;
019import org.elasticsearch.index.analysis.TokenFilterFactory;
020import org.elasticsearch.index.settings.IndexSettingsModule;
021import org.elasticsearch.indices.analysis.IndicesAnalysisModule;
022import org.elasticsearch.indices.analysis.IndicesAnalysisService;
023import org.junit.Assert;
024import org.junit.Test;
025import org.xbib.elasticsearch.plugin.analysis.german.AnalysisGermanPlugin;
026
027import java.io.IOException;
028import java.io.StringReader;
029
030public class GermanNormalizationTests extends Assert {
031
032    @Test
033    public void test() throws IOException {
034        AnalysisService analysisService = createAnalysisService();
035        TokenFilterFactory tokenFilter = analysisService.tokenFilter("umlaut");
036
037        String source = "Ein schöner Tag in Köln im Café an der Straßenecke";
038
039        String[] expected = {
040            "Ein",
041            "schoner",
042            "Tag",
043            "in",
044            "Koln",
045            "im",
046            "Café",
047            "an",
048            "der",
049            "Strassenecke"
050        };
051
052        Tokenizer tokenizer = new StandardTokenizer(Version.LUCENE_4_9, new StringReader(source));
053
054        assertSimpleTSOutput(tokenFilter.create(tokenizer), expected);
055
056    }
057
058    private AnalysisService createAnalysisService() {
059        Settings settings = ImmutableSettings.settingsBuilder().loadFromClasspath("org/xbib/elasticsearch/index/analysis/german_normalization_analysis.json").build();
060
061        Index index = new Index("test");
062
063        Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings),
064                new EnvironmentModule(new Environment(settings)),
065                new IndicesAnalysisModule())
066                .createInjector();
067
068        AnalysisModule analysisModule = new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class));
069        new AnalysisGermanPlugin().onModule(analysisModule);
070
071        Injector injector = new ModulesBuilder().add(
072                new IndexSettingsModule(index, settings),
073                new IndexNameModule(index),
074                analysisModule)
075                .createChildInjector(parentInjector);
076
077        return injector.getInstance(AnalysisService.class);
078    }
079
080    private void assertSimpleTSOutput(TokenStream stream, String[] expected) throws IOException {
081        stream.reset();
082        CharTermAttribute termAttr = stream.getAttribute(CharTermAttribute.class);
083        assertNotNull(termAttr);
084        int i = 0;
085        while (stream.incrementToken()) {
086            assertTrue(i < expected.length);
087            assertEquals(expected[i], termAttr.toString());
088            i++;
089        }
090        assertEquals(i, expected.length);
091        stream.close();
092    }
093}
