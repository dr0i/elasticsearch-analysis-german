036public class BaseformTokenFilterTests extends Assert {
037
038    @Test
039    public void testOne() throws IOException {
040        AnalysisService analysisService = createAnalysisService();
041        TokenFilterFactory tokenFilter = analysisService.tokenFilter("baseform");
042        assertThat(tokenFilter, instanceOf(BaseformTokenFilterFactory.class));
043
044        String source = "Die Jahresfeier der Rechtsanwaltskanzleien auf dem Donaudampfschiff hat viel ?kosteuer gekostet";
045
046        String[] expected = {
047            "Die",
048            "Die",
049            "Jahresfeier",
050            "Jahresfeier",
051            "der",
052            "der",
053            "Rechtsanwaltskanzleien",
054            "Rechtsanwaltskanzlei",
055            "auf",
056            "auf",
057            "dem",
058            "der",
059            "Donaudampfschiff",
060            "Donaudampfschiff",
061            "hat",
062            "haben",
063            "viel",
064            "viel",
065            "?kosteuer",
066            "?kosteuer",
067            "gekostet",
068            "kosten"
069        };
070
071        Tokenizer tokenizer = new StandardTokenizer(Version.LUCENE_46, new StringReader(source));
072        assertSimpleTSOutput(tokenFilter.create(tokenizer), expected);
073
074    }
075
076    @Test
077    public void testTwo() throws IOException {
078        AnalysisService analysisService = createAnalysisService();
079        TokenFilterFactory tokenFilter = analysisService.tokenFilter("baseform");
080        assertThat(tokenFilter, instanceOf(BaseformTokenFilterFactory.class));
081
082        String source = "Das sind Autos, die Nudeln transportieren.";
083
084        String[] expected = {
085                "Das",
086                "Das",
087                "sind",
088                "sind",
089                "Autos",
090                "Auto",
091                "die",
092                "der",
093                "Nudeln",
094                "Nudel",
095                "transportieren",
096                "transportieren"
097        };
098
099        Tokenizer tokenizer = new StandardTokenizer(Version.LUCENE_46, new StringReader(source));
100        assertSimpleTSOutput(tokenFilter.create(tokenizer), expected);
101    }
102
103    private AnalysisService createAnalysisService() {
104        Settings settings = ImmutableSettings.settingsBuilder()
105                .loadFromClasspath("org/xbib/elasticsearch/index/analysis/baseform_de.json").build();
106
107        Index index = new Index("test");
108
109        Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings),
110                new EnvironmentModule(new Environment(settings)),
111                new IndicesAnalysisModule())
112                .createInjector();
113
114        AnalysisModule analysisModule = new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class));
115        new AnalysisGermanPlugin().onModule(analysisModule);
116
117        Injector injector = new ModulesBuilder().add(
118                new IndexSettingsModule(index, settings),
119                new IndexNameModule(index),
120                analysisModule)
121                .createChildInjector(parentInjector);
122
123        return injector.getInstance(AnalysisService.class);
124    }
125
126    private void assertSimpleTSOutput(TokenStream stream, String[] expected) throws IOException {
127        stream.reset();
128        CharTermAttribute termAttr = stream.getAttribute(CharTermAttribute.class);
129        assertNotNull(termAttr);
130        int i = 0;
131        while (stream.incrementToken()) {
132            assertTrue(i < expected.length);
133            assertEquals(expected[i++], termAttr.toString(), "expected different term at index " + i);
134        }
135        assertEquals(i, expected.length, "not all tokens produced");
136        stream.close();
137    }
138}
