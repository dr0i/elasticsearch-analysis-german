001package org.xbib.elasticsearch.index.analysis.combo;
002
003import org.junit.Test;
004
005import java.io.Reader;
006import java.io.StringReader;
007
008import static org.hamcrest.MatcherAssert.assertThat;
009import static org.hamcrest.Matchers.equalTo;
010import static org.hamcrest.Matchers.is;
011import static org.hamcrest.Matchers.not;
012
013public class StringReaderClonerTests {
014
015    @Test
016    public void test() throws Exception {
017        String content = "test\n";
018        StringReader reader = new StringReader(content);
019        StringReaderCloner cloner = new StringReaderCloner();
020        cloner.init(reader);
021
022        Reader clone1 = cloner.giveAClone();
023        // The original Reader can be returned with this implementation.
024        // Check it actually is.
025        assertThat("returns original reader", clone1, is((Reader)reader));
026        assertThat("same content", ReaderContent.readWhole(clone1), equalTo(content));
027        assertThat("empty after reading", clone1.read(), equalTo(-1));
028
029        Reader clone2 = cloner.giveAClone();
030        assertThat("do not return the previous clone", clone2, not(is(clone1)));
031        assertThat("same content", ReaderContent.readWhole(clone2), equalTo(content));
032        assertThat("empty after reading", clone2.read(), equalTo(-1));
033    }
034
035}
