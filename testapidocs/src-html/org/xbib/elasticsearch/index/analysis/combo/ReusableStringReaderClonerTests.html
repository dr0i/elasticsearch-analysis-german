001package org.xbib.elasticsearch.index.analysis.combo;
002
003import org.testng.annotations.Test;
004
005import java.io.Reader;
006
007import static org.hamcrest.MatcherAssert.assertThat;
008import static org.hamcrest.Matchers.equalTo;
009import static org.hamcrest.Matchers.is;
010import static org.hamcrest.Matchers.not;
011
012@Test
013public class ReusableStringReaderClonerTests {
014
015    @Test
016    public void test() throws Exception {
017        String content = "test\n";
018        ReusableStringReader reader = new ReusableStringReader();
019        reader.setValue(content);
020        ReusableStringReaderCloner cloner = new ReusableStringReaderCloner();
021        cloner.init(reader);
022
023        Reader clone1 = cloner.giveAClone();
024        // The original Reader can be returned with this implementation.
025        // Check it actually is.
026        assertThat("returns original reader", clone1, is((Reader) reader));
027        assertThat("same content", ReaderContent.readWhole(clone1), equalTo(content));
028        assertThat("empty after reading", clone1.read(), equalTo(-1));
029
030        Reader clone2 = cloner.giveAClone();
031        assertThat("do not return the previous clone", clone2, not(is(clone1)));
032        assertThat("same content", ReaderContent.readWhole(clone2), equalTo(content));
033        assertThat("empty after reading", clone2.read(), equalTo(-1));
034    }
035
036    @Test
037    public void testReuseAfterUse() throws Exception {
038        String content = "test\n";
039        ReusableStringReader reader = new ReusableStringReader();
040        reader.setValue(content);
041        ReusableStringReaderCloner cloner = new ReusableStringReaderCloner();
042        cloner.init(reader);
043
044        Reader clone1 = cloner.giveAClone();
045        // The original Reader can be returned with this implementation.
046        // Check it actually is.
047        assertThat("returns original reader", clone1, is((Reader) reader));
048        assertThat("same content", ReaderContent.readWhole(clone1), equalTo(content));
049        assertThat("empty after reading", clone1.read(), equalTo(-1));
050
051        // Now change the content!
052        String otherContent = content + " CHANGED";
053        assertThat("contents are not equal", otherContent, not(equalTo(content)));
054        reader.setValue(otherContent);
055
056        Reader clone2 = cloner.giveAClone();
057        assertThat("do not return the previous clone", clone2, not(is(clone1)));
058        assertThat("same content as before", ReaderContent.readWhole(clone2), equalTo(content));
059        assertThat("empty after reading", clone2.read(), equalTo(-1));
060    }
061
062    @Test
063    public void testReuseBeforeUse() throws Exception {
064        String content = "test\n";
065        ReusableStringReader reader = new ReusableStringReader();
066        reader.setValue(content);
067        ReusableStringReaderCloner cloner = new ReusableStringReaderCloner();
068        cloner.init(reader);
069
070        // Now change the content!
071        String otherContent = content + " CHANGED";
072        assertThat("contents are not equal", otherContent, not(equalTo(content)));
073        reader.setValue(otherContent);
074
075        Reader clone1 = cloner.giveAClone();
076        // The original Reader can be returned with this implementation.
077        // Check it actually is.
078        assertThat("returns original reader", clone1, is((Reader) reader));
079        assertThat("has the new content", ReaderContent.readWhole(clone1), equalTo(otherContent)); // alas, but desirable performance-wise
080        assertThat("empty after reading", clone1.read(), equalTo(-1));
081
082        Reader clone2 = cloner.giveAClone();
083        assertThat("do not return the previous clone", clone2, not(is(clone1)));
084        String actualContent = ReaderContent.readWhole(clone2);
085        assertThat("same content as before", actualContent, equalTo(content));
086        assertThat("empty after reading", clone2.read(), equalTo(-1));
087    }

089}
