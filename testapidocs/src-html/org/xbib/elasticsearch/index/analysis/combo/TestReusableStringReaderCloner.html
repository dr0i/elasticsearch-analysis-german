002package org.xbib.elasticsearch.index.analysis.combo;
003
004import org.testng.annotations.Test;
005import org.xbib.elasticsearch.index.analysis.BaseTokenStreamTest;
006
007import java.io.IOException;
008import java.io.Reader;
009
010/**
011 * Testcase for {@link ReusableStringReaderCloner}
012 */
013@Test
014public class TestReusableStringReaderCloner extends BaseTokenStreamTest {
015
016    @Test
017    public void testCloningReusableStringReader() throws IOException, ClassNotFoundException, NoSuchMethodException, IllegalAccessException, InstantiationException {
018        // This test cannot be located inside TestReaderCloneFactory
019        // because of the ReusableStringReader class being package private
020        // (and it's a real pain to use Java reflection to gain access to
021        //  a package private constructor)
022        Reader clone;
023        ReusableStringReader reader = new ReusableStringReader();
024        reader.setValue("test string");
025        ReaderCloneFactory.ReaderCloner<Reader> cloner = ReaderCloneFactory.getCloner(reader);
026        assertNotNull(cloner);
027        assertEquals(cloner.getClass().getName(), ReusableStringReaderCloner.class.getName());
028        clone = cloner.giveAClone();
029        ReaderContent.assertReaderContent(clone, "test string");
030        clone = cloner.giveAClone();
031        ReaderContent.assertReaderContent(clone, "test string");
032
033        // Test reusability
034        ReaderCloneFactory.ReaderCloner<ReusableStringReader> forClassClonerStrict = ReaderCloneFactory.getClonerStrict(ReusableStringReader.class);
035        assertNotNull(forClassClonerStrict);
036        assertEquals(forClassClonerStrict.getClass().getName(), ReusableStringReaderCloner.class.getName());
037        reader.setValue("another test string");
038        forClassClonerStrict.init(reader);
039        clone = forClassClonerStrict.giveAClone();
040        ReaderContent.assertReaderContent(clone, "another test string");
041        clone = forClassClonerStrict.giveAClone();
042        ReaderContent.assertReaderContent(clone, "another test string");
043        reader.setValue("test string");
044        forClassClonerStrict.init(reader);
045        clone = forClassClonerStrict.giveAClone();
046        ReaderContent.assertReaderContent(clone, "test string");
047        clone = forClassClonerStrict.giveAClone();
048        ReaderContent.assertReaderContent(clone, "test string");
049
050        ReaderCloneFactory.ReaderCloner<Reader> forClassCloner = ReaderCloneFactory.getCloner(ReusableStringReader.class);
051        assertNotNull(forClassCloner);
052        assertEquals(forClassCloner.getClass().getName(), ReusableStringReaderCloner.class.getName());
053        reader.setValue("another test string");
054        forClassCloner.init(reader);
055        clone = forClassCloner.giveAClone();
056        ReaderContent.assertReaderContent(clone, "another test string");
057        clone = forClassCloner.giveAClone();
058        ReaderContent.assertReaderContent(clone, "another test string");
059        reader.setValue("test string");
060        forClassCloner.init(reader);
061        clone = forClassCloner.giveAClone();
062        ReaderContent.assertReaderContent(clone, "test string");
063        clone = forClassCloner.giveAClone();
064        ReaderContent.assertReaderContent(clone, "test string");
065    }
066
067}
