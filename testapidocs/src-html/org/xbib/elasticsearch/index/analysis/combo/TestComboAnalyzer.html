139    @Test
140    public void testReuseSequentialMultithreading() throws IOException, InterruptedException {
141        // Create the analyzer
142        final ComboAnalyzer cb = new ComboAnalyzer(TEST_VERSION_CURRENT, new WhitespaceAnalyzer(TEST_VERSION_CURRENT));
143        final NamedAnalyzer namedAnalyzer = new NamedAnalyzer("name", AnalyzerScope.INDEX, cb);
144        // Use N threads, each running M times
145        Thread[] threads = new Thread[4];
146        final int runs = 4;
147        // The lock ensures only one thread is running at a given time
148        final Lock lock = new ReentrantLock();
149        // This integer ensures each thread runs with a different input
150        // Inputs must not be exchanged from one thread to another during object reuse
151        final AtomicInteger sequence = new AtomicInteger(0);
152        final AtomicBoolean abort = new AtomicBoolean(false);
153        // The barrier ensures that each thread gets a chance to execute, for each run
154        // We must use extra care so that all threads can exit as soon as one fails
155        final CyclicBarrier latch = new CyclicBarrier(threads.length);
156        // Code executed on each thread
157        Runnable code = new Runnable() {
158            @Override
159            public void run() {
160                // Run multiple times before quitting
161                for (int run = 0 ; run < runs ; ++run) {
162                    try {
163                        // Serialize runs
164                        lock.lock();
165                        // Get unique sequence number
166                        int i = sequence.getAndIncrement();
167                        // Check the analysis went well, including the unique sequence number
168                        assertTokenStreamContents(namedAnalyzer.tokenStream("field", new StringReader("just a little test " + i)),
169                                new String[]{"just", "a", "little", "test", Integer.toString(i)},
170                                new int[]{0, 5, 7, 14, 19},
171                                new int[]{4, 6, 13, 18, 19 + ("" + i).length()},
172                                new int[]{1, 1, 1, 1, 1});
173                    } catch (Exception e) {
174                        e.printStackTrace();
175                        // Make other fail,
176                        abort.set(true); // if they will soon be waiting,
177                        latch.reset(); // and if they are already waiting
178                        // Now we can fail!
179                        assertNull(e);
180                    } finally {
181                        lock.unlock();
182                    }
183                    // Wait for other threads, so calls are well interleaved between threads
184                    try {
185                        if (abort.get()) return;
186                        latch.await();
187                    } catch (Exception e) {
188                        e.printStackTrace();
189                        // Make other fail,
190                        abort.set(true); // if they will soon be waiting,
191                        latch.reset(); // and if they are already waiting
192                        // Now we can fail!
193                        assertNull(e);
194                    }
195                }
196            }
197        };
198        // Create the threads
199        for (int i = 0 ; i < threads.length ; i++)
200            threads[i] = new Thread(code);
201        // Start the threads
202        for (int i = 0 ; i < threads.length ; i++)
203            threads[i].start();
204        // Wait for completion
205        for (int i = 0 ; i < threads.length ; i++)
206            threads[i].join();
207        // Ensure all desired runs have been performed
208        assertThat(abort.get(), equalTo(false));
209        assertThat(sequence.get(), equalTo(runs * threads.length));
210    }

212    @Test
213    public void testReuseConcurrentMultithreading() throws IOException, InterruptedException {
214        // Create the analyzer
215        final ComboAnalyzer cb = new ComboAnalyzer(TEST_VERSION_CURRENT, new WhitespaceAnalyzer(TEST_VERSION_CURRENT));
216        final NamedAnalyzer namedAnalyzer = new NamedAnalyzer("name", AnalyzerScope.INDEX, cb);
217        // Use N threads, each running M times
218        Thread[] threads = new Thread[4];
219        final int runs = 4000; // leave time for threads to run concurrently
220        // This integer ensures each thread runs with a different input
221        // Inputs must not be exchanged from one thread to another during object reuse
222        final AtomicInteger sequence = new AtomicInteger(0);
223        // The barrier ensures that each thread gets a chance to execute, for each run
224        final CyclicBarrier latch = new CyclicBarrier(threads.length);
225        // Code executed on each thread
226        Runnable code = new Runnable() {
227            @Override
228            public void run() {
229                try {
230                    latch.await();
231                    // Run multiple times before quitting
232                    for (int run = 0 ; run < runs ; ++run) {
233                        // Get unique sequence number
234                        int i = sequence.getAndIncrement();
235                        // Check the analysis went well, including the unique sequence number
236                        assertTokenStreamContents(namedAnalyzer.tokenStream("field", new StringReader("just a little test " + i)),
237                                new String[]{"just", "a", "little", "test", Integer.toString(i)},
238                                new int[]{0, 5, 7, 14, 19},
239                                new int[]{4, 6, 13, 18, 19 + ("" + i).length()},
240                                new int[]{1, 1, 1, 1, 1});
241                    }
242                } catch (Exception e) {
243                    // Fail!
244                    assertNull(e);
245                }
246            }
247        };
248        // Create the threads
249        for (int i = 0 ; i < threads.length ; i++)
250            threads[i] = new Thread(code);
251        // Start the threads
252        for (Thread thread : threads) {
253            thread.start();
254        }
255        // Wait for completion
256        for (Thread thread : threads) {
257            thread.join();
258        }
259        // Ensure all desired runs have been performed
260        assertThat(sequence.get(), equalTo(runs * threads.length));
261    }
