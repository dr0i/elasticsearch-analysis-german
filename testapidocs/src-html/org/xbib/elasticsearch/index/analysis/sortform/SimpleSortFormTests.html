001package org.xbib.elasticsearch.index.analysis.sortform;
002
003import org.elasticsearch.common.inject.Injector;
004import org.elasticsearch.common.inject.ModulesBuilder;
005import org.elasticsearch.common.settings.ImmutableSettings;
006import org.elasticsearch.common.settings.Settings;
007import org.elasticsearch.common.settings.SettingsModule;
008import org.elasticsearch.env.Environment;
009import org.elasticsearch.env.EnvironmentModule;
010import org.elasticsearch.index.Index;
011import org.elasticsearch.index.IndexNameModule;
012import org.elasticsearch.index.analysis.AnalysisModule;
013import org.elasticsearch.index.analysis.AnalysisService;
014import org.elasticsearch.index.analysis.NamedAnalyzer;
015import org.elasticsearch.index.settings.IndexSettingsModule;
016import org.elasticsearch.indices.analysis.IndicesAnalysisModule;
017import org.elasticsearch.indices.analysis.IndicesAnalysisService;
018import org.junit.Test;
019import org.xbib.elasticsearch.index.analysis.BaseTokenStreamTest;
020
021public class SimpleSortFormTests  extends BaseTokenStreamTest {
022
023    @Test
024    public void testBasicUsage() throws Exception {
025        Settings settings = ImmutableSettings.settingsBuilder()
026                .put("index.analysis.analyzer.myanalyzer.type", "sortform")
027                //.put("index.analysis.analyzer.myanalyzer.pattern", "<<.*?>>\\s*")
028                .build();
029        AnalysisService analysisService = createAnalysisService(settings);
030        NamedAnalyzer myanalyzer = analysisService.analyzer("myanalyzer");
031        assertAnalyzesTo(myanalyzer, "<<Der>> Titel des Buches", new String[]{"Titel des Buches"});
032    }
033
034    @Test
035    public void testMABUsage() throws Exception {
036        Settings settings = ImmutableSettings.settingsBuilder()
037                .put("index.analysis.analyzer.myanalyzer.type", "sortform")
038                .build();
039        AnalysisService analysisService = createAnalysisService(settings);
040        NamedAnalyzer myanalyzer = analysisService.analyzer("myanalyzer");
041        assertAnalyzesTo(myanalyzer, "\u0098Der\u009c Titel des Buches", new String[]{"Titel des Buches"});
042    }
043
044    @Test
045    public void testDateUsage() throws Exception {
046        Settings settings = ImmutableSettings.settingsBuilder()
047                .put("index.analysis.analyzer.myanalyzer.type", "sortform")
048                .put("index.analysis.analyzer.myanalyzer.pattern", "[^0-9]")
049                .build();
050        AnalysisService analysisService = createAnalysisService(settings);
051        NamedAnalyzer myanalyzer = analysisService.analyzer("myanalyzer");
052        assertAnalyzesTo(myanalyzer, "ersch. ca. 1988", new String[]{"1988"});
053    }
054
055    private AnalysisService createAnalysisService(Settings settings) {
056        Index index = new Index("test");
057        Injector parentInjector = new ModulesBuilder()
058                .add(new SettingsModule(settings),
059                        new EnvironmentModule(new Environment(settings)),
060                        new IndicesAnalysisModule()).createInjector();
061        AnalysisModule analysisModule = new AnalysisModule(settings,
062                parentInjector.getInstance(IndicesAnalysisService.class));
063        analysisModule.addAnalyzer("sortform", SortformAnalyzerProvider.class);
064        Injector injector = new ModulesBuilder().add(
065                new IndexSettingsModule(index, settings),
066                new IndexNameModule(index),
067                analysisModule)
068           .createChildInjector(parentInjector);
069        return injector.getInstance(AnalysisService.class);
070    }
071}
